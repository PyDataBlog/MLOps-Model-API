Public Class Form1

    End Sub

    Private Sub Busa_Click(sender As Object, e As EventArgs) Handles Busa.Click

        TBusa.Text = 40
        TBusa.Enabled = False
        Busa.Enabled = False

        If Busa.Enabled = False And Balemania.Enabled = False And Brusia.Enabled = False And Baustralia.Enabled = False Then
            Bcampeon.Enabled = True
        End If

    End Sub
