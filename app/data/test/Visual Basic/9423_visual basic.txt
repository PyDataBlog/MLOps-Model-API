Public Class formInicio
    Private Sub AltaToolStripMenuItem_Click(sender As Object, e As EventArgs) Handles AltaToolStripMenuItem.Click
        If Me.MdiChildren.Length < 1 Then
            Dim formAlta As New formAltaAlumno
            formAlta.MdiParent = Me
            formAlta.Show()
        End If
    End Sub

    Private Sub NotasToolStripMenuItem_Click(sender As Object, e As EventArgs) Handles NotasToolStripMenuItem.Click
        If Me.MdiChildren.Length < 1 Then
            Dim formNotasAlta As New formNotasAlta
            formNotasAlta.MdiParent = Me
            formNotasAlta.Show()
        End If
    End Sub

    Private Sub AlumnoToolStripMenuItem_Click(sender As Object, e As EventArgs) Handles AlumnoToolStripMenuItem.Click
        If Me.MdiChildren.Length < 1 Then
            Dim formImprimirAlumno As New formImprimirAlumno
            formImprimirAlumno.MdiParent = Me
            formImprimirAlumno.Show()
        End If
    End Sub

    Private Sub formInicio_Load(sender As Object, e As EventArgs) Handles MyBase.Load

        dam.Text = "DAM"
        daw.Text = "DAW"

        comboImprimirCurso.Items.Add(dam)
        comboImprimirCurso.Items.Add(daw)
    End Sub

    Private Sub comboImprimirCurso_SelectedIndexChanged(sender As Object, e As EventArgs) Handles comboImprimirCurso.SelectedIndexChanged
        If comboImprimirCurso.SelectedItem.ToString.Equals("DAM") Then
            If Me.MdiChildren.Length < 1 Then
                Dim formImprimirCurso As New formImprimirCurso
                formImprimirCurso.MdiParent = Me
                formImprimirCurso.Tag = "DAM"
                formImprimirCurso.Show()
            End If
        Else
            If Me.MdiChildren.Length < 1 Then
                Dim formImprimirCurso As New formImprimirCurso
                formImprimirCurso.MdiParent = Me
                formImprimirCurso.Tag = "DAW"
                formImprimirCurso.Show()
            End If
        End If
    End Sub
End Class
