Namespace YahooManaged.Geo.GeoPlanet


    Public Class GeoPlaceData
        Implements IPlace

        Public ReadOnly Property Address As String Implements IPlace.Address
            Get
                Return Me.Name
            End Get
        End Property
        Public Property Radius As Integer Implements IPlace.Radius
            Get
                With Me.BoundingBox
                    Dim a = Math.Abs(.NorthWest.Latitude - .SouthWest.Latitude) / 180 * 42000
                    Dim b = Math.Abs(.SouthWest.Longitude - .SouthEast.Longitude) / 360 * 42000
                    Dim c As Double = (Math.Sqrt(a ^ 2 + b ^ 2) / 2) * 1000
                    Return CInt(c)
                End With
            End Get
            Set(ByVal value As Integer)
                Throw New NotSupportedException("Setting a value is not supported")
            End Set
        End Property
        Public Property Name As String = String.Empty
        Public Property Type As PlaceType = PlaceType.Undefined Implements IPlace.WOEType
        Public Property WOEID As Long = -1 Implements IPlace.WOEID
        Public Property Center As Coordinates Implements IPlace.Position
        Public Property BoundingBox As CoordinatesRectangle Implements IPlace.BoundingBox
        Public Property PostalCode As String = String.Empty Implements IPlace.ZipCode
        Public Property Country As AdminArea
        Public Property LocalAdmin As LocalAdminArea
        Public Property RegionalAdmin As RegionalAdminArea
        Public Property FederalAdmin As FederalAdminArea
        Public Property Locality1 As Locality
        Public Property Locality2 As Locality
        Public Property AreaInSquareKilometers As ValueRange
        Public Property PopulationCount As ValueRange

    End Class




End Namespace
