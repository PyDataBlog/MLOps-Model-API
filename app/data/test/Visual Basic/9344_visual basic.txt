Imports MySql.Data.MySqlClient

Namespace Dao.MySQL.Licitacao
    Public Class BuildLicitacaoItens
        Implements Dao.IModel(Of SGV.ItensLicitacao)
        Public Function BuildModel(ByVal myData As MySqlDataReader) As ItensLicitacao Implements IModel(Of ItensLicitacao).BuildModel
            Dim licitacaoItem As New ItensLicitacao()
            licitacaoItem.GetID = myData.GetInt32(0)
            licitacaoItem.GetIDAcreditacao = myData.GetInt32(1)
            licitacaoItem.GetIDSubarea = myData.GetInt32(2)
            licitacaoItem.GetIDLicitacao = myData.GetInt32(3)
            If Not myData.IsDBNull(4) Then licitacaoItem.GetIDEquipamento = myData.GetInt32(4)
            If Not myData.IsDBNull(5) Then licitacaoItem.GetIDProduto = myData.GetString(5)
            licitacaoItem.GetStatus = myData.GetBoolean(6)
            licitacaoItem.GetLote = myData.GetString(7)
            licitacaoItem.GetItem = myData.GetString(8)
            licitacaoItem.GetDescricao = myData.GetString(9)
            licitacaoItem.GetQTD = myData.GetInt32(10)
            licitacaoItem.GetUnitario = myData.GetDouble(11)
            licitacaoItem.GetQTDOrc = myData.GetInt32(12)
            licitacaoItem.GetQTDREC = myData.GetInt32(13)
            If Not myData.IsDBNull(14) Then licitacaoItem.GetIDServico = myData.GetInt32(14)
            Return licitacaoItem
        End Function
    End Class
End Namespace
