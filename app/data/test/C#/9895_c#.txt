using System;

class P09_SortArray
{
    static void SwapPositionsInArray(int[] array, int firstIndex, int secondIndex)
    {
        int tempVar = array[firstIndex];
        array[firstIndex] = array[secondIndex];
        array[secondIndex] = tempVar;
    }

    static int FindBiggestInRange(int[] array, int start, int end)
    {
        int biggest = int.MinValue;

        for (int i = start; i <= end; i++)
        {
            if (array[i] > biggest)
            {
                biggest = array[i];
            }
        }

        return biggest;
    }

    static void PrintArray(int[] array)
    {
        for (int i = 0; i <= array.GetUpperBound(0); i++)
        {
            Console.Write("{0, 4}", array[i]);
        }

        Console.WriteLine();
    }

    static int[] CreateRandomArray(int sizeOfArray)
    {
        int[] array = new int[sizeOfArray];
        Random digit = new Random();

        for (int i = 0; i < sizeOfArray; i++)
        {
            array[i] = digit.Next(-100, 100);
        }

        return array;
    }

    static void Main()
    {
        int[] array = CreateRandomArray(10);
        Console.Write("random array is:");
        PrintArray(array);
        int arrayLastIndex = array.GetUpperBound(0);

        for (int i = 0; i < arrayLastIndex - 1; i++)
        {
            int biggest = FindBiggestInRange(array, 0, arrayLastIndex - i);
            int indexOfBiggest = Array.IndexOf(array, biggest);
            SwapPositionsInArray(array, indexOfBiggest, arrayLastIndex - i);
        }

        Console.Write(" ascending sort:");
        PrintArray(array);

        for (int i = 0; i < arrayLastIndex - 1; i++)
        {
            int biggest = FindBiggestInRange(array, i, arrayLastIndex);
            int indexOfBiggest = Array.IndexOf(array, biggest);
            SwapPositionsInArray(array, indexOfBiggest, i);
        }

        Console.Write("descending sort:");
        PrintArray(array);
    }
}
