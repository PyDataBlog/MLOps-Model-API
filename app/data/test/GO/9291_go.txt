package main

import (
	"github.com/mycaosf/gui"
	"github.com/mycaosf/winapi"
)

type aboutDialog struct {
	gui.BaseDialog
}

func (p *aboutDialog) Reg() {
	p.BaseDialog.Reg()
	p.RegCmd(winapi.IDOK, p.OnOk)
}

func (p *aboutDialog) OnOk(cmd *winapi.CMD) {
	p.End(winapi.IDOK)
}

func runAboutDialog(res *gui.Resource, parent winapi.HWND) {
	var dlg aboutDialog
	dlg.Reg()
	dlg.DlgBox(res.Get(), "AboutDlg", parent)
}
