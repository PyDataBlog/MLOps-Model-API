package cmd

import (
	"github.com/spf13/cobra"

	log "github.com/Sirupsen/logrus"
)

// buildCmd represents the build command
var buildCmd = &cobra.Command{
	Use:     "build <project name(s)>",
	Aliases: []string{"b"},
	Short:   "Build index.html for these projects once.",
	Long: `Build can be used to precompile all relevant sources and
generate the index.html from the compiled sources as well
as the index template only once, without continous watching
for changes.`,

	Run: func(cmd *cobra.Command, args []string) {
		for _, pDir := range args {
			empty := isEmptyProject(pDir)
			if empty {
				log.WithFields(log.Fields{
					"project": pDir,
				}).Error(log_all_projnotexist)
				continue
			}

			if err := build(pDir); err != nil {
				continue
			}
		}
	},
}

func init() {
	RootCmd.AddCommand(buildCmd)
}
