package termuiwrapper_test

import (
	"projectanima/AnimaUtilities/termuiwrapper"
	"testing"
)

func TestProcessUI_Start(t *testing.T) {

	pUI := termuiwrapper.ProcessUI{}

	errorFound := pUI.Start()

	if errorFound != nil {
		t.Errorf("%s; %s", errorFound.Error(), "termUI error")
	}

	defer pUI.End()
}
