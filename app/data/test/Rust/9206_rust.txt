use std::fmt;

/// Specify the current location for the parser.
#[derive(Debug)]
pub struct Loc {
    /// The name of the thing being parsed.  Typically a file name.
    pub name: &'static str,
    /// The column.
    pub column: u32,
    /// The line.
    pub line: u32,
}

impl fmt::Display for Loc {
    fn fmt(&self, form: &mut fmt::Formatter) -> fmt::Result {
        write!(form, "{}:{}:{}", self.name, self.line, self.column)
    }
}
