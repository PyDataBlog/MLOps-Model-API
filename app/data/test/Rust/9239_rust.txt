#![allow(dead_code, unused_variables)]

use std::thread::sleep_ms;

mod xcb;

fn main() {
    let c = xcb::connect().unwrap();
    let iter = c.screen_iterator();
    println!("abc");
    for screen in iter {
        println!("width {}", screen.width());
        c.bar(screen);
    }
    sleep_ms(10000);
}
