*        The status of the tests.

      IMPLICIT NONE              ! No implicit typing

*     None

      INTEGER STATUS

*  Global Constants:
      INCLUDE 'SAE_PAR'          ! Standard SAE constants
      INCLUDE 'CHR_ERR'

*  Local Variables:
      CHARACTER*1 C1             ! CHARACTER values
      CHARACTER*1 C2

*.

*    Test CHR_SWAP

      C1 = 'A'
      C2 = 'B'
      CALL CHR_SWAP (C1, C2)
      IF ((C1 .EQ. 'B') .AND. (C2 .EQ. 'A')) THEN
         PRINT *, 'CHR_SWAP OK'
      ELSE
         PRINT *, 'CHR_SWAP FAILS'
         STATUS = SAI__ERROR
      ENDIF

      END
