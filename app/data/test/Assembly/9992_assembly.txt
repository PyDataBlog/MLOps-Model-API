	.model SMALL

	include pf.inc

	.CODE

func BFLUSH	; bflush() {
	push	BP
	mov	BP,SP
	push	DS

	;arg	bf:word
	bf	= (RETSIZE+1)*2

	mov	DS,[BP+bf]		; BFILE\¢ÌÌZOg

	mov	CX,DS:[b_pos]
	jcxz	short _ok

	mov	BX,DS:[b_hdl]
	lea	DX,DS:[b_buff]
	msdos	WriteHandle
	jc	short _error

	sub	DS:[b_pos],AX
	jnz	short _error

_ok:
	clr	AX
	pop	DS
	pop	BP
	ret	(1)*2

_error:
	mov	AX,EOF
	pop	DS
	pop	BP
	ret	(1)*2
endfunc		; }
