	include func.inc

	.CODE
	EXTRN DOS_GETDRIVE:CALLMODEL

func DOS_GETCWD	; dos_getcwd() {
	push	BP
	mov	BP,SP
	push	SI

	; ø
	drive	= (RETSIZE+1+DATASIZE)*2
	buf	= (RETSIZE+1)*2

	or	AL,AL
	jnz	short GO
	_call	DOS_GETDRIVE
	inc	AL

	mov	DL,AL
	add	AL,'@'
	mov	AH,':'

	_lds	BX,[BP+buf]
