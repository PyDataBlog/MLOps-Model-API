	.MODEL SMALL
	include func.inc
	EXTRN FONT_WRITE:CALLMODEL
	EXTRN FONT_READ:CALLMODEL

	.CODE

func FONT_ENTRY_KCG	; font_entry_kcg() {
	push	BP
	mov	BP,SP
	sub	SP,32
	push	SI
	push	DI

	temp = -32

	; space
	push	SS
	pop	ES
	lea	DI,[BP+temp]
	mov	CX,16/2
	xor	AX,AX
	rep	stosw

	mov	AX,32	; space
	push	AX
	_push	SS
	lea	AX,[BP+temp]
	push	AX
	call	FONT_WRITE

	; !©çÌ[v
	mov	SI,33
	mov	DI,2921h
	jmp	short GO
KLOOP:
	mov	SI,0a1h
GO:
	push	DI
	_push	SS
	lea	AX,[BP+temp]
	push	AX
	call	FONT_READ

	push	SI
	_push	SS
	lea	AX,[BP+temp]
	push	AX
	call	FONT_WRITE

	inc	DI
	cmp	DI,297fh
	jne	short SKIP
	mov	DI,2a21h
SKIP:
	inc	SI
	cmp	SI,7fh
	je	short KLOOP
	cmp	SI,0e0h
	jl	short GO

	pop	DI
	pop	SI
	mov	SP,BP
	pop	BP
	ret
endfunc		; }
