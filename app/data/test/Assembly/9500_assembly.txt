	.MODEL SMALL
	include func.inc

	.CODE

func DOS_SEEK
	push	BP
	mov	BP,SP
	fh	= (RETSIZE+4)*2
	offs	= (RETSIZE+2)*2
	mode	= (RETSIZE+1)*2

	mov	BX,[BP+fh]
	mov	DX,[BP+offs]
	mov	CX,[BP+offs+2]
	mov	AL,[BP+mode]
	mov	AH,42h
	int	21h
	jnc	short SUCCESS
	neg	AX	; cy=1
	cwd
SUCCESS:
	pop	BP
	ret	8
endfunc
