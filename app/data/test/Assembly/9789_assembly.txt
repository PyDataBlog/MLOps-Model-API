	.MODEL SMALL
	include func.inc

	.CODE

func DOS_PUTS2		; dos_puts2() {
	mov	BX,SP
	mov	CX,SI	; save SI
	_push	DS
	_lds	SI,SS:[BX+RETSIZE*2]
	lodsb
	or	AL,AL
	je	short EXIT
	mov	AH,2
CLOOP:	cmp	AL,0ah		; '\n'
	jne	short NOT_LF
	mov	DL,0dh
	int	21h
	mov	AL,0ah
NOT_LF:
	mov	DL,AL
	int	21h
	lodsb
	or	AL,AL
	jne	short CLOOP
EXIT:
	_pop	DS
	mov	SI,CX	; restore SI
	ret	DATASIZE*2
endfunc			; }
