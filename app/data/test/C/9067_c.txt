#import "TLDNode.h"

@interface TLDDrawNode : TLDNode <TLDDrawableNode>

@property (assign, nonatomic) TLDColor color;

- (instancetype)init NS_DESIGNATED_INITIALIZER;
- (instancetype)initWithColor:(TLDColor)color;

- (void)clear;

/*!
    Please note that the drawPoints methods must be called manually.
 */
- (void)drawPoints:(TLDPoint *)points amount:(GLsizei)amount;
- (void)drawPoints:(TLDPoint *)points amount:(GLsizei)amount mode:(GLenum)mode;

- (void)drawLineFromPoint:(TLDPoint)point1 toPoint:(TLDPoint)point2;
- (void)drawLineWithCenter:(TLDPoint)center radius:(GLfloat)radius andAngle:(GLfloat)angle;

- (void)drawRectangleWithRect:(TLDRect)rect;

- (void)drawArcWithCenter:(TLDPoint)center radius:(GLfloat)radius angle:(GLfloat)angle arcLength:(GLfloat)arcLength numSegments:(GLsizei)numSegments shouldDrawLineToCenter:(BOOL)shouldDrawLineToCenter;

@end
