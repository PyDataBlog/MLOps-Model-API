#!/bin/bash

. $(dirname $0)/include.rc
. $(dirname $0)/volume.rc

function pidgrep()
{
    ps ax | grep "$1" | awk '{print $1}' | head -1
}

## Start glusterd
glusterd;
pidof glusterd;
$CLI volume info;

mkdir -p $B0/brick1/brick $B0/brick2/brick;

## Lets create volume
$CLI volume create $V0 $H1:$B0/brick1/brick   $H1:$B0/brick2/brick;


## Start volume and verify
$CLI volume start $V0;
