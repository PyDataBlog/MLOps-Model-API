OPTS=( \
    "-h:echo \"hello\"&echo \"what\"&echo \"is\"" \
)

for i in "${!OPTS[@]}"; do
    printf "%s\t%s\n" "$i" "${OPTS[$i]}"
    var=${OPTS[$i]}
    echo "-var: $var"
    echo "-opt: ${var%%:*}"
    echo "-ste: ${var##*:}"

echo .......
    line=${var##*:}
    echo "line: ${line}"
echo .......

    p=0
    while [ "${line}" != "" ]; do
        arr[p]=${line%%&*}
        line=${line#*&}
        more=`echo "${line}" | grep "&"`
        ((p++))
        if [ "${more}" == "" ]; then
            arr[p]=${line}
            break
        fi
    done
    
    for var in "${arr[@]}"; do
        echo "var: $var"
    done
done
