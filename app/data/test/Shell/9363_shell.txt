#!/bin/bash

# Desktop background
if [ -x /usr/bin/feh ] ; then
  feh --bg-fill /home/latunov/Pictures/wallpapers/mountain.jpg &
fi

# Start xmobar
xmobar &

# Set keyboard mappings
setxkbmap -layout us,bg -variant ,phonetic -option grp:alt_shift_toggle

# Start terminal
gnome-terminal -x bash -c "screenfetch; exec $SHELL" &

# Start xmonad
xmonad
