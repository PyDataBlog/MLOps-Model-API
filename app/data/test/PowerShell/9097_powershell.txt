param($installPath, $toolsPath, $package, $project)

$config = $xml.configuration

Write-Host $config

Install-Files $installPath $project "sitecore"

Mount-Databases $project "App_Data/databases" $config.sql.instance $config.sql.owner $config.databases.replace $config.databases.with 

Set-ConnectionStrings $project "App_Config\ConnectionStrings.config" $config.sql.instance $config.sql.owner $config.sql.password "Sitecore_"
