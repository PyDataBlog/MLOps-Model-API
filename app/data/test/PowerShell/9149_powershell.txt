Function Get-ActiveExcelWorksheet
{
	<#
	.SYNOPSIS
		Get the current active worksheet
	
	.DESCRIPTION
		The Get-ActiveExcelWorkshet cmd-let always returns the current active working worksheet

	.EXAMPLE
		PS C:\> Get-ActiveExcelWorksheet
	#>
Param
(
)
    Test-ExcelPackage -Force $True -Full $True | Out-Null
    $Script:ActiveExcelWorksheet
}
