Function New-CsController() {
	param( [Parameter(Mandatory=$true)][string]$tableName, [string]$projName = "", [switch]$create, [switch]$read, [switch]$update, [switch]$delete, [switch]$list, [switch]$filter)
	[SolnInfo]$solnInfo = [SolnInfo]::Load()
	New-CsControllerPrivate $solnInfo $solnInfo.webCsprojInfo $solnInfo.dbInfo $tableName $create $read $update $delete $list $filter
}
