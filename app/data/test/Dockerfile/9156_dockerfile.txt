# Create image based on official Node 6 image from dockerhub.
FROM node:6

# Create a directory where our app will be placed.
RUN mkdir -p /srv/www

# Change to the directory where we want our commands to be executed from.
WORKDIR /srv/www

# Copy dependency definitions separate from code to require less build time.
COPY package.json /srv/www

# Install the dependencies.
RUN npm install

# Get the rest of the code needed to run the app.
COPY . /srv/www

# Start the application.
CMD ["npm", "start"]
