BEGIN

Declare output int;

If  _id = 0 Then 

	Insert into 
		appmenu (pid, title, detail, url, weight )
	Values
		(case when _pid = '' then null else _pid end, _title, _detail, _url, _weight);	
		
	Set output = LAST_INSERT_ID();
	
Else 

	Update
		appmenu
	Set
		pid = _pid,
		title = _title,
		detail = _detail,
		url = _url, 
		weight = _weight
	Where
		id = _id;

End if;
