drop table if exists COMPANY_BENEFICIAL;
drop table if exists COMPANY;
drop table if exists COUNTRY;
drop table if exists APP_LANGUAGES;
drop sequence if exists COMPANY_SEQ;
drop sequence if exists BENEFICIAL_SEQ;

create table APP_LANGUAGES(
    LANG NUMBER(2) NOT NULL,
    PRIMARY KEY (LANG)
);

create table COUNTRY(
    CODE CHAR(2) NOT NULL,
    LANG NUMBER (2) NOT NULL,
    NAME VARCHAR(100) NOT NULL,
    PRIMARY KEY (CODE, LANG),
    FOREIGN KEY (LANG) REFERENCES APP_LANGUAGES(LANG)
);

create table COMPANY(
    COMPANY_ID INTEGER NOT NULL,
    NAME VARCHAR(100) NOT NULL,
    ADDRESS VARCHAR(100) NOT NULL,
    CITY VARCHAR(100) NOT NULL,
    COUNTRY_CODE CHAR(2) NOT NULL,
    EMAIL VARCHAR(64) NULL,
    PHONE VARCHAR(16) NULL,
    PRIMARY KEY(COMPANY_ID)
);

create table COMPANY_BENEFICIAL (
    COMPANY_ID NUMBER NOT NULL,
    BENEFICIAL_ID NUMBER NOT NULL,
    NAME VARCHAR(100) NOT NULL,
    PRIMARY KEY (COMPANY_ID, BENEFICIAL_ID)
);

create sequence COMPANY_SEQ;
create sequence BENEFICIAL_SEQ;
