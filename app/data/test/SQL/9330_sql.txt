-- Description:	Conta o número de vezes que um projeto é referenciado na tabela de tarefas.
-- =============================================
CREATE PROCEDURE dbo.task_check_project
	@project_id INT
   ,@result INT OUT
AS
BEGIN
	SET NOCOUNT ON;

    SELECT @result = COUNT(*)
	FROM dbo.task
	WHERE project_id = @project_id
END
