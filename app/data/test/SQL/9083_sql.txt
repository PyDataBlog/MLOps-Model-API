create table admin
(
  id       int unsigned auto_increment
    primary key,
  username varchar(100) not null,
  password varchar(100) not null,
  constraint username
  unique (username)
);

create table expense
(
  id          int unsigned auto_increment
    primary key,
  pcn         varchar(10)    not null,
  water       decimal(20, 2) null,
  electricity decimal(20, 2) null,
  cleaning    decimal(20, 2) null,
  cctv        decimal(20, 2) null,
  constraint pcn
  unique (pcn)
);

create table job
(
  id     int unsigned auto_increment
    primary key,
  titlen int          null,
  title  varchar(100) null,
  constraint titlen
  unique (titlen)
);

create table salary
(
  id      int unsigned auto_increment
    primary key,
  titlen  int            not null,
  wage    decimal(20, 2) null,
  jwage   decimal(20, 2) null,
  benefit decimal(20, 2) null,
  constraint titlen
  unique (titlen)
);

create table staff
(
  id         int unsigned auto_increment
    primary key,
  pcn        varchar(10)  not null,
  idn        varchar(30)  null,
  name       varchar(100) null,
  gender     varchar(10)  null,
  age        int          null,
  seniority  int          null,
  department varchar(100) null,
  titlen     int          not null,
  phone      varchar(100) null,
  mobile     varchar(100) null,
  constraint pcn
  unique (pcn)
);
