local lang = vRP.lang

local items = {}

local function play_drink(player)
  local seq = {
    {"mp_player_intdrink","intro_bottle",1},
    {"mp_player_intdrink","loop_bottle",1},
    {"mp_player_intdrink","outro_bottle",1}
  }

  vRPclient.playAnim(player,{true,seq,false})
end

local pills_choices = {}
pills_choices[lang.items.actions.take()] = {function(player,choice)
  local user_id = vRP.getUserId(player)
  if user_id ~= nil then
    if vRP.tryGetInventoryItem(user_id,"pills",1) then
      vRPclient.varyHealth(player,{25})
      vRPclient.notify(player,{lang.items.pills.use()})
      play_drink(player)
      vRP.closeMenu(player)
    end
  end
end}

items["pills"] = {lang.items.pills.name(),lang.items.pills.description(),function(args) return pills_choices end,0.1}

return items
