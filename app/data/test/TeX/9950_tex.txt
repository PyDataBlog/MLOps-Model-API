\def\dominoH{10pt}
\def\dominoW{5pt}

\def\drawDominoH#1#2{\draw[fill=white] (#1,#2) rectangle (#1+\dominoH,#2+\dominoW);}
\def\drawDominoV#1#2{\draw[fill=white] (#1,#2) rectangle (#1+\dominoW,#2+\dominoH);}

\def\rdomino#1#2 {
   \ifthenelse{\equal{#1}{=}}{
     \draw (0,0) rectangle (\dominoH,\dominoW);
     \draw (0,\dominoW) rectangle (\dominoH,2*\dominoW);
     \begin{scope}[xshift=\dominoH]
   }{
   \ifthenelse{\equal{#1}{|}}{
    \draw (0,0) rectangle (\dominoW,\dominoH);
    \begin{scope}[xshift=\dominoW]
   }{
   \ifthenelse{#1 > 0 \AND #1 < 10}{
    \draw[pattern=north east lines] (0,0) rectangle (#1*\dominoW,\dominoH);
    \begin{scope}[xshift=#1*\dominoW]
   }{}}}
   \ifthenelse{\equal{#2}{}}{
     % tail is empty
   }{
     \rdomino#2 
   }   
   \end{scope}
}

\def\drawDomino#1{%
 \begin{tikzpicture}[inner sep=0pt,baseline=2pt]
   \rdomino#1 
 \end{tikzpicture}%
}

\def\rdominothree#1#2 {
   \ifthenelse{\equal{#1}{E}}{
     \draw (0,0) rectangle (\dominoH,\dominoW);
     \draw (0,\dominoW) rectangle (\dominoH,2*\dominoW);
     \draw (0,2*\dominoW) rectangle (\dominoH,3*\dominoW);
     \begin{scope}[xshift=\dominoH]
   }{
   \ifthenelse{\equal{#1}{n}}{
     \draw (0,0) rectangle (\dominoH,\dominoW);
     \draw (0,\dominoW) rectangle (\dominoW,\dominoW+\dominoH);
     \draw (\dominoW,\dominoW) rectangle (2*\dominoW,\dominoW+\dominoH);
     \begin{scope}[xshift=\dominoH]
   }{
   \ifthenelse{\equal{#1}{u}}{
     \draw (0,0) rectangle (\dominoW,\dominoH);
     \draw (\dominoW,0) rectangle (2*\dominoW,\dominoH);
     \draw (0,\dominoH) rectangle (\dominoH,\dominoW+\dominoH);
     \begin{scope}[xshift=\dominoH]
   }{
   \ifthenelse{\equal{#1}{i}}{
    \drawDominoV{0}{0}
    \begin{scope}[xshift=\dominoW]
   }{
   \ifthenelse{\equal{#1}{|}}{
    \drawDominoV{0}{\dominoW}
    \begin{scope}[xshift=\dominoW]
   }{
   \ifthenelse{\equal{#1}{S}}{
     \drawDominoH{0}{0}
     \drawDominoH{0}{\dominoW}
     \drawDominoH{\dominoW}{2*\dominoW}
     \begin{scope}[xshift=\dominoH]
   }{
   \ifthenelse{\equal{#1}{Z}}{
     \drawDominoH{\dominoW}{0}
     \drawDominoH{0}{\dominoW}
     \drawDominoH{0}{2*\dominoW}
     \begin{scope}[xshift=\dominoH]
   }{
   \ifthenelse{\equal{#1}{r}}{
     \drawDominoV{0}{0}
     \drawDominoH{0}{\dominoH}
     \begin{scope}[xshift=\dominoW]
   }{
   \ifthenelse{\equal{#1}{L}}{
     \drawDominoH{0}{0}
     \drawDominoV{0}{\dominoW}
     \begin{scope}[xshift=\dominoW]
   }{
   \ifthenelse{\equal{#1}{<}}{
     \begin{scope}[xshift=-\dominoW]
   }{
   \ifthenelse{#1 > 0 \AND #1 < 10}{
    \draw[pattern=north east lines] (0,0) rectangle (#1*\dominoW,3*\dominoW);
    \begin{scope}[xshift=0]
   }{}
   }}}}}}}}}}
   \ifthenelse{\equal{#2}{}}{
     % tail is empty
   }{
     \rdominothree#2 
   }   
   \end{scope}
}

\def\drawDominoThree#1{%
 \begin{tikzpicture}[inner sep=0pt,baseline=4pt]
   \rdominothree#1 
 \end{tikzpicture}%
}
