#!/usr/bin/env perl

use strict;
use warnings;

use File::Slurp qw/read_file/;
use Algorithm::Combinatorics qw/permutations combinations/;
use List::Util qw/max/;
use Data::Dumper;

my $input = 377;
my ($a, $b) = (0,0);

my @res = (0);
my $pos = 0;
for (my $i = 1; $i <= 2017; $i++) {
	$pos = ($pos+$input) % $i;
	$pos += 1;
	splice(@res, $pos, 0, $i);
}
$a = $res[$pos+1];
print "Part One Result: $a\n";

$pos = 0;
for (my $i = 1; $i <= 50000000; $i++) {
	$pos = ($pos+$input) % $i;
	$pos += 1;
	$b = $i if $pos == 1;
}
print "Part Two Result: $b\n";
