ifeq ($(VELOX_COREDIR),)
$(error error : VELOX_COREDIR undefined,  source configure-velox-project.sh in Velox Project)
else
include $(VELOX_COREDIR)/etc/common_defines.mk
include $(VELOX_COREDIR)/$(LOCAL_TOOLCHAIN_DIR)/etc/defines.mk
endif

PORTNAME=gcc
PORTVERSION=linaro-5.3.1-2016.05-x86_64_arm-linux-gnueabihf
DISTNAME=$(PORTNAME)-$(PORTVERSION)
DISTFILE=$(DISTNAME).tar.xz
DISTURL=http://releases.linaro.org/components/toolchain/binaries/5.3-2016.05/arm-linux-gnueabihf/
DISTMD5=9815a2f56fea6f1ff14905ab5997326a
MAINTAINER=Stewart Maguire <stewart.maguire@s2ack.com>
DOSTRIP=no

all: build

extract: genericextract

build: fetch extract
	@$(ECHO_TARGET_NOTHING)

install_tmp: create_tmp_dir
	$(TOOLCHAIN_CP) $(VELOX_TOOLCHAIN_PATH)/$(VELOX_HOST_TRIPLET)/lib $(TOOLCHAIN_TMPDIR)
	$(TOOLCHAIN_CP) $(VELOX_TOOLCHAIN_PATH)/$(VELOX_HOST_TRIPLET)/lib/libstdc++*.so* $(TOOLCHAIN_TMPDIR)/usr/lib
	$(TOOLCHAIN_CP) $(VELOX_TOOLCHAIN_PATH)/$(VELOX_HOST_TRIPLET)/libc/lib $(TOOLCHAIN_TMPDIR)
	$(TOOLCHAIN_CP) $(VELOX_TOOLCHAIN_PATH)/$(VELOX_HOST_TRIPLET)/libc/usr/lib $(TOOLCHAIN_TMPDIR)/usr
	$(TOOLCHAIN_CP) $(VELOX_TOOLCHAIN_PATH)/$(VELOX_HOST_TRIPLET)/libc/usr/bin $(TOOLCHAIN_TMPDIR)/usr
	$(TOOLCHAIN_CP) $(VELOX_TOOLCHAIN_PATH)/$(VELOX_HOST_TRIPLET)/libc/usr/sbin $(TOOLCHAIN_TMPDIR)/usr
	$(TOOLCHAIN_CP) $(VELOX_TOOLCHAIN_PATH)/$(VELOX_HOST_TRIPLET)/libc/usr/share $(TOOLCHAIN_TMPDIR)/usr
	$(TOOLCHAIN_CP) $(VELOX_TOOLCHAIN_PATH)/$(VELOX_HOST_TRIPLET)/libc/sbin/ldconfig* $(TOOLCHAIN_TMPDIR)/sbin

#so some reason cross compile wont work using the GNU ld scripts
LD_FIXES=/usr/lib/libc.so /usr/lib/libpthread.so

install: install_tmp install_target
	$(MAKE) fix_ld
	$(MAKE) portinstalled

include $(VELOX_COREDIR)/$(LOCAL_TOOLCHAIN_DIR)/etc/rules.mk
