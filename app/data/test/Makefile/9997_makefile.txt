CXX = g++
CXXFLAGS = -Wall -Wextra -g -O0
LDFLAGS = 

TARGET = machine
SOURCES = $(shell find -name '*.cpp')
OBJECTS = $(shell find -name '*.cpp' | sed 's/cpp/o/')
HEADERS = $(shell find -name '*.hpp')

$(TARGET): $(OBJECTS)
	$(CXX) $(OBJECTS) -o $(TARGET) $(LDFLAGS)

$(OBJECTS): $(HEADERS)

.PHONY: clean
clean:
	rm $(OBJECTS) $(TARGET)

.PHONY: run
run: $(TARGET)
	./$(TARGET)

.PHONY: valgrind
valgrind: $(TARGET)
	valgrind ./$(TARGET)
