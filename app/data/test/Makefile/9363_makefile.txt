GO=go
GODOCKER=CGO_ENABLED=0 GOOS=linux GOARCH=amd64 go
TAG=latest
BIN=bigbrother
IMAGE=dailyhotel/$(BIN)

build:
	glide install
	$(GODOCKER) build -a -ldflags '-w -s' -installsuffix cgo -o bin/$(BIN) .

test: build
	$(GO) test -race -coverprofile=coverage.txt -covermode=atomic

image:
	docker build -t $(IMAGE):$(TAG) .

deploy: image
	docker push $(IMAGE):$(TAG)

.PHONY: clean

clean:
	rm -rf bin/
	rm -f coverage.txt

cleanall: clean
	rm -rf vendor/
