.PHONE: devenv-start

APP_NAME = mymo
DEVENV_DOCKER_COMPOSE = docker-compose -p $(APP_NAME) -f $(shell pwd)/docker/docker-compose.yml

devenv-start:
	${DEVENV_DOCKER_COMPOSE} up -d

build: src/**/*.go pkg/**/*.go
	go build -o out/mymo-api cmd/mymo-api/main.go

run: build
	./mymo-api serve

migration: build
	./mymo-api migration up
