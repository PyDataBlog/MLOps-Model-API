#pragma once

namespace pgraph {

template<class Value>
class OneElementRange {
public:
  using iterator = Value*;
  using const_iterator = const Value*;
  
  explicit OneElementRange(const Value& value)
    : m_value{value} {
  }

  iterator begin() {
    return m_value;
  }
  const_iterator begin() const {
    return m_value;
  }
  iterator end() {
    return m_value + 1;
  }
  const_iterator end() const {
    return m_value + 1;
  }
  bool empty() const {
    return false;
  }
  
private:
  Value m_value[1];
};
  
} // namespace pgraph
