package org.xmly.thrall.core.protocol;

public enum WebResponseStatus {
	
	SUCCESS(200, "success"),
	AUTH(401,"auth"),
	FAIL(400, "fail");
	
	WebResponseStatus(int code, String desc) {
		this.code = code;
		this.desc = desc;
	}
	
	private int code;
	private String desc;
	
	public int getCode() {
		return code;
	}
	public String getDesc() {
		return desc;
	}
	
}
