<?php

namespace Cloudtek\DynamoDM\Mapping;

use Doctrine\Common\Persistence\Mapping\MappingException as BaseMappingException;

/**
 * Exception indicating that no valid mapping data was found for the requested class.
 *
 * @author      Jonathan H. Wage <jonwage@gmail.com>
 * @author      Aaron Miller <aaronm@cldtk.com>
 */
class MappingExceptionClassNotMapped extends BaseMappingException
{
    /**
     * @param string $className
     */
    public function __construct($className)
    {
        parent::__construct("Class $className is not a valid document or mapped super class.");
    }
}
