<?php

namespace Proweb\CommonBundle;

use Proweb\CommonBundle\DependencyInjection\Compiler\MapperCollectionPass;
use Proweb\CommonBundle\DependencyInjection\Compiler\PropelEventDispatcherCompilerPass;
use Proweb\CommonBundle\DependencyInjection\Compiler\PropelQueryAsServiceCompilerPass;
use Symfony\Component\DependencyInjection\ContainerBuilder;
use Symfony\Component\HttpKernel\Bundle\Bundle;

/**
 * @author Marek Kalnik <marekk@theodo.fr>
 * @package NosbProwebCommonBundle
 */
class ProwebCommonBundle extends Bundle
{
    public function build(ContainerBuilder $container)
    {
        parent::build($container);

        $container->addCompilerPass(new PropelQueryAsServiceCompilerPass());
        $container->addCompilerPass(new MapperCollectionPass());
        $container->addCompilerPass(new PropelEventDispatcherCompilerPass());
    }
}
