import * as _ from "lodash";

export interface ReduxAction {
    type: string;
    payload?: any;
    error?: boolean;
    meta?: _.Dictionary<any>;
}

export function createAction(type: string, payload?: any, error?: boolean, meta?: _.Dictionary<any>): ReduxAction {
    return {
        type,
        payload,
        error,
        meta
    };
}
