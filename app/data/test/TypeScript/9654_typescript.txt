'use strict';

import {
  Component
} from '@angular/core';

import {
  Router
} from '@angular/router';

import { OrderService } from '../services/order.service';
import { ProductFlowService } from '../services/product-flow.service';

@Component({

  selector: 'place-order-button',

  templateUrl: 'views/components/place-order-button.html',

  styleUrls: [
    'styles/components/place-order-button.component.css'
  ],

  'providers': [

    OrderService

  ]

})

export class PlaceOrderButtonComponent {

  constructor (
    private productFlow: ProductFlowService,
    private order: OrderService,
    private router: Router
  ) {

    return this;

  }

  public addToOrder (): void {

    this
      .order
      .addItem(
        this
          .productFlow
          .currentViewedProduct
      );

    this
      .router
      .navigate([
        '/order'
      ]);

  }

}
