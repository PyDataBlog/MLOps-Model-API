package org.oxbow.codebridge.ui.dialog

import java.io.File
import org.controlsfx.control.textfield.CustomTextField
import org.oxbow.codebridge.ui.common.Glyphs
import javafx.scene.Cursor
import javafx.scene.control.ComboBox
import javafx.scene.control.Label
import javafx.scene.input.MouseEvent
import javafx.scene.layout.GridPane
import javafx.scene.layout.Priority
import javafx.stage.DirectoryChooser
import javafx.stage.Stage
import org.oxbow.codebridge.ui.common.FileSelectorField
import javafx.scene.control.TextField
import javafx.scene.layout.Region

class NewRepositoryPage extends GridPane {

    setVgap(7)
    setHgap(5)

    val cbType = new ComboBox[String]
    cbType.getItems.add("Git")
    cbType.getSelectionModel().select(0)
    
    val txName = new TextField
    val txPath = new FileSelectorField("Select repository location")

    add(new Label("Type"),     0, 0); add(cbType, 1, 0)
    add(new Label("Name"),     0, 1); add(txName, 1, 1)
    add(new Label("Location"), 0, 2); add(txPath, 1, 2)
    
    GridPane.setHgrow(txPath, Priority.ALWAYS)
    GridPane.setHgrow(txName, Priority.ALWAYS)
    setPrefSize( Region.USE_COMPUTED_SIZE, Region.USE_COMPUTED_SIZE)
    
}
