package com.twitter.searchbird

import com.twitter.app.App
import com.twitter.util.Await

object DemoClient extends App {
  def main() {
    val client = SearchbirdClient.Robust(":8000")

    println(Await.result(client.put("Hello", "World")))
    println(Await.result(client.get("Hello")))
    println(Await.result(client.search("World")))
  }
}
