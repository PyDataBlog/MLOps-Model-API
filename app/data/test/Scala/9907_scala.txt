package errors

import play.api.libs.json.Json
import play.api.mvc.Results.Ok

trait ErrorResults {

  def error(msg: String) = Ok(Json.obj("error" -> msg))

  def invalidLang(lang: String) = error(s"no supported language with code '$lang'.")

  def noSuchCourse(courseCode: String) = error(s"no course with code '$courseCode'.")

  def noSuchExam(courseCode: String, examCode: String) =
    error(s"no exam with code '$examCode' for course with code '$courseCode'.")

  def noStats(lang: String) = error(s"no stats are available in language with code '$lang'.")
}
