  * Given a list of integers, count the number of subsets of those integers which exactly equal the given sum.
  *
  * Running time = O(2^n * n).
  *
  * TODO: Use dynamic programming to create a more time-efficient solution, as well as print the subsets.
  *
  */

object SubsetSum {
  def subsetSum(sum: Int, values: List[Int]): Int = {
    println("*")
    if (sum == 0) 1
    else if (sum != 0 && values.isEmpty) 0
    else if (values.head > sum) subsetSum(sum, values.tail)
    else subsetSum(sum - values.head, values.tail) + subsetSum(sum, values.tail)
  }
}
