package cwe.scala.library.reasoning
import cwe.scala.library.math.PositiveRational

/**
 * Represents a commitment of type T
 * R is the type used by the PositiveRational (should have a numeric box)
 */
trait Commitment[T, R] {
	/**
	 * Returns the committed value
	 */
	def committed: T
	
	/**
	 * The degree of commitment
	 */
	def degree: PositiveRational[R]
	
	/**
	 * Returns true if commitment degree is full (equal to one)
	 */
	def isFull: Boolean = degree != null && degree.isOne
	/**
	 * Returns true if commitment degree is null (equal to zero)
	 */
	def isNull: Boolean = degree == null || degree.isZero
}
