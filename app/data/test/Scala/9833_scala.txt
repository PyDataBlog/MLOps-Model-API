package olympian.party.engine

/**
 * @author kevinbayes
 *
 */
trait RepositoryBackedAccessor {
  
  val repositoryName: String
  
  def useRepository[T,REP](f:(REP) => Option[T]): Option[T] = EngineRegistry.getSingleton(repositoryName) match {
    case Some(repository: REP) => f(repository)
    case None => None
  }

}
