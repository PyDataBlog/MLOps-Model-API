include $(TOPDIR)/rules.mk
include $(INCLUDE_DIR)/kernel.mk
include $(INCLUDE_DIR)/version.mk

PKG_NAME:=superscale-base-files
PKG_RELEASE:=1
PKG_FLAGS:=nonshared

include $(INCLUDE_DIR)/package.mk

define Package/superscale-base-files
  SECTION:=base
  CATEGORY:=Superscale
  DEPENDS:=
  TITLE:=Base filesystem for superscale
  URL:=http://fap/
  VERSION:=$(PKG_RELEASE)-$(REVISION)
endef

define Package/superscale-base-files/description
 This package contains a base filesystem and system scripts for OpenWrt.
endef

define Build/Prepare
	mkdir -p $(PKG_BUILD_DIR)
endef

define Build/Compile/Default

endef

define Package/superscale-base-files/install
	$(CP) ./files/* $(1)/
	chmod 600 $(1)/etc/dropbear/authorized_keys
endef

$(eval $(call BuildPackage,superscale-base-files))
