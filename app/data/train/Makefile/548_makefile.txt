ifeq ($(VELOX_COREDIR),)
$(error error : VELOX_COREDIR undefined,  source configure-velox-project.sh in Velox Project)
else
include $(VELOX_COREDIR)/etc/common_defines.mk
include $(VELOX_COREDIR)/$(LOCAL_PORTS_DIR)/etc/defines.mk
include $(VELOX_COREDIR)/$(LOCAL_PORTS_DIR)/etc/kernel_module.mk
endif

PORTNAME=ft6x06
PORTVERSION=1.0
DISTNAME=$(PORTNAME)-$(PORTVERSION)
MAINTAINER=Stewart Maguire <stewart.maguire@s2ack.com> 
DISTMTS=yes

ifneq ($(VELOX_ARCH), $(VELOX_ARCH_NATIVE))
all: build
else
all: build
endif

extract: 
	@$(ECHO_TARGET_NOTHING)

$(CONFIGURE):
	mkdir -p $(VELOX_WORK)/$(DISTNAME)
	cp -R files/* $(VELOX_WORK)/$(DISTNAME)

build: configure
	cd $(VELOX_WORK)/$(DISTNAME) && $(MAKE) $(JOBS) $(KERNEL_MAKE_ARGS)

install_modules:
	cd $(VELOX_WORK)/$(DISTNAME) && $(MAKE) $(KERNEL_MAKE_ARGS) modules_install
	
install:
	$(PORTINSTALL) $(MAKE) install_modules
	$(MAKE) portinstalled
	
include $(VELOX_COREDIR)/$(LOCAL_PORTS_DIR)/etc/rules.mk
