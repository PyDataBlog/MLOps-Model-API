ifeq ($(VELOX_COREDIR),)
$(error error : VELOX_COREDIR undefined,  source configure-velox-project.sh in Velox Project)
else
include $(VELOX_COREDIR)/etc/common_defines.mk
include $(VELOX_COREDIR)/$(LOCAL_PORTS_DIR)/etc/defines.mk

PORTNAME=Cheetah
PORTVERSION=2.4.4
DISTNAME=$(PORTNAME)-$(PORTVERSION)
DISTFILE=$(DISTNAME).tar.gz
DISTURL=http://pypi.python.org/packages/source/C/Cheetah
DISTMD5=853917116e731afbc8c8a43c37e6ddba
MAINTAINER=Stewart Maguire <stewart.maguire@s2ack.com> 
DISTMTS=yes
DEPENDS=devel/python devel/python-markdown

ifneq ($(VELOX_ARCH),$(VELOX_ARCH_NATIVE))
all: native build
else
all: build
endif

extract: genericextract

$(CONFIGURE):
	@$(ECHO_TARGET_NOTHING)

build: genericbuild
	cd $(VELOX_WORK)/$(DISTNAME) && $(SETUPTOOLS_ENV) $(PYTHON_INTERPRETER) ./setup.py build $(SETUPTOOLS_BUILD)

install:
	cd $(VELOX_WORK)/$(DISTNAME) && $(SETUPTOOLS_ENV) $(PORTINSTALL) $(PYTHON_INTERPRETER) ./setup.py install $(SETUPTOOLS_INSTALL) 
	$(MAKE) portinstalled

include $(VELOX_COREDIR)/$(LOCAL_PORTS_DIR)/etc/rules.mk
