.PHONY: all
all:
	@$(MAKE) --no-print-directory -C test1
	@$(MAKE) --no-print-directory -C test2
.PHONY: run1
run1:
	@$(MAKE) --no-print-directory -C test1 run
.PHONY: run2
run2:
	@$(MAKE) --no-print-directory -C test2 run
.PHONY: clean
clean:
	@$(MAKE) --no-print-directory -C test1 clean
	@$(MAKE) --no-print-directory -C test2 clean
