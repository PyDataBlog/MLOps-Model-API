BUILD_DIR := ../build
INCLUDE_DIR := ../include
SH_LIBS_HOME := ../../../shLibs

OPTIONS := -Wall -g -I$(INCLUDE_DIR) -I$(SH_IMAGE_HOME)/include -I$(SH_MACROS_HOME)/include

$(BUILD_DIR)/shfont.o: shfont.cpp $(INCLUDE_DIR)/shfont.h
	g++ -o $(BUILD_DIR)/shfont.o -c shfont.cpp $(OPTIONS)
