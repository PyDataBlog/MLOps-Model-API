include ../config

MAGIC_OBJS:= luamagic.o

all: $(MAGIC_SO)

$(MAGIC_SO): $(MAGIC_OBJS)
	$(LD) $(LDFLAGS) -o $@ $(MAGIC_OBJS) -lmagic

# List of dependencies.
luamagic.o: luamagic.c luamagic.h

clean:
	rm -f $(MAGIC_SO) $(MAGIC_OBJS)
