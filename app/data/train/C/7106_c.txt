#ifndef MESSAGE_H
#define MESSAGE_H

#include <QString>

#include "enums.h"
#include "abstractdata.h"

namespace QAprs {
    class Message : public AbstractData
    {
        public:
            Message();

            // Getters
            QString inline getRecipient() { return recipient; }
            QString inline getText() { return text; }
            int inline getId() { return id; }
            MessageType inline getType() { return type; }
            bool inline isMessage() { return (type == NORMAL); }

            // Setters
            bool setRecipient(QString recipient);
            bool setText(QString text);
            bool setId(int id = 1);
            void inline setType(MessageType type = NORMAL) { this->type = type; }

        protected:
            QString recipient;
            QString text;
            int id;
            MessageType type;
    };
}

#endif // MESSAGE_H
