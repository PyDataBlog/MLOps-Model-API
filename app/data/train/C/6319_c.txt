#include <unistd.h>
#include <sys/types.h>
#include <stdio.h>
#include <pthread.h> //include library to handle threads

void *Print(){
	pthread_t t1 = pthread_self();
	printf("Thread: %u\n", (unsigned int) t1);
	pthread_exit(NULL);	
}

int main(){
	pthread_t thread1;
	pthread_create(&thread1,NULL,Print,NULL);
	pthread_join(thread1,NULL);
	pthread_exit(NULL);
	return 0;
}
