#include <stdio.h>
#include <stdlib.h>
#include <windows.h>

int main()
{
    HDC hDC = GetDC(GetConsoleWindow());
    HPEN Pen = CreatePen( PS_SOLID, 2, RGB(255, 255, 255));

    SelectObject( hDC, Pen );

    MoveToEx( hDC, 20, 50, NULL );
    LineTo( hDC, 100, 50 );
    MoveToEx( hDC, 20, 60, NULL );
    LineTo( hDC, 100, 60 );

    system("pause");
    return 0;
}
