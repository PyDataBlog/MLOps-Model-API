#import <UIKit/UIKit.h>

typedef void(^YYZAlertViewActionHandler)(UIAlertAction *action);

NS_CLASS_AVAILABLE_IOS(8_0) @interface YYZAlertView : NSObject

+ (void)yyz_alertViewWithTitle:(NSString *)title
                       message:(NSString *)message
             cancelActionTitle:(NSString *)cancelActionTitle;

+ (void)yyz_alertViewWithTitle:(NSString *)title
                       message:(NSString *)message
                   actionTitle:(NSString *)actionTitle
                 actionHandler:(YYZAlertViewActionHandler)actionHandler;

+ (void)yyz_alertViewWithTitle:(NSString *)title
                       message:(NSString *)message
             cancelActionTitle:(NSString *)cancelActionTitle
              otherActionTitle:(NSString *)otherActionTitle
                 actionHandler:(YYZAlertViewActionHandler)actionHandler;

@end
