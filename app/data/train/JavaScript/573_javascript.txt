(function(){
	'use strict'

	angular
		.module("jobDetail")
		.service("jobDetailService",jobDetailService);

	jobDetailService.$inject = ['apiService','apiOptions'];

	function jobDetailService(apiService,apiOptions)
	{
		var jobId;
		this.getJobDetail=function(jobId)
		{
			// return "ok";
			return apiService.get("job/"+jobId);

		};
	};


})();
