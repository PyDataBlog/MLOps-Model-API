module.exports = {
	User: function(id) {
		this.id = id;
	},
	UserSummary: function(id, href, badgeCount, pointCount) {
		this.id = id;
		this.href = href;
		this.badgeCount = badgeCount;
		this.pointCount = pointCount;
	},
	UserDetails: function(id, href, badgeCount, pointCount, awards) {
		this.id = id;
		this.href = href;
		this.badgeCount = badgeCount;
		this.pointCount = pointCount;
		this.awards = awards;
	}
};
