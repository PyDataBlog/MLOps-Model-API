'use strict';

(function(module) {
  const splashView = {};

  splashView.defaultView = function(){
    $('#about').hide();
    $('#map').css('z-index', -500);
    $('form').fadeIn();
    $('#splash-page').fadeIn();
  }
  module.splashView = splashView;
})(window);

$(function(){
  $('#home').on('click', function(){
    splashView.defaultView();
  });
});
