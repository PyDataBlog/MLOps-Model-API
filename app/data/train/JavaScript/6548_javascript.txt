const {
    Command
} = require("discord.js-commando");

module.exports = class Say extends Command {
    constructor(client) {
        super(client, {
            name: "say",
            aliases: ["echo", "speak"],
            group: "util",
            memberName: "say",
            description: "Says what you want it to say",
            examples: ["say hello"],
            autoAliases: true,
            args: [{
                key: "sayText",
                prompt: "What do you want me to say?",
                type: "string"
            }]
        });
    }


    run(msg, {
        sayText
    }) {
        if (msg.deletable === true) {
            msg.delete();
        }
        return msg.say(sayText);
    };
}
