module.exports.sum = function (arr, prop, exp) {
    var total = 0
    for (var i = 0, _len = arr.length; i < _len; i++) {
        var value = arr[i][prop];
        if (exp) {
            if (arr[i][exp.field] == exp.value) {
                total += value * 1;
            }
        } else {
            total += value * 1;
        }
    }
    return total
};
