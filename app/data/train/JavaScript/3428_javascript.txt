(function () {
    var Demo = {
        init: function () {
            this.syntaxHighlight();
            this.sticky();
        },

        syntaxHighlight: function () {
            hljs.initHighlighting();
        },

        sticky: function () {
            var $sticky = $('[data-sticky]');

            $sticky.sticky({
                topSpacing: 10
            });
        }
    };

    Demo.init();
})();
