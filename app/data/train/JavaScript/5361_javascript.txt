import React from 'react';

class Icon extends React.Component {
    constructor(props) {
        super(props);
        this.displayName = 'Icon';
        this.icon = "fa " + this.props.icon + " " + this.props.size;
    }
    render() {
        return (
			<i className={this.icon}></i> 
        	);
    }
}

export default Icon;
