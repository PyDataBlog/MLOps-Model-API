Function Get-XMLWebRequest($url) {
  Write-Host "Called for $url"
  $xml = New-Object xml
  $resolver = New-Object -TypeName System.Xml.XmlUrlResolver
  $reader = New-Object -TypeName System.Xml.XmlReaderSettings
  $reader.XmlResolver = $resolver
  $xml.Load([System.Xml.XmlReader]::Create($url, $reader))
  Return $xml
}
