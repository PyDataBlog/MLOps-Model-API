# Variables
[string]$strWorkDirectory = "D:\NewMovies"

# Define working directory
Set-Location $strWorkDirectory

# Delete all "Proof" folders in all subfolders
Get-Childitem -Path $strWorkDirectory -Include "Proof" -Recurse -Directory | Remove-Item -Recurse -Force

# Delete all "Sample" folders in all subfolders
Get-Childitem -Path $strWorkDirectory -Include "Sample" -Recurse -Directory | Remove-Item -Recurse -Force

# Delete all nfo files, except movie.nfo, in all subfolders
Get-ChildItem -Path $strWorkDirectory -Filter *.nfo -Exclude movie.nfo -Recurse -File | Remove-Item

# Delete all jpg files, except folder.jpg, fanart.jpg, in all subfolders
Get-ChildItem -Path $strWorkDirectory -Filter *.jpg -Exclude folder.jpg,fanart.jpg,thumb*.jpg -Recurse -File | Remove-Item

# Delete all txt files in all subfolders
Get-ChildItem -Path $strWorkDirectory -Filter *.txt -Recurse -File | Remove-Item

# Delete all sfc files in all subfolders
Get-ChildItem -Path $strWorkDirectory -Filter *.sfv -Recurse -File | Remove-Item

# Delete all url files in all subfolders
Get-ChildItem -Path $strWorkDirectory -Filter *.url -Recurse -File -Exclude "OFDb - *" | Remove-Item

# Delete all html files in all subfolders
Get-ChildItem -Path $strWorkDirectory -Filter *.html -Recurse -File | Remove-Item
