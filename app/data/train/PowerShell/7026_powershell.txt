Class LogConfig {
    [string]$LogFilePath
    [string]$LogFileName
    [int]$LogFileMaxSizeInMB
    [int]$LogFileMaxGenerations
    [string]$SQLInstance
    [string]$SQLDatabase
    [string]$SQLTable
    [string]$EventLogName
    [string]$EventSourceName
    [int]$EventID
    [bool]$LoggingLevel
}

Add-Type -TypeDefinition @"
    [System.Flags]
    public enum LogType {
        None = 0,
        LogFile = 1,
        SQL = 2,
        EventLog = 4
    }
"@

$a = [Logtype]"Logfile" -bor [LogType]"SQL"
[LogType] "LogFile,SQL"

$a
($a -and [Logtype]"LogFile")
($a -and [Logtype]"SQL")
