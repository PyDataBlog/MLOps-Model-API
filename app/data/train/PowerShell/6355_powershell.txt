$source = @"
public class Calc
{
    public int Add(int a, int b)
    {
        return a+b;
    }
    
    public int Mul(int a, int b)
    {
        return a*b;
    }
    
    public static float Divide(int a, int b)
    {
        return (float)a/b;
    }
}
"@

Add-Type -TypeDefinition $source

$CalInstance = New-Object -TypeName Calc

$CalInstance.Add(2,3)
[Calc]::Divide(12,5)
