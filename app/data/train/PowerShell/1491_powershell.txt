#Requires -RunAsAdministrator

# Update default Administrator
$Admin_Password = Read-Host -Prompt "Passsword for user Administrator" -AsSecureString
$Admin_Account = Get-LocalUser -Name "Administrator"
$Admin_Account | Set-LocalUser -Password $Admin_Password

# Adapt local groups
Add-LocalGroupMember -Group "Administratoren" -Member "Administrator"
Add-LocalGroupMember -Group "Remotedesktopbenutzer" -Member "Administrator"

Add-LocalGroupMember -Group "Hauptbenutzer" -Member "Christian"
Add-LocalGroupMember -Group "Remotedesktopbenutzer" -Member "Christian"
