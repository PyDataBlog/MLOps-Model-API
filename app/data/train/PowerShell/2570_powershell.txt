# https://www.saotn.org/windows-server-2012-r2-disk-cleanup-dism/

# analyze the size of the Component Store (WinSxS folder)
dism.exe /online /Cleanup-Image /AnalyzeComponentStore

# removes superseded and unused system files from a system
dism.exe /online /Cleanup-Image /StartComponentCleanup

# all superseded versions of every component in the component store is removed
dism.exe /online /Cleanup-Image /StartComponentCleanup /ResetBase
