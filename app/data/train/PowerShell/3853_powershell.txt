function Move-DomainComputer {
[CmdletBinding()]
<#
.SYNOPSIS
Move-DomainComputer will move a computer object in Active Directory to the specified OU.
.PARAMETER computername
The computer to move
.PARAMETER targetOU
The Active Directory object into which the computer object will be moved.
.EXAMPLE
Move-DomainComputer -computername testWorkstation1 -targetOU AppSenseEnabledWorkstations
.EXAMPLE
Move-DomainComputer -computername testWorkstation1,testWorkstation2 -targetOU AppSenseEnabledWorkstations
.EXAMPLE
Get-Content Computerlist.txt | Move-DomainComputer -targetOU AppSenseEnabledWorkstations
.NOTES
    NAME:	Move-DomainComputer.ps1
    AUTHOR:	Paul D'Antignac
    DATE:	2017/01/16
    Version: 1.0
#>
param(
[parameter(
    Mandatory=$true,
    ValuefromPipeline=$true)]
[string[]]$computername,
[parameter(
    Mandatory=$true)]
[ValidateSet('_NewComputers','UserWorkstations','StandardWorkstations','AppSenseEnabledWorkstations')]
[string]$targetOU
    )
$searcherOU = [adsisearcher]"(&(objectclass=organizationalunit)(name=$targetOU))"
$targetpath = [adsi]$searcherOU.findall().path
Process{
    Foreach ($item in $computername){
        $searcher = [adsisearcher]"(&(objectCategory=computer)(name=$item))"
        $srcComp = $searcher.Findone().Path
        if($srcComp -eq $null){
            Write-Warning "Computer object $item not found.";
            Break
        }else{
            $source = [adsi]$searcher.FindOne().Path
            $source.MoveTo($targetpath)
            }
        }
    }
}
