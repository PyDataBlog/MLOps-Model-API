<#
.Synopsis
Retrieves Incoterms
#>
function Get-TranssmartIncoterm
{
    Param
    (
        [Parameter(Mandatory)]
        [string]$Url,

        [Parameter(Mandatory)]
        [string]$UserName,

        [Parameter(Mandatory)]
        [AllowNull()]
        [string]$Password,

        [int]$ID
    )

    Begin
    {
        $Client = New-Object -TypeName Transsmart.Api.Client -ArgumentList $Url, $UserName, $Password
    }
    Process
    {
        if ($ID)
        {
            $Client.GetIncoterm($ID)
        }        
    }
    End
    {
        if (-not $ID)
        {
            $Client.GetIncoterms()
        }
    }
}
