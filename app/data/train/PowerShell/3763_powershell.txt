Try {
	If (Test-Path $($args[1]) ){ # Destination
		Copy-Item $($args[0]) -Destination "$($args[1])" -verbose -ErrorAction Stop
	}
	else {	
		Write-Host "Please Create: $($args[1])"
		exit 3
	}
}
Catch {
	Write-Host "Error in Copy-Item"
	$error[0]
	exit 2
}
