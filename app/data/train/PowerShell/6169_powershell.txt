function Get-UninstallString {
    [CmdletBinding()]
    Param(
        [Parameter(Mandatory=$true)]
        [String]$InstallDate
    )

    Begin {
        $RegPaths = 'HKLM:\Software\Microsoft\Windows\CurrentVersion\Uninstall',
        'HKLM:\Software\Wow6432Node\Microsoft\Windows\CurrentVersion\Uninstall'
        $Names = @()
        foreach ($Object in $RegPaths) {
            $Items = Get-ChildItem -Path $Object -Recurse | Select-Object -ExpandProperty PSChildName
            $Names += $Items
        }
    }
    Process {
        foreach ($RegPath in $RegPaths) {
            foreach ($Object in $Names) {
                Get-ItemProperty -Path "$RegPath\$Object" -ErrorAction SilentlyContinue| Where-Object {($_.InstallDate -eq $InstallDate) -and $_.UninstallString} |
                    Select-Object -Property UninstallString,InstallDate,DisplayName
            }
        }
    }
    End {}
}
