cls
$today = Get-Date
$today_a = $today.ToShortDateString()
$yesterday = $today.AddDays(-1)
$body = "<html><head></head><body><table border ='1'><th>SamAccountName</th><th>LastName</th><th>FirstName</th><th>EmployeeNumber</th><th>CreatedBy</th>"
$sendEmail = $false
$users = Get-QADUser -SizeLimit 0 -CreatedOn $yesterday
foreach ($user in $users) {
	$sendEmail = $true
	$detailedUser = Get-ADUser -Identity $user.DN -Properties *
	$body += "<tr><td>" + $detailedUser.SamAccountName + "</td><td>" + $detailedUser.Surname + "</td><td>" + $detailedUser.GivenName + "</td><td>" + $detailedUser.EmployeeNumber + "</td><td>" + ($detailedUser.nTSecurityDescriptor.owner).replace("CHART-IND\","") + "</td></tr>"
	
}

if ($sendEmail) {
	Send-MailMessage -SmtpServer "charthubcasus01.chart-ind.local" -To "kevin.merolla@chartindustries.com" -From "AD-Audit@chartindustries.com" -Subject "AD Audit - Daily AD Users Report - $today_a" -Body $body -BodyAsHtml
}
