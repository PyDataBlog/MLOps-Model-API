#Remove Portgroups from All Hosts in a given cluster

function Remove-VLANs 
{

Param (
[string]$VC,[string]$VLAN,[string]$VSWITCH,[string]$Cluster
)
connect-viserver $VC 
$ESXS = get-cluster $Cluster | Get-VMhost * -server $VC  
    ForEach ($ESX in $ESXS) {
        Get-VMHost $ESX | Get-VirtualSwitch -Name $VSWITCH | Get-VirtualPortGroup -Name $VLAN | Remove-VirtualPortGroup -Confirm:$false
        }
Disconnect-VIServer $VC 
}
