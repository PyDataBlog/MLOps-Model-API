Function Set-ActiveExcelWorksheet
{
	<#
	.SYNOPSIS
		Set the current active excel worksheet
	
	.DESCRIPTION
		The Set-ExcelWorkshet cmd-let always set the active excel worksheet to first one, unless define otherwise
	
	.PARAMETER WorksheetName
		Specifies the name of the worksheet

	.EXAMPLE
		PS C:\> Add-ExcelWorksheet -WorksheetName "Sheet1"
	#>
Param
(
    [Int]$WorksheetIndex = 1,
    [String]$WorksheetName
)
    Test-ExcelPackage -Force $True -Full $True | Out-Null
    $Script:ActiveExcelWorksheet = $Script:ExcelPackage.Workbook.Worksheets.Item($WorksheetIndex)
    If
    (
        $WorksheetName
    )
    {
        $MatchingWorksheet = $Script:ExcelPackage.Workbook.Worksheets | Where-Object {$_.Name -eq $WorksheetName}
        If
        (
            $MatchingWorksheet
        )
        {
            $Script:ActiveExcelWorksheet = $MatchingWorksheet
        }
        Else
        {
            Write-Warning $DefaultActiveExcelWorkseetWarning
        }
    }
}
