Function New-ExcelPackage
{
	<#
	.SYNOPSIS
		Return a new excel package
	
	.DESCRIPTION
		The New-ExcelPackage cmd-let returns an OfficeOpenXml.ExcelPackage object which allows user to perform excel application related operations
	
	.PARAMETER Full
		Specifies if user wants to create a fully functional excel file

	.EXAMPLE
		PS C:\> New-ExcelPackage -Full $True
	#>
Param
(
    [Boolean]$Full = $False
)
    $Script:ExcelPackage = New-Object -TypeName OfficeOpenXml.ExcelPackage
    If
    (
        $Full
    )
    {
        Add-ExcelWorksheet -WorksheetName "Sheet1"
        Set-ActiveExcelWorksheet
    }
}
