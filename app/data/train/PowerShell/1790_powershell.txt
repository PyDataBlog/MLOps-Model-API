function Get-VMwareToolsStatus
{

Param(
[string]$vcenter,[string]$vm
)
connect-viserver $vcenter
$a = New-VIProperty -Name ToolsVersion -ObjectType VirtualMachine -ValueFromExtensionProperty 'Config.tools.ToolsVersion'-Force 
$b = New-VIProperty -Name ToolsVersionStatus -ObjectType VirtualMachine -ValueFromExtensionProperty 'Guest.ToolsVersionStatus' -Force
Get-VM $vm | Select Name, Version, ToolsVersion, ToolsVersionStatus
disconnect-viserver $vcenter -confirm:$false

}
