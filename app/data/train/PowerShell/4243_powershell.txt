Import-Module sqlserver -Verbose:$false

$SQLserverName = "automasjonsql.felles.ds.nrk.no"
$DatabaseName  = "GenerellAutomasjon"
$date = Get-Date -uformat "%d.%m.%Y-%H%M"

$Config = Invoke-Sqlcmd -ServerInstance $SQLserverName -Database $DatabaseName -Query "SELECT * FROM dbo.Digas_Config_Master" -ErrorAction Stop 
$FolderConfig = Invoke-Sqlcmd -ServerInstance $SQLserverName -Database $DatabaseName -Query "SELECT * FROM dbo.Digas_Config_Folders_Master" -ErrorAction Stop

$Config | Export-Csv D:\Planlagte_oppgaver\SCOM\DigasMPConfig\Export\Config_$date.csv -Encoding UTF8 -Force -NoTypeInformation
$FolderConfig | Export-Csv D:\Planlagte_oppgaver\SCOM\DigasMPConfig\Export\FolderConfig_$date.csv -Encoding UTF8 -Force -NoTypeInformation
