Add-PSSnapin Microsoft.SharePoint.PowerShell
Set-location $home

$siteUrl = "http://jkxfs8waagaa04t/sites/team"
$termStoreName = "Managed Metadata Service"
$termgroupName = "People"
$termSetName = "Department"
$termName = "DEV"


$site = Get-spsite $siteUrl
$session = New-Object Microsoft.SharePoint.Taxonomy.TaxonomySession($site)
$termStore = $session.TermStores.Item($termStoreName)
$termGroup = $termStore.Groups.Item($termgroupName)
$termSet = $termGroup.TermSets.Item($termSetName)
$term = $termSet.Terms.Item($termName)

$termSet.CustomProperties | Format-List
$term.CustomProperties | Format-List
$term.LocalCustomProperties | Format-List
