Function Set-ExcelCellValue
{
	<#
	.SYNOPSIS
		Set the value of an excell cell
	
	.DESCRIPTION
		The Set-ExcelCellFormula cmd-let specifies the value of a cell
	
	.PARAMETER CellAddress
        Specify the address of the cell

 	.PARAMETER CellValue
        Specify the value of the cell

	.EXAMPLE
		PS C:\> Set-ExcelCellValue -CellAdress "A1" -CellValue "60"
	#>
Param
(
    [String]$CellAddress = $(Read-Host $SpecifyCellAddressMessage),
    [String]$CellValue = $(Read-Host $SpecifyCellValueMessage)
)
    Test-ActiveExcelWorksheet -Force $True | Out-Null
    $Script:ActiveExcelWorksheet.Cells[$CellAddress].Value = $CellValue
}
