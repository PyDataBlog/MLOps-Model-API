$search_updatecatalog_psm1 = "$(Split-Path -Parent $MyInvocation.MyCommand.Definition)\modules\Search-UpdateCatalog.psm1"
$module_basename = Get-Item -Path $search_updatecatalog_psm1 -ErrorAction Stop | Select-Object -ExpandProperty BaseName

if ( (Get-Module | Where-Object {$_.Name -eq $module_basename}) ) { Remove-Module $module_basename }
Import-Module $search_updatecatalog_psm1

Write-Host "Search-UpdateCatalog -Searchterms ""Windows 2000 SP4"" -ForceLanguage ""en-US"" -MaxResults 1 -AllowPopups -Verbose"
$result = Search-UpdateCatalog -Searchterms "Windows 2000 SP4" -ForceLanguage "en-US" -MaxResults 1 -AllowPopups -Verbose
$result | Format-List
