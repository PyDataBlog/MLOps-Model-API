# Default Editor
function Open-DefaultEditor { Open-SublimeText $args}

# Editors
function Open-SublimeText{& "$env:ProgramFiles\Sublime Text 3\sublime_text.exe" $args}
function Open-NPP{& "$env:ProgramFiles(x86)\Notepad++\notepad++.exe" $args}
function Open-VSCode{code -n $args}

#alias editing
function Open-Environment{Open-DefaultEditor "$shellRoot\env.ps1"}
function Open-Movement{Open-DefaultEditor "$shellRoot\modules\movement.ps1"}
function Open-Tools{Open-DefaultEditor "$shellRoot\modules\tools.psm1"}
function Open-Aliases{Open-DefaultEditor "$shellRoot\modules\aliases.ps1"}
