-- Add Ligatabelle and Teamlist
CREATE TABLE ligalist(
    ligaid CHAR(5) NOT NULL PRIMARY KEY,
    fullname VARCHAR(255),
    shortcut VARCHAR(10)
);
CREATE TABLE teamlist(
    teamid CHAR(6) NOT NULL PRIMARY KEY,
    fullname VARCHAR(255),
    shortcut VARCHAR(10),
    patch VARCHAR(5),
    status TINYINT(1)
);
CREATE TABLE liga_teams(
    liga_teams_id CHAR(7) NOT NULL PRIMARY KEY,
    ligaid CHAR(5) NOT NULL,
    teamid CHAR(6) NOT NULL
);
ALTER TABLE
    teamlist ADD status TINYINT(1);
    
SELECT * FROM liga_teams WHERE ligaid = '00000' JOIN teamlist ON teamlist.teamid = liga_teams.teamid
