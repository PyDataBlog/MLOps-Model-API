CREATE VIEW [Process].[vewSurveyTimeMostRecent]
WITH SCHEMABINDING 
AS
SELECT        TOP (100) PERCENT SubjectTag, MAX(SurveyYear) AS SurveyYearMostRecent, MAX(AgeCalculateYears) AS AgeCalculateOldest
FROM            Process.tblSurveyTime
WHERE        (SurveyTaken = 1)
GROUP BY SubjectTag
ORDER BY SubjectTag
