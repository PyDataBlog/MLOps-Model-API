SET SQL_MODE="NO_AUTO_VALUE_ON_ZERO";

CREATE TABLE IF NOT EXISTS `Penguins` (
    `ID`	smallint	NOT	NULL AUTO_INCREMENT COMMENT 'Player ID',
	`Username`	varchar(12) NOT NULL COMMENT 'Nickname',
	`Password`	tinytext	NOT	NULL COMMENT 'Hashed Password',
	`Email`		tinytext	NOT	NULL COMMENT 'User E-mail',
	`JoinDate`	timestamp NOT	NULL DEFAULT CURRENT_TIMESTAMP COMMENT 'Player join time in UNIX',
	`Active`	tinyint(1)	NOT	NULL DEFAULT '0' COMMENT 'Account activation status',
	`Banned`	tinyint(1)	NOT	NULL DEFAULT '0' COMMENT 'Account banned status',
	`LoginKey`	varchar(35) NOT NULL COMMENT 'Login key used for game authentication',
	`Inventory`	mediumblob	NOT NULL DEFAULT '' COMMENT 'Player item list',
	`Color`		tinyint(2)	NOT	NULL DEFAULT '0' COMMENT 'Current player color item',
	`Head`		smallint	NOT	NULL DEFAULT '0' COMMENT 'Current player head item',
	`Face`		smallint	NOT	NULL DEFAULT '0' COMMENT 'Current player face item',
	`Neck`		smallint	NOT	NULL DEFAULT '0' COMMENT 'Current player neck item',
	`Body`		mediumint	NOT	NULL DEFAULT '0' COMMENT 'Current player body item',
	`Hands`		mediumint	NOT	NULL DEFAULT '0' COMMENT 'Current player hand(s) item',
	`Feet`		mediumint	NOT	NULL DEFAULT '0' COMMENT 'Current player feet item',
	`Photo`		smallint	NOT	NULL DEFAULT '0' COMMENT 'Current player photo item',
	`Flag`		smallint	NOT	NULL DEFAULT '0' COMMENT 'Current player flag item',
	`Coins`		mediumint(7)	NOT	NULL DEFAULT '5000' COMMENT 'Player coins',
	`Buddies`	blob	NOT	NULL DEFAULT '' COMMENT 'Player buddies',
	`Ignores`	blob	NOT	NULL DEFAULT ''COMMENT 'Player ignores',
	`Mod`		tinyint(1)	NOT	NULL DEFAULT '0' COMMENT 'Determines player moderator status',
	`Rank`		tinyint(1)	NOT	NULL DEFAULT '1' COMMENT 'Player rank',
	PRIMARY KEY (`ID`)
) ENGINE=MyISAM  DEFAULT CHARSET=ascii AUTO_INCREMENT=1;
