/* 스타 */
CREATE TABLE star (
	star_id NUMBER(10) NOT NULL, /* 스타 id */
	ingsta_id NUMBER(10) NOT NULL, /* 잉스타 id */
	state VARCHAR2(10 CHAR), /* 상태 */
	reg_date DATE DEFAULT SYSDATE, /* 등록일 */
	mod_date DATE, /* 수정일 */
	del_date DATE /* 삭제일 */
);

COMMENT ON TABLE star IS '스타';

COMMENT ON COLUMN star.star_id IS '스타 id';

COMMENT ON COLUMN star.ingsta_id IS '잉스타 id';

COMMENT ON COLUMN star.state IS '상태';

COMMENT ON COLUMN star.reg_date IS '등록일';

COMMENT ON COLUMN star.mod_date IS '수정일';

COMMENT ON COLUMN star.del_date IS '삭제일';

CREATE UNIQUE INDEX pk_star
	ON star (
		star_id ASC,
		ingsta_id ASC
	);

CREATE INDEX fk_star_star_id
	ON star (
		star_id ASC
	);

CREATE INDEX fk_star_ingsta_id
	ON star (
		ingsta_id ASC
	);
