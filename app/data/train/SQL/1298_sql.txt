CREATE TABLE [dbo].[Turma] (
    [id]            INT           IDENTITY (1, 1) NOT NULL,
    [nome]          VARCHAR (100) NOT NULL,
    [numeroVagas]   INT           NOT NULL,
    [idInstituicao] INT           NOT NULL,
    [idAno]         INT           NOT NULL,
    [idSerie]       INT           NOT NULL,
    [descricao]     VARCHAR (200) NULL,
    CONSTRAINT [PK_Turma] PRIMARY KEY CLUSTERED ([id] ASC),
    CONSTRAINT [FK_Turma_Ano] FOREIGN KEY ([idAno]) REFERENCES [dbo].[Ano] ([id]),
    CONSTRAINT [FK_Turma_Instituicao] FOREIGN KEY ([idInstituicao]) REFERENCES [dbo].[Instituicao] ([id]),
    CONSTRAINT [FK_Turma_Serie] FOREIGN KEY ([idSerie]) REFERENCES [dbo].[Serie] ([id])
);
