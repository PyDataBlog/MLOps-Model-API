SELECT p.label, p.comment, pro.label, pf.label, p.productPropertyTextual1, p.productPropertyTextual2, p.productPropertyTextual3, p.productPropertyNumeric1, p.productPropertyNumeric2, p.productPropertyTextual4, p.productPropertyTextual5, p.productPropertyNumeric4
FROM Product as p 
JOIN Producer as pro ON (pro.publisher = p.publisher AND pro.id = p.producer) 
JOIN ProductXproductFeature as ppf ON (ppf.Product = p.id)
JOIN ProductFeature as pf ON (pf.id = p.id)
WHERE
	ppf.productFeature AND
	p.label = "Product1";
