CREATE TABLE BREAST_CANCER_BIOMARKERS
(
  ID                     NUMBER(10,2),
  ESTROGEN_RECEPTOR      VARCHAR2(255 BYTE),
  PROGESTERONE_RECEPTOR  VARCHAR2(255 BYTE),
  HER2_STATUS            VARCHAR2(255 BYTE),
  HER2_TEST_TYPE         VARCHAR2(255 BYTE),
  EGFR_STATUS            VARCHAR2(252 BYTE)
)
LOGGING 
NOCACHE
NOPARALLEL;


CREATE UNIQUE INDEX PK_BREAST_CANCER_BIOMARKERS ON BREAST_CANCER_BIOMARKERS
(ID)
LOGGING
NOPARALLEL;


ALTER TABLE BREAST_CANCER_BIOMARKERS ADD (
  CONSTRAINT PK_BREAST_CANCER_BIOMARKERS PRIMARY KEY (ID));

CREATE INDEX EGFR_STATUS_IND
 ON BREAST_CANCER_BIOMARKERS(EGFR_STATUS);
