CREATE TABLE lesson (
	id uuid NOT NULL DEFAULT uuid_generate_v4(),
	slug varchar NOT NULL,
	label varchar NOT NULL,
	description varchar NOT NULL DEFAULT '',
	media JSONB NOT NULL DEFAULT '{}'::JSONB,
	language_id uuid NOT NULL,
	level integer NOT NULL DEFAULT 1,
	tags varchar NOT NULL DEFAULT '',
	primary_tag varchar NOT NULL DEFAULT '',
	category varchar NOT NULL DEFAULT '',
	identifier varchar NOT NULL DEFAULT '',
	CONSTRAINT lesson_pk PRIMARY KEY (id),
	CONSTRAINT lesson_slug_uniq UNIQUE(slug),
	CONSTRAINT language_lesson_fk FOREIGN KEY (language_id) REFERENCES language(id) ON DELETE CASCADE
);
