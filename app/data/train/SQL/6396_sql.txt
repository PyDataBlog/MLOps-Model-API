CREATE DATABASE IF NOT EXISTS ${TARGET_DB};
USE ${TARGET_DB};

DROP TABLE IF EXISTS ${TARGET_DB}.concept_ancestor;

CREATE TABLE ${TARGET_DB}.concept_ancestor(
    ancestor_concept_id 		BIGINT 	   ,
    descendant_concept_id 		BIGINT 	   ,
    max_levels_of_separation 	INT
)
PARTITIONED BY (min_levels_of_separation INT)
STORED AS ${STORED_FORMAT}
TBLPROPERTIES ("orc.compress"="${COMPRESS}", "orc.stripe.size"="${STRIPSIZE}", "orc.row.index.stride"="${ROWINDEXSTRIPE}")
;

FROM ${SOURCE_DB}.concept_ancestor ca
INSERT OVERWRITE TABLE concept_ancestor PARTITION (min_levels_of_separation)
SELECT
        ca.ancestor_concept_id
        ,ca.descendant_concept_id
        ,ca.max_levels_of_separation
        ,ca.min_levels_of_separation
WHERE   ca.min_levels_of_separation<5
INSERT OVERWRITE TABLE concept_ancestor PARTITION (min_levels_of_separation)
SELECT
        ca.ancestor_concept_id
        ,ca.descendant_concept_id
        ,ca.max_levels_of_separation
        ,ca.min_levels_of_separation
WHERE ca.min_levels_of_separation>=5
SORT BY ca.ancestor_concept_id
;
