use football;

drop table if exists pred;

create table pred ( season char(9) not null, 
	     	    matchdate date not null,
                    hometeam varchar(25) not null,
                    awayteam varchar(25) not null,
		    method   char(1) not null, 
                    homescore smallint not null,
		    awayscore smallint not null, 
		    result char(1),
		    rank float
		      );

create UNIQUE index id_pred ON pred( season, matchdate, hometeam, awayteam, method); 
create index id_pred_season ON pred( season ); 
create index id_pred_hometeam ON pred( hometeam ); 
create index id_pred_awayteam ON pred( awayteam ); 
create index id_pred_matchdate ON pred( matchdate ); 
create index id_pred_method ON pred( method ); 


commit;
