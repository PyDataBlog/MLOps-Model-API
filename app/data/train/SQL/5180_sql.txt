-- список журналів для реєстрування кореспонденції
drop table if exists dovidnyk.dov_jorn;
--
drop sequence if exists dovidnyk.seq_jornid;
create sequence dovidnyk.seq_jornid start 1;
grant usage, select on sequence dovidnyk.seq_jornid to approle;
--
create table dovidnyk.dov_jorn
 (id     int default 1,
  orgid  int,
  idx    varchar(20),
  nazvat varchar(500),
  nazvas varchar(500),
  nazvan varchar(500),
  typ    int,
  bdt    date,
  edt    date,
  active int default 0,
  constraint dov_jorn_pk primary key (id));
--
comment on table dovidnyk.dov_jorn  is 'список журналів для реєстрування кореспонденції';
comment on column dovidnyk.dov_jorn.id is 'ід. журналу';
comment on column dovidnyk.dov_jorn.orgid is 'ідентифікатор актуальної назви організації';
comment on column dovidnyk.dov_jorn.idx is 'індекс журналу (номер справи)';
comment on column dovidnyk.dov_jorn.nazvat is 'титульна назва';
comment on column dovidnyk.dov_jorn.nazvas is 'коротка назва';
comment on column dovidnyk.dov_jorn.nazvan is 'повна назва (коментар)';
comment on column dovidnyk.dov_jorn.typ is 'тип журналу: 1-вхідна кореспонденція, 2-';
comment on column dovidnyk.dov_jorn.bdt is 'дата початку';
comment on column dovidnyk.dov_jorn.edt is 'дата закінчення';
--
grant all on table dovidnyk.dov_jorn to approle;
--
