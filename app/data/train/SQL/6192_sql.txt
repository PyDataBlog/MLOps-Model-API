CREATE TABLE pokemon_species_names (
	pokemon_species_id INTEGER NOT NULL, 
	local_language_id INTEGER NOT NULL, 
	name VARCHAR(79), 
	genus TEXT, 
	CONSTRAINT pk_pokemon_species_names_pokemon_species_id_local_language_id PRIMARY KEY (pokemon_species_id, local_language_id), 
	CONSTRAINT fk_pokemon_species_names_pokemon_species FOREIGN KEY(pokemon_species_id) REFERENCES pokemon_species (id), 
	CONSTRAINT fk_pokemon_species_names_languages FOREIGN KEY(local_language_id) REFERENCES languages (id)
);
GO
CREATE INDEX ix_pokemon_species_names_name ON pokemon_species_names (name);
