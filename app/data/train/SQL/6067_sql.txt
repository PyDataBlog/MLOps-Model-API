/*Associationテーブル*/
--DROP TABLE Association;
CREATE TABLE Association(
	/*作成者ID*/
	CreateUserId BIGINT NOT NULL,
	/*作成日時*/
	CreateDateTime TIMESTAMP NOT NULL,
	/*更新者ID*/
	UpdateUserId BIGINT NOT NULL,
	/*更新日時*/
	UpdateDateTime TIMESTAMP NOT NULL,
	/*プログラムID*/
	ProgramId VARCHAR NOT NULL,
	/*AssociationID*/
	AssociationId BIGSERIAL NOT NULL,
	/*Associationタイプ*/
	AssociationType VARCHAR NOT NULL,
	/*sourceObject*/
	SourceObject VARCHAR NOT NULL,
	/*sourceObject(ORG)*/
	SourceObjectOrg VARCHAR NOT NULL,
	/*targetObject*/
	TargetObject VARCHAR NOT NULL,
	/*targetObject(ORG)*/
	TargetObjectOrg VARCHAR NOT NULL,
	/*オブジェクトタイプ*/
	ObjectType VARCHAR NOT NULL,
	/*entryUUID*/
	EntryUuid VARCHAR NOT NULL,
	/*entryUUID(ORG)*/
	EntryUuidOrg VARCHAR NOT NULL,
	CONSTRAINT Association_PKey PRIMARY KEY (AssociationId)
);
COMMENT ON TABLE Association IS 'Associationテーブル';
COMMENT ON COLUMN Association.CreateUserId IS '作成者ID';
COMMENT ON COLUMN Association.CreateDateTime IS '作成日時';
COMMENT ON COLUMN Association.UpdateUserId IS '更新者ID';
COMMENT ON COLUMN Association.UpdateDateTime IS '更新日時';
COMMENT ON COLUMN Association.ProgramId IS 'プログラムID';
COMMENT ON COLUMN Association.AssociationId IS 'AssociationID';
COMMENT ON COLUMN Association.AssociationType IS 'Associationタイプ';
COMMENT ON COLUMN Association.SourceObject IS 'sourceObject';
COMMENT ON COLUMN Association.SourceObjectOrg IS 'sourceObject(ORG)';
COMMENT ON COLUMN Association.TargetObject IS 'targetObject';
COMMENT ON COLUMN Association.TargetObjectOrg IS 'targetObject(ORG)';
COMMENT ON COLUMN Association.ObjectType IS 'オブジェクトタイプ';
COMMENT ON COLUMN Association.EntryUuid IS 'entryUUID';
COMMENT ON COLUMN Association.EntryUuidOrg IS 'entryUUID(ORG)';
