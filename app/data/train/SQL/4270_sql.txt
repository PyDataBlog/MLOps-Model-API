-- USUARIO1: Se desea que el usuario2 no tenga accesos a toda la tabla.
revoke all on tbl_empleado from usuario2;

-- USUARIO1: Crear una vista sin los campos restringidos
create view usuario1.v_tbl_empleado
as
select empleado_id, nombre1, apellido1
from usuario1.tbl_empleado;

-- SYSTEM: usuario1 no tiene acceso para crear una vista
grant create view to usuario1;

-- USUARIO1: Otorgar accesos a la vista para el usuario2
grant select on v_tbl_empleado to usuario2;
