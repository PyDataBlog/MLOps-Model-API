CREATE TABLE [dbo].[Country]
(
	[Id] INT NOT NULL PRIMARY KEY, 
    [Version] INT NULL, 
    [CreationDate] DATETIME NOT NULL, 
    [CreationUser] NVARCHAR(50) NOT NULL, 
    [ModificationDate] DATETIME NULL, 
    [ModificationUser] NVARCHAR(50) NULL, 
    [Name] NVARCHAR(50) NOT NULL, 
    [NameFR] NVARCHAR(50) NOT NULL, 
    [IsoAlpha3Code] CHAR(10) NOT NULL
)

GO


CREATE INDEX [IX_Country_Name] ON [dbo].[Country] ([Name])

GO

CREATE INDEX [IX_Country_NameFR] ON [dbo].[Country] ([NameFR])
