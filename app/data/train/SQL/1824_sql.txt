CREATE TABLE generation_names (
	generation_id INTEGER NOT NULL, 
	local_language_id INTEGER NOT NULL, 
	name VARCHAR(79) NOT NULL, 
	CONSTRAINT pk_generation_names_generation_id_local_language_id PRIMARY KEY (generation_id, local_language_id), 
	CONSTRAINT fk_generation_names_generations FOREIGN KEY(generation_id) REFERENCES generations (id), 
	CONSTRAINT fk_generation_names_languages FOREIGN KEY(local_language_id) REFERENCES languages (id)
);
GO
CREATE INDEX ix_generation_names_name ON generation_names (name);
