CREATE DEFINER=`sudeesh`@`localhost` PROCEDURE `getConfig`(
	IN p_ID INTEGER
)
BEGIN
	
    SELECT
		ID,
        ConfigName,
        Description,
        IsSystem,
        Created,
        Modified,
        OperatorID,
        Inactive
	FROM
		Config 
	WHERE
		ID = p_ID;
        
	SELECT
		ID,
        ConfigID,
        KeyName,
        Value,
        IsSystem,
        Created,
        Modified,
        OperatorID,
        Inactive
	FROM
		ConfigItem
	WHERE
		ConfigID = p_ID;
        
END
