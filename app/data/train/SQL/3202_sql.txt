SELECT
    r.alunno as IDAlunno,
    p.nome,
    p.cognome,
    r.disciplina as IDDisciplina,
    d.nome AS disciplina,
    d.unita,
    r.risultato
FROM risultati r
LEFT JOIN alunni p
    ON p.id = r.alunno
LEFT JOIN discipline d
    ON d.id = r.disciplina
WHERE !ISNULL(r.risultato)
ORDER BY timestamp DESC
LIMIT 10
