

SET SQL_MODE = "NO_AUTO_VALUE_ON_ZERO";



CREATE TABLE IF NOT EXISTS `page_views` (
  `ID` int(255) AUTO_INCREMENT PRIMARY KEY,
  `SSID_USER` varchar(30) DEFAULT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;



CREATE TABLE IF NOT EXISTS `customer_mastertable` (
  `CUST_ID` varchar(10) NOT NULL,
  `CUST_NAME` varchar(50) DEFAULT NULL,
  `CUST_ADDRESS` varchar(50) DEFAULT NULL,
  `CUST_CITY` varchar(20) DEFAULT NULL,
  `CUST_STATE` varchar(10) DEFAULT NULL,
  `CUST_PHONE` decimal(10,0) DEFAULT NULL,
  `CUST_EMAIL` varchar(40) DEFAULT NULL,
  `CUST_DOB` varchar(200) DEFAULT NULL,
  `INCOME_GROUP` varchar(100) DEFAULT NULL,
  `OCCUPATION` varchar(20) DEFAULT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;



CREATE TABLE IF NOT EXISTS `customer_ordering` (
  `ID` int(150) AUTO_INCREMENT PRIMARY KEY,
  `CUST_ID` varchar(10) DEFAULT NULL

CREATE TABLE IF NOT EXISTS `discount_allowed` (
  `ID` int(255) AUTO_INCREMENT PRIMARY KEY,

) ENGINE=InnoDB DEFAULT CHARSET=utf8;

  `LAST_UPDATE` varchar(200) DEFAULT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
