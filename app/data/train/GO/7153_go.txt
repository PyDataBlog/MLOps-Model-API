// Package soundcloud implements the OAuth2 protocol for authenticating users through Facebook.
// This package can be used as a reference implementation of an OAuth2 provider for Goth.
package soundcloud

import (
	"bytes"
	"encoding/json"
	"io"
	"io/ioutil"
	"net/url"
	"net/http"
	"strings"
	"strconv"

	"github.com/laicosly/goth"
	"golang.org/x/oauth2"
)

const (
	authURL         string = "https://soundcloud.com/connect"
	tokenURL        string = "https://api.soundcloud.com/oauth2/token"
	endpointProfile string = "https://api.soundcloud.com/me"
)

// New creates a new Facebook provider, and sets up important connection details.
// You should always call `soundcloud.New` to get a new Provider. Never try to create
// one manually.
func New(clientKey, secret, callbackURL string, scopes ...string) *Provider {
	p := &Provider{
		ClientKey:   clientKey,
		Secret:      secret,
		CallbackURL: callbackURL,
	}
	p.config = newConfig(p, scopes)
	return p
}

// Provider is the implementation of `goth.Provider` for accessing Facebook.
type Provider struct {
	ClientKey   string
	Secret      string
	CallbackURL string
	config      *oauth2.Config
}

// Name is the name used to retrieve this provider later.
func (p *Provider) Name() string {
	return "soundcloud"
}

// Debug is a no-op for the soundcloud package.
func (p *Provider) Debug(debug bool) {}

// BeginAuth asks Facebook for an authentication end-point.
func (p *Provider) BeginAuth(state string, params url.Values) (goth.Session, error) {
	url := p.config.AuthCodeURL(state)
	session := &Session{
		AuthURL: url,
		State: state,
		Params: params,
	}
	return session, nil
}

// FetchUser will go to Facebook and access basic information about the user.
func (p *Provider) FetchUser(session goth.Session) (*goth.User, error) {
	sess := session.(*Session)
	user := goth.User{
		AccessToken: sess.AccessToken,
		RefreshToken: sess.RefreshToken,
	}
	url := endpointProfile + "?oauth_token=" + url.QueryEscape(sess.AccessToken)
	response, err := http.Get(url)
	if err != nil {
		return nil, err
	}
	defer response.Body.Close()

	bits, err := ioutil.ReadAll(response.Body)
	if err != nil {
		return nil, err
	}

	err = json.NewDecoder(bytes.NewReader(bits)).Decode(&user.RawData)
	if err != nil {
		return nil, err
	}

	err = userFromReader(bytes.NewReader(bits), &user)
	return &user, err
}

// UnmarshalSession will unmarshal a JSON string into a session.
func (p *Provider) UnmarshalSession(data string) (goth.Session, error) {
	sess := &Session{}
	err := json.NewDecoder(strings.NewReader(data)).Decode(sess)
	return sess, err
}

func userFromReader(reader io.Reader, user *goth.User) error {

	u := struct {
		ID          int `json:"id"`
		Permalink   string `json:"permalink"`
		Username    string `json:"username"`
		Description string `json:"description"`
		Picture     string `json:"avatar_url"`
		Name        string `json:"full_name"`
		Country     string `json:"country"`
		City        string `json:"city"`
	}{}

	err := json.NewDecoder(reader).Decode(&u)
	if err != nil {
		return err
	}

	user.Name = u.Name
	user.NickName = u.Username
	user.Description = u.Description
	user.AvatarURL = u.Picture
	user.UserID = strconv.Itoa(u.ID)
	user.Location = u.City + " " + u.Country

	return err
}

func newConfig(provider *Provider, scopes []string) *oauth2.Config {
	c := &oauth2.Config{
		ClientID:     provider.ClientKey,
		ClientSecret: provider.Secret,
		RedirectURL:  provider.CallbackURL,
		Endpoint: oauth2.Endpoint{
			AuthURL:  authURL,
			TokenURL: tokenURL,
		},
		Scopes: []string{
			"non-expiring",
		},
	}

	defaultScopes := map[string]struct {}{
		"non-expiring": {},
	}

	for _, scope := range scopes {
		if _, exists := defaultScopes[scope]; !exists {
			c.Scopes = append(c.Scopes, scope)
		}
	}

	return c
}
