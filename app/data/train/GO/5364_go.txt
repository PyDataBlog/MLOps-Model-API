package lifegame

type Universe struct {
	aliveCells []Cell
}

type Cell struct{}

func NewUniverse() *Universe {
	return &Universe{}
}

func (u *Universe) HasAliveCell() bool {
	return len(u.AliveCells()) != 0
}

func (u *Universe) AliveCells() []Cell {
	return u.aliveCells
}

func (u *Universe) NextGeneration() {
	u.aliveCells = []Cell{}
}

func (u *Universe) BornCellAtLocation(x, y int) {
	u.aliveCells = append(u.aliveCells, Cell{})
}

func (c *Cell) Location() (int, int) {
	return 0, 0
}
