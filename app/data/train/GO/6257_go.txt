type Rocket struct{
    Name string
}

func (r *Rocket) Launch(){

type InSet struct{
    word []uint64
}

func (s *InSet) Has(x uint64) bool{
    word,bit:=x/64,uint64(x%64)
    return word<len(s.words)&& s.words[word]&(1<<bit)!=0
}



func main() {
    r1:=Rocket{"Mike"}
