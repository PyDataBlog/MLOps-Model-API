/*
Package menu contains structure and methods for the creation and display of menus.
*/
package menu

import "fmt"

type MenuFunc func()

/*
The MenuEntry struct is used to populate the menu values for a CLI.
*/
type MenuEntry struct {
	MenuText     string
	MenuSelector int
	MenuFunction MenuFunc
	StopRunning  bool
}

/*
PrintEntry causes the MenuEntry to print to standard out in human readable form.
*/
func (entry MenuEntry) PrintEntry() {
	fmt.Printf("%d - %v\n", entry.MenuSelector, entry.MenuText)
}

/*
IsSelected determines whether or not the provided selector indicates this MenuEntry has been chosen.
*/
func (entry MenuEntry) IsSelected(selector int) bool {
	return entry.MenuSelector == selector
}
