package tmpl

// A wrapper around the data to be passed to the template. It automatically
// adds a bit of useful information, such as a message the author may want to
// send, the Goji context in which the template is being executeds, and the
// name of the page being rendered.
type PageData struct {
	Name    string                 // The name of the page
	Context map[string]interface{} // The current c.Env
	Data    interface{}            // The data to be used in the page
}
