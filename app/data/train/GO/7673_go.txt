package main

import "fmt"

// 两个参数和
func plus(a int, b int) int {
	return a + b
}

// 相同参数类型的多个参数声明方式
func plusPlus(a, b, c int) int {
	return a + b + c
}

// https://gobyexample.com/functions
func main() {
	res := plus(1, 2)
	fmt.Println("1+2 = ", res)

	res = plusPlus(1, 2, 3)
	fmt.Println("1+2+3 =", res)
}
