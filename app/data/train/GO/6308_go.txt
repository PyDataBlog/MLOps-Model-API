package db

import (
	"gopkg.in/mgo.v2/bson"
)

type User struct {
	ID      bson.ObjectId `bson:"_id,omitempty"`
	Name    string
	Friends []string
	Groups  []string
}

type Message struct {
	ID      bson.ObjectId `bson:"_id,omitempty"`
	Name    string
	Content string
	From    string
	To      []string
}
type Group struct {
	ID    bson.ObjectId `bson:"_id,omitempty"`
	Users []string
}
