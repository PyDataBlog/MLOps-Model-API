package stripedash

import (
	"fmt"
	"strconv"
)

func Uint64ToDollars(i uint64) string {
	d := []byte(strconv.FormatUint(i, 10))

	// Cents only
	if len(d) < 3 {
		return fmt.Sprintf("0.%s", d)
	}

	// Dollars and cents
	centStart := len(d) - 2

	dollars := d[0:centStart]
	cents := d[centStart:len(d)]
	return fmt.Sprintf("%s.%s", string(dollars), string(cents))
}
