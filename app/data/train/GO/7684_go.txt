package int64s

import "testing"

func TestMaxReturnsZeroWhenGivenZeroArgs(t *testing.T) {
	expected := int64(0)
	if actual := Max(); expected != actual {
		t.Error("Expected", expected, "got", actual)
	}
}

func TestMaxReturnsInputArgWhenGivenOneArgs(t *testing.T) {
	expected := int64(7)
	if actual := Max(7); expected != actual {
		t.Error("Expected", expected, "got", actual)
	}
}

func TestMaxReturnsLargestArgWhenGivenMultipleArgs(t *testing.T) {
	expected := int64(9)
	if actual := Max(1, 3, 5, 7, 9); expected != actual {
		t.Error("Expected", expected, "got", actual)
	}
}
