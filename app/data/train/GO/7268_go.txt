package config

import (
	"os"
)

type config struct {
	BasePath string
	DestPath string
}

var current = config{}

// GetBasePath returns the base directory to backup
func GetBasePath() string {
	return current.BasePath
}

// GetDestPath returns the backup directory
func GetDestPath() string {
	return current.DestPath
}

func init() {
	current.BasePath = "./"
	if len(os.Args) > 1 {
		current.BasePath = os.Args[1]
	}
}
