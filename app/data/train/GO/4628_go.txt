package main


// Redirect the user to url
type Redirect struct {
	HTTPStatusCode int `json:"-"`
	URL string `json:"url"`
}

// RedirectToLogin redirect -> /login
var RedirectToLogin = &Redirect {
	HTTPStatusCode: 401, //StatusUnauthorized
	URL: "/login",
}

// RedirectToLogout redirect -> /logout
var RedirectToLogout = &Redirect {
	HTTPStatusCode: 307, //StatusTemporaryRedirect
	URL: "/logout",
}
