package fbaOutbound

// Address	邮寄地址信息
type Address struct {
	Name                string `json:"Name" label:"收件人的姓名"`
	Line1               string `json:"Line1" label:"街道地址信息"`
	Line2               string `json:"Line2" label:"其他街道地址信息"`
	Line3               string `json:"Line3" label:"其他街道地址信息"`
	DistrictOrCounty    string `json:"DistrictOrCounty" label:"区或县"`
	City                string `json:"City" label:"城市"`
	StateOrProvinceCode string `json:"StateOrProvinceCode" label:"省/自治区/直辖市"`
	CountryCode         string `json:"CountryCode" label:"国家/地区"`
	PostalCode          string `json:"PostalCode" label:"邮政编码"`
	PhoneNumber         string `json:"PhoneNumber" label:"电话"`
}

// CODSettings	您为货到付款(COD) 订单设置的货到付款金额。请注意，货到付款配送订单只适用于中国 (CN) 和日本 (JP)。
type CODSettings struct {
	IsCODRequired     bool     `json:"IsCODRequired" label:"是否要求货到付款"`
	CODCharge         Currency `json:"CODCharge" label:"货到付款订单金额"`
	CODChargeTax      Currency `json:"CODChargeTax" label:"货到付款订单金额税费"`
	ShippingCharge    Currency `json:"ShippingCharge" label:"货到付款运费金额"`
	ShippingChargeTax Currency `json:"ShippingChargeTax" label:"货到付款运费金额税费"`
}

// CreateFulfillmentOrderItem	用于创建配送订单的商品信息
type CreateFulfillmentOrderItem struct {
	SellerSKU                    string   `json:"SellerSKU" label:"商品的卖家SKU"`
	SellerFulfillmentOrderItemId string   `json:"SellerFulfillmentOrderItemId" label:"创建的用于跟踪配送订单商品的配送订单商品编号"`
	Quantity                     int      `json:"Quantity" label:"商品数量"`
	GiftMessage                  string   `json:"GiftMessage" label:"给礼物收件人的消息"`
	DisplayableComment           string   `json:"DisplayableComment" label:"在面向收件人的材料（如出库货件装箱单）中显示的商品详情文本"`
	FulfillmentNetworkSKU        string   `json:"FulfillmentNetworkSKU" label:"商品的亚马逊配送网络SKU"`
	OrderItemDisposition         string   `json:"OrderItemDisposition" label:""`
	PerUnitDeclaredValue         Currency `json:"PerUnitDeclaredValue" label:"卖家为此商品指定的价格"`
	PerUnitPrice                 Currency `json:"PerUnitPrice" label:"买家需要对货到付款 (COD) 订单中的此商品支付的金额"`
	PerUnitTax                   Currency `json:"PerUnitTax" label:"买家需要对货到付款 (COD) 订单中的此商品支付的金额的税费"`
}

// Currency	货币类型和金额
type Currency struct {
	CurrencyCode string `json:"CurrencyCode" label:"货币代码"`
	Value        string `json:"Value" label:"货币金额"`
}

// DeliveryWindow	发送计划配送订单的时间范围。
type DeliveryWindow struct {
	StartDateTime string `json:"StartDateTime" label:"计划配送时间范围的开始日期和时间"`
	EndDateTime   string `json:"EndDateTime" label:"计划配送时间范围的结束日期和时间"`
}

// DeliveryWindowList
type DeliveryWindowList struct {
	member []DeliveryWindow
}

// Fee	费用类型和金额。
type Fee struct {
	Name   string   `json:"Name" label:"费用类型"`
	Amount Currency `json:"Amount" label:"费用金额"`
}

// FeeList
type FeeList struct {
	member []Fee
}
