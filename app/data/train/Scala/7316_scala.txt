package wtf.shekels.alice.als.command.commands

import wtf.shekels.alice.als.command.{Command, CommandInfo}

/**
  * @author alice
  * @since 10/15/17.
  */
object Lob extends Command("Lob", "Translates message to Lob via mobgobc") {

	override def execute(command: CommandInfo): Unit = {
		val pattern = """[aeiouAEIOU]""".r
		val msg = command.message.getContent.split(" ").tail.mkString(" ")
		command.message.getChannel.sendMessage(pattern.replaceAllIn(msg, "ob").toLowerCase)
	}

	override def messageIsValid(command: CommandInfo): Boolean = {
		val msg = command.message.getContent
		!msg.isEmpty && msg.contains(" ")
	}
}
