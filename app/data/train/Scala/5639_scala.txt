package chapter.five

import ExerciseTwo.BankAccount

import org.scalatest._
import org.scalatest.junit.JUnitRunner
import org.junit.runner.RunWith

@RunWith(classOf[JUnitRunner])
class ExerciseTwoSpec extends FlatSpec with Matchers {

  "BankAccount" should "should increase the balance of an account by x when deposit is called with x" in {
    val account = new BankAccount
    val x = 10.00

    account.deposit(x)
    account.balance shouldBe 10.00
  }

  "BankAccount" should "should decrease the balance of an account by x when withdraw is called with x" in {
    val account = new BankAccount
    val initial = 20.00
    val x = 1.00

    account.deposit(initial)
    account.withdraw(x)
    account.balance shouldBe 19.00
  }

}
