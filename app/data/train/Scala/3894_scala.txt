package chessagents
import akka.actor.ActorRef
class ProposalMove(move: (Int,Int,Int,Int), points: Int, ref: ActorRef) {
  var move1 : (Int,Int,Int,Int) = move
  var points1 = points
  var ref1 = ref
  def getPoints(): Int = {
    return points1;
  }
  def getMove(): (Int,Int,Int,Int) = {
    return move1;
  }
  def getRef(): ActorRef = {
    return ref1;
  }
  def setMove(from_x: Int, from_y: Int, to_x: Int, to_y: Int ): Unit = {
    move1 = (from_x,from_y,to_x,to_y)
  }
  def setPoints(points : Int): Unit ={
    points1 = points
  }
  def setPieceActor (ref: ActorRef): Unit = {
    ref1 = ref
  }
}
