package colonlc.mpm

/**
	* Representation of the available cli options
	*
	* @param verbose Verbose output
	* @param quiet No Output
	* @param task Which task to run
	* @param args the left arguments
	*/
private[mpm] class Options(
							 val verbose: Boolean,
							 val quiet: Boolean,

							 val task: Tasks.MyVal,
							 val args: Array[String]
						 ) {
	override def toString: String =
		s"""Options:
			 |	verbose: $verbose
			 |	quiet: $quiet
			 |	command: $task
			 |	args: ${args.mkString(", ")}
		 """.stripMargin
}
