package kumoi.shell.vm

import java.net.InetAddress


abstract class VMGraphics

case class SDL(display: String, xauth: String, fullScreen: Boolean) extends VMGraphics
case object SDLAuto extends VMGraphics

case class VNC(port: Int, listen: InetAddress, passwd: String, keymap: String) extends VMGraphics
case object VNCAuto extends VMGraphics

case class Spice(port: Int, tlsPort: Int, channels: List[(String, String)]) extends VMGraphics
case object SpiceAuto extends VMGraphics

case class Rdp(port: Int, replaceUser: Boolean, mutiUser: Boolean) extends VMGraphics
case object RdpAuto extends VMGraphics

case class VBoxDesktop(display: String, fullScreen: Boolean) extends VMGraphics
case object VBoxDesktopAuto extends VMGraphics
