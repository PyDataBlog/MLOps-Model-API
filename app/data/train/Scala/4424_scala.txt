package mr.merc.log

import org.slf4j.LoggerFactory

trait Logging {
  @transient private lazy val logger = LoggerFactory.getLogger(getClass())

  def debug(msg: => String): Unit = {
    if (logger.isDebugEnabled()) {
      logger.debug(msg)
    }
  }

  def debug(msg: => String, er: Throwable): Unit = {
    if (logger.isDebugEnabled()) {
      logger.debug(msg, er)
    }
  }

  def info(msg: => String): Unit = {
    if (logger.isInfoEnabled()) {
      logger.info(msg)
    }
  }

  def info(msg: => String, er: Throwable): Unit = {
    if (logger.isInfoEnabled()) {
      logger.info(msg, er)
    }
  }

  def warn(msg: => String): Unit = {
    if (logger.isWarnEnabled()) {
      logger.warn(msg)
    }
  }

  def warn(msg: => String, er: Throwable): Unit = {
    if (logger.isWarnEnabled()) {
      logger.warn(msg, er)
    }
  }

  def error(msg: => String): Unit = {
    if (logger.isErrorEnabled()) {
      logger.error(msg)
    }
  }

  def error(msg: => String, er: Throwable): Unit = {
    if (logger.isErrorEnabled()) {
      logger.error(msg, er)
    }
  }
}
