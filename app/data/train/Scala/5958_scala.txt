package com.hooboy.etjo.editor.main

import org.newdawn.slick._

object Main {
  def main(args: Array[String]) {
    try {
      val app = new AppGameContainer(new Editor);
      app setTitle "Electronic Turbo Jerk-Off"
      app setDisplayMode (1600, 800, false)
      app setShowFPS false
      app start
    } catch {
      case e: SlickException => e printStackTrace
    }
  }
}
