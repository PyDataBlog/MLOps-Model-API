package nl.ebpi.yaidom.queryapitests.indexed

import java.{ util => jutil }

import org.junit.runner.RunWith
import org.scalatest.junit.JUnitRunner

import nl.ebpi.yaidom.indexed.Elem
import nl.ebpi.yaidom.parse.DocumentParserUsingDom
import nl.ebpi.yaidom.queryapitests.AbstractRobustQueryTest
import nl.ebpi.yaidom.resolved

/**
 * Query test case for indexed elements.
 *
 * @author Chris de Vreeze
 */
@RunWith(classOf[JUnitRunner])
class RobustQueryTest extends AbstractRobustQueryTest {

  private val logger: jutil.logging.Logger = jutil.logging.Logger.getLogger("nl.ebpi.yaidom.queryapitests.indexed")

  final type E = Elem

  protected final val contactsElem: Elem = {
    val docParser = DocumentParserUsingDom.newInstance

    val is = classOf[RobustQueryTest].getResourceAsStream("/nl/ebpi/yaidom/queryapitests/contacts.xml")

    Elem(Some(new java.net.URI("")), docParser.parse(is).documentElement)
  }

  protected final def toResolvedElem(elem: E): resolved.Elem =
    resolved.Elem(elem.elem)
}
