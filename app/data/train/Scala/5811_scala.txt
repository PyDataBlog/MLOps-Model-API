package chapter.eleven

import ExerciseThree.Fraction

import org.scalatest._
import org.scalatest.junit.JUnitRunner
import org.junit.runner.RunWith

@RunWith(classOf[JUnitRunner])
class ExerciseThreeSpec extends FlatSpec with Matchers {

  "Fraction" should "have a addition operator" in {
    val a = Fraction(2, 4)
    val b = Fraction(3, 6)

    // todo: override equals?
    a + b should be (Fraction(4, 4))
  }

  "Fraction" should "have a subtraction operator" in {
    val a = Fraction(1, 2)
    val b = Fraction(1, 3)

    a - b should be (Fraction(1, 6))
  }

  "Fraction" should "have a multiplication operator" in {
    val a = Fraction(7, 10)
    val b = Fraction(1, 7)

    a * b should be (Fraction(7, 70))
  }

  "Fraction" should "have a division operator" in {
    val a = Fraction(1, 2)
    val b = Fraction(4, 2)

    a / b should be (Fraction(1, 4))
  }

  "Fraction" should "have an overridden toString" in {
    Fraction(2, 10).toString should be ("1/5")
  }
}
