package org.smartpony.consumer.jvm.actor

import akka.actor.ActorSystem
import com.typesafe.config.ConfigFactory
import org.smartpony.core.provider.ProviderActorSystem

trait RunningActorSystem extends ProviderActorSystem {

  private val systemName = "jvm-consumer-system"
  private val config = ConfigFactory.load().getConfig(systemName)

  override lazy val providerActorSystem: ActorSystem = ActorSystem(systemName, config)

}
