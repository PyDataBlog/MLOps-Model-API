package com.github.probe.backend.util

import akka.actor.ActorRef
import akka.dispatch.Future
import akka.pattern.ask
import akka.util.Timeout

object AkkaUtil {

  trait ResponseType[T, R]

  def askT[T, R](actor: ActorRef, msg: T)(implicit timeout: Timeout, rt: ResponseType[T, R], m: Manifest[R]): Future[R] = ask(actor, msg).mapTo(m)
}
