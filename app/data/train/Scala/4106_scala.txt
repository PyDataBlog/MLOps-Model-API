
object Tuples extends App {
  
  val list = Seq(
    (Foo(1), Bar(1)), (Foo(2), Bar(2)), (Foo(3), Bar(3))
  )
  
  //bad
  list.foreach { tuple =>
    println("Foo: " + tuple._1)
    println("Foo: " + tuple._2)
  }
  
  //good
  list.foreach { case (foo: Foo, bar: Bar) =>
    println("Foo: " + foo)
    println("Foo: " + bar)
  }
  
  case class Foo(id: Int)
  case class Bar(id: Int)
}
