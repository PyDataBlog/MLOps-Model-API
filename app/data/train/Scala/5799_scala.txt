package controllers

import play.api._
import play.api.mvc._
import play.api.data._
import play.api.data.Forms._
import play.api.data.validation.Constraints._

import models._
import models.form._

object Login extends Controller {
  
  val loginForm = Form(
      tuple(
      "user" -> text,
      "password" -> text
    ) verifying ("ログインまたはパスワードが違います。", result => result match {
      case (u, p) => Account.authenticate(u, p).isDefined
    })
  )
  
  def index = Action {
    Ok(views.html.login(loginForm))
  }
  def submit = Action { implicit request =>
    loginForm.bindFromRequest.fold(
      errors => BadRequest(views.html.login(errors)),
      user => Redirect(routes.AccountView.index).withSession("user" -> user._1)
    )
  }
  
  /**
   * Logout and clean the session.
   */
  def logout = Action {
    Redirect(routes.Login.index).withNewSession.flashing(
      "success" -> "You've been logged out"
    )
  }
}
