package com.plasmaconduit

import com.plasmaconduit.conveyance.Box
import com.plasmaconduit.edge.http.HttpServerResponse

package object framework {

  import scala.language.implicitConversions

  implicit def toBox[F](src: HttpResponse): Box[F, HttpResponse] = {
    Box.of(src)
  }

  implicit def fromServerResponse(r: HttpServerResponse): HttpResponse = {
    HttpResponse(r.body, r.status, r.headers)
  }

  implicit def observableFromServerResponse[F](r: HttpServerResponse): Box[F, HttpResponse] = {
    Box.of(HttpResponse(r.body, r.status, r.headers))
  }

}
