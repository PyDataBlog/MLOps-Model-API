;9. A byte string S is given. Obtain the string D1 which contains all the positive numbers of S and the string D2 which contains all the negative numbers of S.
;Exemple:
;S: 1, 3, -2, -5, 3, -8, 5, 0
;D1: 1, 3, 3, 5, 0
;D2: -2, -5, -8
assume CS:code, DS:data
data segment
    s db 1, 3, -2, -5, 3, -8, 5, 1
    l EQU $-s
    d1 db l dup (?)
    d2 db l dup (?)
    d1i dw 0
    d2i dw 0
data ends
code segment
start:
    mov AX, data
    mov DS, AX
    
    mov cx, l
    mov si, 0
    jcxz sfarsit

Repeta:
    mov al, byte ptr s[si]
    cmp al, 0
    ;cbw
    ;push ax
    jge push_positive
    jl push_negative

push_negative:
    ;pop ax
    push si
    mov si, d2i
    mov d2[si], byte ptr al
    inc si
    mov d2i, si
    pop si
    jmp LoopReturn

push_positive:
    ;pop ax
    push si
    mov si, d1i
    mov d1[si], byte ptr al
    inc si
    mov d1i, si
    pop si
    jmp LoopReturn

LoopReturn:
    inc si
    ;jcxz sfarsit
    loop Repeta

sfarsit:
    mov AX, 4C00h
    int 21h
code ends
end start
