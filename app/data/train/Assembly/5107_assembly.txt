assume CS:code, DS:data
data segment
    ;(a*a+b)/(a+c/a)
    ;a,c-byte; b-doubleword
    ;UNSIGNED
    a DB 15
    b DD 10
    c DB 50
    result DD ?
data ends
code segment
start:
    mov AX, data
    mov DS, AX
    
    xor AX,AX ; clean ax
    mov AL, a ; move a in AL
    mul a ; a*a
    add word ptr b, AX ; add least part first
    adc word ptr b+2, 0 ; just add the carry to b because ax was a byte

    xor AX, AX ; clean 
    xor DX, DX ; clean more
    mov AL, c ; move c into AL for division
    div a ; divide by a so it's (c/a)
    mov AH, 0 ; remove the result of c%a
    add AL, a ; add a to AL
    adc AH, 0 ; add the carry to AH
    mov BX, AX ; move a+c/a into BX

    mov AX, word ptr b ; prepare AX for double word by word divison
    mov DX, word ptr b+2 ; move most into DX for division
    div BX ; double word by word division (a*a+b)/(a+c/a)
    
    mov AX, 4C00h
    int 21h
code ends
end start
