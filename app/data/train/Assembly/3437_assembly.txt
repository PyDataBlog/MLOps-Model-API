; This procedure deals with the WM_DESTROY windows message (event)
; It simple send a postquitmessage to the windows message loop which in his turn
; exits the application.

include QuitEvent.inc

.code

public QuitEvent
 
QuitEvent proc nExitCode: qword

	push	rbp
	mov		rbp, rsp
	mov		rcx, nExitCode
	sub		rsp, 20h
	call	PostQuitMessage
	add		rsp, 20h
	mov		rsp, rbp
	pop		rbp
	ret

QuitEvent endp

end
