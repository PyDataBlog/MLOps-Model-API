include ExitMessageBox.inc

.data
	msgboxtext     db "Do you really want to exit?", 0

.code

public ExitMessageBox

ExitMessageBox proc
	push	rbp
	mov		rbp, rsp
	push	MB_YESNO
	lea		rax, application + APPLICATION.caption
	push	rax
	lea		rax, msgboxtext
	push	rax
	push	application + APPLICATION.handle
	call	MessageBox
	mov		rsp, rbp
	pop		rbp
	ret							; return to caller
ExitMessageBox endp

end
