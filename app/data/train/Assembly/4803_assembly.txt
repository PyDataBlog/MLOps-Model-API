TITLE seagullbird_t27
.MODEL SMALL
.DATA
	A DB 15 DUP (1)
	B DB 20 DUP (1)
	C DB 15 DUP (0)
.CODE
MAIN PROC FAR
	ASSUME CS:_TEXT, DS:_DATA
	PUSH DS
	XOR AX, AX
	PUSH AX
	MOV AX, @DATA
	MOV DS, AX

	LEA BX, A
	LEA BP, C

	FOR_1:
		CMP BX, LENGTH A
		JGE FINAL
		LEA SI, B
		MOV AX, [BX]
		FOR_2:
			CMP SI, LENGTH B
			JGE END_FOR_2
			CMP AX, DS:[SI]
			JZ CONTINUE
			INC SI
			JMP FOR_2

		CONTINUE:
			MOV DS:[BP], AX
			INC BP
			JMP END_FOR_2

		END_FOR_2:
			INC BX
			JMP FOR_1

	FINAL:
		RET
MAIN ENDP
	END MAIN
