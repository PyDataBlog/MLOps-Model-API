;
; My Code
; User inputs an initial pattern and the the coordinates of an LED, that LED will blink on and off in flashes of 500ms

.data
x			DWORD	0
y			DWORD	0
p			DWORD   0
patterning	BYTE	"Input Pattern" 	

.code
	main:nop

		invoke version
		invoke setScreenSize, 120, 120
		invoke readIntegerWithMessage, addr patterning
		mov p, eax
		invoke readInteger
		mov x, eax
		invoke readInteger
		mov y, eax
		invoke setPattern, p
		jmp flash		

		flash:
			invoke readRow, y
			mov ecx, x
			mov ebx, 1
			shl ebx, CL
			xor eax, ebx
			invoke writeRow, y, eax
			invoke Sleep, 500
			jmp flash


	end main
