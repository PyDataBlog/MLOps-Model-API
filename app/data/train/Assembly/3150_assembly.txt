#Purpose Computes (larger-5)* (smaller+2)
		.text
first:	
	la $a0, q1 #asks first question
	li $v0, 4
	syscall
	
	li $v0, 5 #gets user input and moves it to $t0
	syscall
	move $t0,$v0
	
second:
	la $a0, q2 #asks second question
	li $v0, 4
	syscall
	
	li $v0, 5 #gets user input and moves it to $t1
	syscall
	move $t1,$v0
	
	bgt $t0,$t1,FIRST_IS_GREATER #branches if first integer is greater than second integer  to FIRST_IS_GREATER

SECOND_IS_GREATER:
	sub $t2,$t1,5 #subtracts 5 from second number and stores result in $t2
	add $t3,$t0,2 #adds 2 to first number and stores result in $t3
	mul $t4,$t3,$t2 #multiplies $t2 and $t3 and stores it in $t4
	
	la $a0,answer #prints first part of the answer
	li $v0,4
	syscall
	
	move $a0, $t4 # prints numerical result
	li $v0, 1
	syscall
	j halt #jumps to halt

FIRST_IS_GREATER:
	sub $t2,$t0,5 #subtracts 5 from the first number and stores result in $t2
	add $t3,$t1,2 #adds 2 to sexond number and stores result in $t3
	mul $t4,$t3,$t2 #multiplies $t2 and $t3 and stores it in $t4
	
	la $a0, answer #prints first part of the answer
	li $v0,4 
	syscall
	
	move $a0, $t4 #prints numerical result
	li $v0, 1
	syscall
	
halt: #ends program
	li $v0,10
	syscall
	
	.data
q1:	.asciiz "Enter the first integer: "
q2:	.asciiz "Enter the second integer: "
answer: .asciiz "(larger - 5) * (smaller + 2) = "
