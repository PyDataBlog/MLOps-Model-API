        .386p

        INCL_DOSERRORS equ 1
        include bseerr.inc
        include devhlp.inc
        include segdef.inc
        include r0thunk.inc

CODE16 segment
        ASSUME CS:CODE16, DS:FLAT

        public thunk16$DevHlp32_VMUnlock

thunk16$DevHlp32_VMUnlock:
        call [DevHelp2]
        jmp far ptr FLAT:thunk32$DevHlp32_VMUnlock
;        jmp32 thunk32$DevHlp32_VMUnlock

CODE16 ends

CODE32 segment
ASSUME CS:CODE32, DS:FLAT, ES:FLAT, SS:NOTHING

        ; public thunk32$DevHlp32_VMUnlock
        public         DevHlp32_VMUnlock

;
; int DH32ENTRY DevHlp32_VMUnlock(
;                                 void *pLockHandle   /* ebp + 8 */
;                                );
;
DevHlp32_VMUnlock proc near
        push ebp
        mov  ebp, esp
        push esi

        mov esi, [ebp + 8]        ; pLockHandle  (FLAT)
        mov edx, DevHlp_VMUnlock
        jmp far ptr thunk16$DevHlp32_VMUnlock
thunk32$DevHlp32_VMUnlock label far:
        jc short @@error           ; if error, EAX = error code
        mov eax, NO_ERROR
@@error:
        pop esi
        leave
        ret
DevHlp32_VMUnlock endp

CODE32  ends

        end
