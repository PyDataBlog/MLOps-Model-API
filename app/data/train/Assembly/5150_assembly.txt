assume CS:code, DS:data
data segment
	shit dd 2
    arr dw 5, -6, 127, 112
	temp dw ?
	l EQU $-arr
    i dw ?
data ends
code segment

BUBBLE proc
    mov i, 1
	outerloop:
	  lea si,arr
	  mov dx,l
	  innerloop: 
			mov ax,[si]
			add si,2
			mov bx,[si]
			cmp ax,bx
			jl noswap
			;jg noswap

		swap:
			;push ax
			;push bx
			mov temp,ax
			mov ax,bx
			mov bx,temp
			mov [si],bx
			sub si,2
			mov [si],ax
			add si,2

		noswap: 
			dec dx
			cmp dx,i
			jne innerloop

	inc i
	mov cx,i
	cmp cx,l
	jne outerloop
	ret
BUBBLE ENDP

start:
    mov ax,data
    mov ds,ax
    
	lea si,arr
	call BUBBLE
	;jmp bubble

quit:
    mov AX, 4C00h
    int 21h

code ends
end start
