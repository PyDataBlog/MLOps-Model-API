from sys import argv

def isprime(x):
    """Checks if x is prime number

    Returns true if x is a prime number, otherwise false.

    """
    if x <= 1:
        return False

    for n in range(2, (x -  1)):
        if x % n == 0:
            return False

    return True


def main():
    """Main"""

    filename, number = argv
    print isprime(int(number))

if __name__ == "__main__":
    main()
