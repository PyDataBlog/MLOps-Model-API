import time
import threading


def play(name,count):
    for i in range(1,count):
        print('%s %d in %d' %(name, i, count))
        time.sleep(1)
    return

if __name__=='__main__':
    t1=threading.Thread(target=play, args=('t1',10))
    # 设置为守护线程
    t1.setDaemon(True)
    t1.start()
    print("main")
    # 等待子线程结束
    t1.join()
    exit(1)
