#!/usr/bin/env python
import argparse

from . import core
from . import gui


def main():
    parser = argparse.ArgumentParser(
        description="Track opponent's mulligan in the game of Hearthstone.")
    parser.add_argument('--gui', action='store_true')
    args = parser.parse_args()
    if args.gui:
        gui.main()
    else:
        core.main()


if __name__ == '__main__':
    main()
