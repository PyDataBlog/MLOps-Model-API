class Solution:
    def dailyTemperatures(self, T):
        ans = []
        m = [None]*101
        for i in range(len(T)-1, -1, -1):
            x = T[i]
            m[x] = i
            ans.append(min([x for x in m[x+1:] if x is not None], default=i)-i)
        ans.reverse()
        return ans
print(Solution().dailyTemperatures([73, 74, 75, 71, 69, 72, 76, 73]))
