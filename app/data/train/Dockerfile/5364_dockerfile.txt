# Dockerfile
FROM python:3.4-onbuild
COPY requirements.txt /src/
RUN pip install --requirement /src/requirements.txt
COPY . /src/app

EXPOSE 8000

# Use the following line to not look at logs
# ENTRYPOINT ["/usr/local/bin/gunicorn", "--config", "/src/app/gunicorn.conf", "-b", ":8000", "main:app"]

#Use the following if you want to watch the Gunicorn logs
ENTRYPOINT ["/usr/local/bin/gunicorn", "--config", "/src/app/gunicorn.conf", "--log-config", "/src/app/logging.conf", "-b", ":8000", "main:app"]
