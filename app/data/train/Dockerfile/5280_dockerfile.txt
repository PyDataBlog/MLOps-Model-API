FROM debian:wheezy
MAINTAINER Orca Health <info@orcahealth.com>
# Thanks to Atlas Health <ops@atlashealth.com> for this base configuration!

ENV DEBIAN_FRONTEND noninteractive
ENV LANG C.UTF-8
ENV LANGUAGE C.UTF-8
ENV LC_ALL C.UTF-8

# Removes docs and man files
ADD ./doc_filters /etc/dpkg/dpkg.cfg.d/01_nodoc
RUN cd /usr/share && rm -rf doc/* man/* info/* lintian/*

# Install baseline packages and clean up
RUN apt-get update -q \
    && apt-get install -qy \
      bzip2 ca-certificates curl wget git openssl procps --no-install-recommends \
    && apt-get clean \
    && cd /var/lib/apt/lists \
    && rm -rf *Release* *Sources* *Packages* \
    && truncate -s 0 /var/log/*log

# https://github.com/docker/docker/issues/4032
ENV DEBIAN_FRONTEND newt
