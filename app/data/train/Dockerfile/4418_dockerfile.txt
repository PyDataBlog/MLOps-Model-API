FROM ubuntu

RUN echo "Generic Container for Abacus"

# Use bash
RUN rm /bin/sh && ln -s /bin/bash /bin/sh

# Base tooling
RUN apt-get update && apt-get install -y \
    git \
    build-essential \
    curl \
    unzip \
    autoconf \
    libtool \
    zlib1g-dev \
    python-pip \
    python-dev \
    python3 \
    python3-pip \
    vim

# Copy common libs
RUN mkdir -p /home/app/common
COPY ./common /home/app/common

WORKDIR /home/app

RUN pip3 install --upgrade pip
