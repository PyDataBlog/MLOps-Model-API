FROM golang:latest
MAINTAINER Zachery Hostens <zacheryph@gmail.com>

RUN apt-get update -y \
    && apt-get install -y unzip libpam0g-dev

RUN curl -L -O https://github.com/google/protobuf/releases/download/v3.1.0/protoc-3.1.0-linux-x86_64.zip \
    && unzip protoc-3.1.0-linux-x86_64.zip -x readme.txt -d /usr/local

RUN go get -u -v google.golang.org/grpc \
                 golang.org/x/tools/cmd/stringer \
                 github.com/golang/protobuf/proto \
                 github.com/golang/protobuf/protoc-gen-go
