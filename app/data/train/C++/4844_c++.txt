#include <algorithm>
#include <unordered_map>
#include <random>

#include "RandNum.hpp"

namespace cm
{
	template <typename T> inline T RandReal(T min, T max);
	template <typename T> inline T RandInteger(T min, T max);
}

namespace cm 
{
	static std::mt19937 Engine = std::mt19937(std::random_device()());
}

template<typename T> T cm::RandReal(T min, T max)
{
	std::uniform_real_distribution<T> distribution(min, max);
	return distribution(Engine);
}

template<typename T> T cm::RandInteger(T min, T max)
{
	std::uniform_int_distribution<T> distribution(min, max);
	return distribution(Engine);
}
