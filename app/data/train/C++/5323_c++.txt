#include <iostream>
using namespace std;
int main(){
	int cv, ce, cs, cp;
	int fv, fe, fs, fp;

	cin >> cv >> ce >> cs;
	cin >> fv >> fe >> fs;

	cp = (cv*3)+ce;
	fp = (fv*3)+fe;

	if( cp > fp){
		cout << "C" << endl;
	}else if( fp > cp){
		cout << "F" << endl;
	}else{
		if(cs>fs){
			cout << "C" << endl;
		}else if(fs>cs){
			cout << "F" << endl;
		}else{
			cout << "=" << endl;
		}
	}

	return 0;
}
