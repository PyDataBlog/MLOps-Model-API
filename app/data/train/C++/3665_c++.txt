long gcd(long m, long n) {
  while (m != 0) {
    long k = n%m;
    n=m;
    m=k;
  }
  return n;
}

bool is_square(long S) {
  long sqr = sqrt(S);
  return sqr*sqr == S;
}

int main() {
  long sum{0};
  while (cin) {
    long n;
    cin >> n;
    if (cin) {if (is_square(n)) {cout << n << endl;sum += n;}}
  }
  cout << endl << sum << endl;
}
