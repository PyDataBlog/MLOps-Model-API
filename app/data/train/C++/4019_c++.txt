#include "sdl_initerror.hpp"

///
/// \brief SDL_InitError::SDL_InitError - Initializes the custom exception class, gets the most recent error message
///
SDL_InitError::SDL_InitError():
    exception(), msg(SDL_GetError()) {}

///
/// \brief SDL_InitError::SDL_InitError - Initializes the custom exception class, sets the error message to a user defined string
/// \param m
///
SDL_InitError::SDL_InitError(const std::string& m):
    exception(), msg(m) {}

///
/// \brief SDL_InitError::~SDL_InitError - Destructor, does nothing, there is nothing ot be released
///
SDL_InitError::~SDL_InitError() throw() {}

///
/// \brief SDL_InitError::what - Returns either the most recent error message or a user defined message, depending on which constructor was used
/// \return - The error message
///
const char* SDL_InitError::what_failed() const throw() {
    return this->msg.c_str();
}
