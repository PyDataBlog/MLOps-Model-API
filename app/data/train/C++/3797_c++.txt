#include <gtest/gtest.h>
#include "solution.h"

TEST(test, testcase0)
{
    Stack s;
    EXPECT_EQ(true, s.empty());
    s.push(1);
    s.push(2);
    s.push(3);
    EXPECT_EQ(false, s.empty());
    EXPECT_EQ(3, s.top());
    s.pop();
    EXPECT_EQ(2, s.top());
    s.pop();
    EXPECT_EQ(1, s.top());
    s.pop();
    EXPECT_EQ(true, s.empty());
}

int main(int argc, char **argv)
{
    ::testing::InitGoogleTest(&argc, argv);
    return RUN_ALL_TESTS();
}
