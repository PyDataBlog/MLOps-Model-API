#!/usr/bin/perl/-w


my @fred = qw / 1 3 5 7 9/;

my $fred_total = total(@fred);
print $fred_total;

print "\n--------------------------\n";

my $user_total = total(<STDIN>);
print "user_total is $user_total\n";

# sub function
sub total{

    $sum=0;
    
    foreach my $number(@_){
    
        $sum += $number;
    
    }
    
    $sum;

}
