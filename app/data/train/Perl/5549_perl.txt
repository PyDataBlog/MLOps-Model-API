#!/usr/bin/perl

use warnings;
use strict;

# map your dir

my @files = glob('/some_dir/*');

# setup replace varibles from old to new :)

my @rplc = (
        { 'old' => 'old_value_0', 'new' => 'new_value_0' },
        { 'old' => 'old_value_1', 'new' => 'new_value_1' }
);

# replace with sed ( util from unix ENV )

for(my $i = 0; $i < @files; ++$i){
        for(my $b = 0; $b < @rplc; ++$b){
                print `/bin/sed -i 's^$rplc[$b]{old}^$rplc[$b]{new}^g' $files[$i]`;
        }
}
