#! /usr/bin/perl -w
use strict;

sub format_money{
	my $money = sprintf "%.2f", shift @_;
	1 while $money =~ s/^(-?\d+)(\d{3})/$1,$2/;
	$money =~ s/^(-?)/$1\$/;
	$money;
}
print format_money(1234567890.1234567890);
