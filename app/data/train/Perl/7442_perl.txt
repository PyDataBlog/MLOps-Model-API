# perl p02.pl
use strict;
use warnings;

my @array = ('a', 'b', 'c', 'd');

# method 1: negative subscripts, works for both array and lists
print $array[-2];

# method 2: using $# to count
print $array[$#array - 1];

# method 3: using scalar to evaluate an array in scalar context
print $array[scalar(@array) - 2];
