package IFTTT::CMS::IFTTT;
use strict;
use warnings;

sub profile {
    my $app  = shift;
    my $user = $app->user;
    $app->redirect(
        $app->uri(
            mode => 'view',
            args => {
                _type => 'author',
                id    => $user->id,
            },
        )
    );
}

1;
