sub binize
{
	my $str = shift;

	$str =~ s/^%//;
	pack 'C', hex($str);
}

for my $arg (@ARGV) {
	$arg =~ s/%([[:xdigit:]]{2})/binize($1)/ge;
	print $arg;
	print "\n";
}
