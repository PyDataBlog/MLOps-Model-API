package Org::Export::Mediawiki::Block;


#** @class Org::Export::Mediawiki::Block
#
#
#
# @author Sebastian Rose <sebastian_rose@gmx.de>
#*


use strict;
use utf8;
use 5.14.0;
use Org::Export::Renderer;
use Moose;


with    'Org::Export::Renderer';


sub format
{
    my( $self, $parser ) = @_;
    my $lines = $parser->{lines};

    return "<pre>\n"
        . join( "", @{$lines} )
        . "</pre>\n";
}




__PACKAGE__->meta()->make_immutable();



1;
