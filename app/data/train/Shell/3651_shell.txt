#!/bin/bash

OLD_DIR=`pwd`           # absolute path
BIN_DIR=`dirname $0` # path relative to pwd

if [[ $1 == "build" ]]; then
  cd $BIN_DIR/../cpp-ws/util/Debug
  make clean
  make
  cd $OLD_DIR
  cd $BIN_DIR/../cpp-ws/itchReader/Debug
  make clean
  make
  cd $OLD_DIR
  cd $BIN_DIR/../cpp-ws/oimes/Debug
  make clean
  make
  cd $OLD_DIR
fi

cd $BIN_DIR/../
. ../minetats_env.sh
bin/http-server -p 7998

cd $OLD_DIR
exit $EXIT_CODE
