#!/bin/bash

BASH_PROFILE_CONTENTS_=(
  "export JAVA_HOME=$(/usr/libexec/java_home -v 1.8)"
  "export SCALA_HOME=/usr/local/opt/scala/libexec"
  "export PATH=/usr/local/sbin:$PATH:$GOPATH"
  "export GOPATH=$ENVIRONMENT_GO_PATH_"
  "export CLICOLOR=1"
  "export LSCOLORS=DxFxCxGxBxegedabagaced"
  "export PATH=/usr/local/opt/gpg-agent/bin:$PATH"
  "export EDITOR=/usr/local/bin/nvim"
  'export HOMEBREW_GITHUB_API_TOKEN="{homebrew access token}"'
)

for content in "${BASH_PROFILE_CONTENTS_[@]}"
do
  echo $content >> $BASH_PROFILE_PATH_
done
