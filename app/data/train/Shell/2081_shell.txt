#!/bin/sh

DIR="$(dirname $0)"
for jar in $DIR/../lib/*.jar; do
  CP="$CP:$jar"
done
if [ ! -d "$DIR/../repo/.git" ]; then
	git init "$DIR/../repo/"
fi
nohup /usr/bin/java -cp "$CP" -Djava.library.path="$DIR/../lib" \
    -Dlog4j.configurationFile="$DIR/../config/slog4j.properties" \
kreon.shivaj.server.WebServerMain \
-Dconfig="$DIR/../config/scanner.config" $@ &
