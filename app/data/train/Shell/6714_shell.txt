#!/bin/bash

now=`date "+%d%h%y_%H%M%S"`
cmd="julia train.jl --trainfiles data/nonsynth/src-train.txt data/nonsynth/targ-train.txt --devfiles data/nonsynth/src-val.txt data/nonsynth/targ-val.txt --savefile nn/model_${now}.jld --encvfile data/nonsynth/src.wv --decvfile data/nonsynth/targ.wv --epochs 25 --embedsize_src 256 --embedsize_trg 32 --hiddensize 512 --batchsize 1 --seed 1 --pdrop 0.25 --lr_init 0.03 --lr_decay 0.5 --gclip 5.0 --maxpredlen 45 --nlayer 2 --beamsize 5 --penalize_length --optim_algo Sgd --min_save_acc 0.40"
logfile=logs/train_$now.log

echo "$cmd" |& tee $logfile
eval "$cmd" |& tee -a $logfile
