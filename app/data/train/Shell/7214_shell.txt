#!/usr/bin/env bash

STRING_SYMBOL_FILE="func.list"
HEAD_FILE="$PROJECT_DIR/ObfuscateHeader.h"
export LC_CTYPE=C

ramdomString()
{
openssl rand -base64 64 | tr -cd 'a-zA-Z' |head -c 16
}

rm -f $HEAD_FILE

touch $HEAD_FILE
echo '#ifndef ObfuscateHeader_h
#define ObfuscateHeader_h' >> $HEAD_FILE
echo "//confuse string at `date`" >> $HEAD_FILE
cat "$STRING_SYMBOL_FILE" | while read -ra line; do
if [[ ! -z "$line" ]]; then
ramdom=`ramdomString`
echo $line $ramdom
insertValue $line $ramdom
echo "#define $line $ramdom" >> $HEAD_FILE
fi
done
echo "#endif" >> $HEAD_FILE
