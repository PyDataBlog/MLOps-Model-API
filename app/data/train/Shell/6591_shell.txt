#!/usr/bin/env bash

JAVA_VER=$(java -version 2>&1 | sed 's/java version "\(.*\)\.\(.*\)\..*"/\1\2/; 1q')

if [ ! "$JAVA_VER" -ge 17 ];
then
  echo "Tinymasq requires java version 7, make sure to install correct version exiting now." 
  exit 1
fi

# Loading local environment settings see local.sh 
[[ -f /usr/share/tinymasq/local.sh ]] &&  source /usr/share/tinymasq/local.sh 

java -XX:MaxPermSize=256m -cp "target/*:/etc/tinymasq/" tinymasq.import $@
