#!/bin/sh

BASEDIR=`dirname $0`
. ${BASEDIR}/00_README.sh

cat << NOTICE
*********** ${0##*/} NOTICE **********
Python周りの環境を整えます．
************************************************
NOTICE
echo -n "Continue? [Press Any Key or Ctrl+c to Cancel]: "
read YN

START=`date +%s`

LOGGING "(${0##*/}) python-allをインストールします…"
APTITUDE python-all

LOGGING "(${0##*/}) python-devをインストールします…"
APTITUDE python-dev

LOGGING "(${0##*/}) easy_installをインストールします…"
APTITUDE python-setuptools

LOGGING "(${0##*/}) PIPをインストールします…"
APTITUDE python-pip
pip install --upgrade pip

LOGGING "(${0##*/}) virtualenvをインストールします…"
pip install --upgrade virtualenv

LOGGING "(${0##*/}) bpythonをインストールします…"
APTITUDE bpython

LOGGING "(${0##*/}) ipythonをインストールします…"
APTITUDE ipython

LOGGING "(${0##*/}) vim-pythonをインストールします…"
APTITUDE vim-python

LOGGING "(${0##*/}) Python Imaging Libraryをインストールします…"
APTITUDE python-imaging

END=`date +%s`
TIME=`expr ${END} - ${START}`
LOGGING "${0##*/}が完了しました (処理時間: ${TIME}秒)"
