#!/bin/bash -e

. ../lfs.comm

build_src() {
    version=4.1.1
    srcfil=gawk-$version.tar.xz
    srcdir=gawk-$version

    tar -xf $LFSSRC/$srcfil && cd $srcdir

    ./configure --prefix=/usr
    make $JOBS
    make check
    make install

    mkdir -v /usr/share/doc/gawk-$version
    cp    -v doc/{awkforai.txt,*.{eps,pdf,jpg}} /usr/share/doc/gawk-$version

    cd .. && rm -rf $srcdir
}

build
