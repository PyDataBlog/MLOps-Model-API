!Configuração Roteador RT-01 Rede: 192.168.10.0/24
enable
	configure terminal
		
    	!Configurando o Gateway Padrão para o ISP (Verificar sua rede)
    	ip default-gateway 209.165.201.XXX
    
    	!Configurando a Rota Padrão para o ISP
	ip route 0.0.0.0 0.0.0.0 serial 0/0/1
		
    	!Configurando o Protocolo de Roteamento Dinâmico OSPF (ID Processo Local 10)
    	router ospf 10
	
  		!Configurando a Identificação do Processo do Roteador rodando o OSPF
  		router-id 1.1.1.1
	
  		!Redistribuindo a Rota Padrão e Rotas Estáticas
  		default-information originate
		redistribute static
	
  		!Desativando os anúcios dos Protocolos nas Interfaces
  		passive-interface gigabitEthernet 0/1.10
		passive-interface serial 0/0/1
  
  		!Configuração da referência de largura de banda para o cálculo de custo dos links
		auto-cost reference-bandwidth 10000
  
  		!Registrar todas as alterações de estado dos links
		log-adjacency-changes detail
	
  		!Declarando as redes fisicamente conectadas, utilizando o recurso de Wildcad Mask (máscara coringa)
  		!Configuração na Área 0 (Backbone)
  		network 192.168.10.0 0.0.0.255 area 0
		network 10.0.0.8 0.0.0.3 area 0
		network 10.0.0.4 0.0.0.3 area 0
		network 10.0.0.0 0.0.0.3 area 0
		end
write

!Comandos de Verificação do OSPF

show running-config
show ip protocols
show ip interface brief
show ip route
show ip route ospf 
show ip route ospf 10
show ip ospf neighbor
show ip ospf database
show ip ospf database router
show ip ospf 10
show ip ospf inteface
