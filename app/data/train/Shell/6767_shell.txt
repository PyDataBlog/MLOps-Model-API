#!/bin/bash

./myscript  3>&1 1>stdout.logfile  2>&3-  |  tee -a stderr.logfile

##here first we duplicate file descript 1 into 3. Then we redirect stdoutput to a file. Then we duplicate  file descriptor 3 into stderror, which causes the stderr and stdout to swap places, thus the output of stderr gets piped (like stdout would in fact) to tee so that it can go to the screen and to a file at the same time.  the minus sign here says to close file descriptor 3 after being done.
#remember stdout can be piped using |  but stderro cannot, so we adopted this swapping of file descriptors to enable that.
