#!/bin/bash


go_inside_directory(){
  directory=$1
  indent=$2
  for f in $( ls $1 ); do
    if [ -d "$1/$f" -a ! -L "$1/$f" ]; then
	    echo $( head -c $2 < /dev/zero | tr '\0' '-' ) "+" $f
      go_inside_directory $1/$f $(($2+3))
    elif [ -e "$1/$f" ]; then
      echo $( head -c $2 < /dev/zero | tr '\0' '-' ) ">" $f
    fi
  done
}

go_inside_directory . 3
