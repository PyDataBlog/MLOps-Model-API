project(Snow_Leopard_Parser_Library)

#
# Version information
#
set(SLP_MAJOR_VERSION 0)
set(SLP_MINOR_VERSION 0)
set(SLP_RELEASE 1)
set(SLP_BUILD 1)
set(SLP_VERSION_LABEL "alpha")
set(SLP_RELEASE_DATE "2014-10-11")
set(SLP_PROGRAM_NAME "Snow Leopard Parser Library")
set(SLP_COPYRIGHT_HOLDERS "Roel Sergeant")
set(SLP_COPYRIGHT_YEARS "2014, 2015, 2016")
set(SLP_COPYRIGHT_EMAILS "rsergeant@panix.com")

configure_file(
    "${PROJECT_SOURCE_DIR}/include/sl/p/version.h.in"
    "${PROJECT_SOURCE_DIR}/include/sl/p/version.h"
)

#
# Configuration files
#
set(P_CONFIG_FILES
    include/sl/p/version.h.in
)

#
# Header files
#
set(P_HEADER_FILES
    include/sl/p/debug.h
    include/sl/p/features.h
    include/sl/p/info.h
    include/sl/p/keywords.h
    include/sl/p/lexer.h
    include/sl/p/numeric_literals.h
    include/sl/p/parse_content.h
    include/sl/p/parser.h
    include/sl/p/parser_settings.h
    include/sl/p/punctuations.h
    include/sl/p/reader.h
    include/sl/p/source.h
    include/sl/p/source_position.h
    include/sl/p/token.h
)

#
# Private header files
#
set(P_PRIVATE_HEADER_FILES
    include/sl/p/_lexer.h
    include/sl/p/_parser.h
    include/sl/p/_reader.h
    include/sl/p/version.h
)

#
# Source files
#
set(P_SOURCE_FILES
    src/p/libslp_information.c
    src/p/print_libslp_information.c
    src/p/keywords/keywords.c
    src/p/lexer/advance.c
    src/p/lexer/initialize_lexer.c
    src/p/lexer/match.c
    src/p/parse_content/create_parse_content.c
    src/p/parser/initialize_parser.c
    src/p/parser/parse.c
    src/p/punctuations/punctuations.c
    src/p/reader/get_character.c
    src/p/source_position/no_source_position.c
    src/p/sources_array/sources_array.c
    src/p/token/create_token.c
)

#
# Private source files
#
set(P_PRIVATE_SOURCE_FILES
    src/p/_lexer/_Lex.c
    src/p/_lexer/_Lex_comment_or_divide.c
    src/p/_lexer/_lexer.c
    src/p/_parser/_parser.c
    src/p/_reader/_Get_character_from_file.c
    src/p/_reader/_Get_character_from_string.c
    src/p/_reader/_reader.c
    src/p/_reader/_Set_position.c
)

#
# All source files
#
set(P_ALL_SOURCE_FILES
    ${P_SOURCE_FILES}
    ${P_PRIVATE_SOURCE_FILES}
    ${C_CONFIGURATION_FILES}
    ${P_HEADER_FILES}
    ${P_PRIVATE_HEADER_FILES}
    ${DH_HEADER_FILES}
    ${UTS_HEADER_FILES}
)

#
# Libraries
#
add_library(slp_shared SHARED ${P_ALL_SOURCE_FILES})
set_target_properties(slp_shared PROPERTIES OUTPUT_NAME "slp")

add_library(slp_static STATIC ${P_ALL_SOURCE_FILES})
set_target_properties(slp_static PROPERTIES OUTPUT_NAME "slp")

#
# Script files
#
set(P_SCRIPT_FILES

)

#
# Documentation files
#
set(P_DOCUMENTATION_FILES
    docs/libslp/libslp_reference.md
    docs/libslp/libslp_user_guide.md
)

#
# Set ALL_TARGETS
#
set(ALL_TARGETS ${ALL_TARGETS} slp_shared slp_static)
