project(Snow_Leopard_Diagnostics_Handler_Library)

#
# Version information
#
set(SLDH_MAJOR_VERSION 0)
set(SLDH_MINOR_VERSION 0)
set(SLDH_RELEASE 1)
set(SLDH_BUILD 1)
set(SLDH_VERSION_LABEL "alpha")
set(SLDH_RELEASE_DATE "2015-08-12")
set(SLDH_PROGRAM_NAME "Snow Leopard Parser Library")
set(SLDH_COPYRIGHT_HOLDERS "Roel Sergeant")
set(SLDH_COPYRIGHT_YEARS "2015")
set(SLDH_COPYRIGHT_EMAILS "rsergeant@panix.com")

configure_file(
    "${PROJECT_SOURCE_DIR}/include/sl/dh/version.h.in"
    "${PROJECT_SOURCE_DIR}/include/sl/dh/version.h"
)

#
# Configuration files
#
set(DH_CONFIG_FILES
    include/sl/dh/version.h.in
)

#
# Header files
#
set(DH_HEADER_FILES
    include/sl/dh/diagnostic.h
    include/sl/dh/diagnostic_codes.h
    include/sl/dh/diagnostic_handler.h
    include/sl/dh/diagnostic_information.h
    include/sl/dh/diagnostic_types.h
    include/sl/dh/info.h
)

#
# Private header files
#
set(DH_PRIVATE_HEADER_FILES
    include/sl/dh/_diagnostics_information.h
    include/sl/dh/version.h
)

#
# Source files
#
set(DH_SOURCE_FILES
    src/dh/libsldh_information.c
    src/dh/print_libsldh_information.c
    src/dh/diagnostic/create_diagnostic_object.c
    src/dh/diagnostic/destroy_diagnostics.c
    src/dh/diagnostic/diagnostic.c
    src/dh/diagnostic/initialize_diagnostics.c
    src/dh/diagnostic_codes/diagnostic_codes.c
    src/dh/diagnostic_handler/abort_fatal_error.c
    src/dh/diagnostic_handler/report_error.c
    src/dh/diagnostic_handler/report_fatal_error.c
)

#
# Private source files
#
set(DH_PRIVATE_SOURCE_FILES
    src/dh/_diagnostics_information/_Diagnostics_information.c
)

#
# All source files
#
set(DH_ALL_SOURCE_FILES
    ${DH_SOURCE_FILES}
    ${DH_PRIVATE_SOURCE_FILES}
    ${C_CONFIGURATION_FILES}
    ${DH_HEADER_FILES}
    ${DH_PRIVATE_HEADER_FILES}
    ${UTS_HEADER_FILES}
)

#
# Libraries
#
add_library(sldh_shared SHARED ${DH_ALL_SOURCE_FILES})
set_target_properties(sldh_shared PROPERTIES OUTPUT_NAME "sldh")

add_library(sldh_static STATIC ${DH_ALL_SOURCE_FILES})
set_target_properties(sldh_static PROPERTIES OUTPUT_NAME "sldh")

#
# Script files
#
set(DH_SCRIPT_FILES
)

#
# Documentation files
#
set(DH_DOCUMENTATION_FILES
    docs/libsldh/libsldh_reference.md
    docs/libsldh/libsldh_user_guide.md
)

#
# Set ALL_TARGETS
#
set(ALL_TARGETS ${ALL_TARGETS} sldh_shared sldh_static)
