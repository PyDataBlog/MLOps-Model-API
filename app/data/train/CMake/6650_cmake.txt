IF(LIBXFCONF_LIBRARIES)
   SET(LIBXFCONF_FIND_QUIETLY TRUE)
ENDIF(LIBXFCONF_LIBRARIES)

find_path(LIBXFCONF_INCLUDE_DIR xfconf/xfconf.h PATH_SUFFIXES xfce4/xfconf-0)
find_library(LIBXFCONF_LIBRARY NAMES xfconf-0)
set(LIBXFCONF_LIBRARIES "${LIBXFCONF_LIBRARY}")

include(FindPackageHandleStandardArgs)
find_package_handle_standard_args(
    Libxfconf
    DEFAULT_MSG
    LIBXFCONF_LIBRARIES
    LIBXFCONF_INCLUDE_DIR)
mark_as_advanced(LIBXFCONF_INCLUDE_DIR LIBXFCONF_LIBRARIES)
