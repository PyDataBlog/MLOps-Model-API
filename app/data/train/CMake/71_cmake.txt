# http://www.cmake.org/Wiki/CMake_Cross_Compiling
# this one is important
SET(CMAKE_SYSTEM_NAME Linux)
#this one not so much
SET(CMAKE_SYSTEM_VERSION 1)

# specify the cross compiler
SET(CMAKE_C_COMPILER   /usr/lib64/madde/linux-x86_64/targets/meego-core-ia32-1.2.0/bin/gcc)
SET(CMAKE_CXX_COMPILER /usr/lib64/madde/linux-x86_64/targets/meego-core-ia32-1.2.0/bin/g++)

# where is the target environment 
SET(CMAKE_FIND_ROOT_PATH /usr/include/bluetooth/ /usr/lib64/madde/linux-x86_64/sysroots/kp-sysroot/)

# search for programs in the build host directories
SET(CMAKE_FIND_ROOT_PATH_MODE_PROGRAM NEVER)
# for libraries and headers in the target directories
SET(CMAKE_FIND_ROOT_PATH_MODE_LIBRARY ONLY)
SET(CMAKE_FIND_ROOT_PATH_MODE_INCLUDE ONLY)
