\section{需求分析}
	\begin{frame}
		\center{\Huge{需求分析}}
	\end{frame}
	
	\begin{frame}
		\begin{block}{平面连杆机构应用广泛}
			连杆机构构件运动形式多样，具有单位面积所受压力较小、制造方便等优点，广泛应用于各种机械、仪表和机电产品中。
		\end{block}
		
		\begin{figure}[!hbtp]
		\centering
  		\includegraphics[scale=0.2]{1.png}
  		\caption{飞机起落架中的连杆机构}
		\end{figure}
	\end{frame}
	
	\begin{frame}
		\begin{block}{现有软件的不足}
			虽然市面上一般的设计软件也可以简单仿真连杆机构运动过程，但是并不能将整个传动过程速度、加速度、位移等量随时间的变化体现出来。
		\end{block}
		

	\end{frame}

		\end{figure}
	\end{frame}

	\begin{frame}
		\frametitle{基础类}
		\begin{columns}[c]

		\column{.45\textwidth} 
		\begin{itemize}
			\item Ponit类和Rod类，分别描述节点和连杆，这两部分是一个杆组的基本组成元素
			\item 两个节点一个连杆组成一级杆组，三个节点两个连杆组成2级杆组，以此类推
		\end{itemize}
		
		\column{.5\textwidth} 
		\begin{figure}[!hbtp]
		\centering
  		\includegraphics[scale=0.5]{basic}
		\end{figure}
		\end{columns}
	\end{frame}

	\begin{frame}
		\frametitle{杆组类}
		\begin{columns}[c]

		\column{.45\textwidth} 
		\begin{itemize}
			\item 根据实际需要，工程中常用的是一级杆组（SR）、二级杆组（RRR、RRP）
			\item 这三类杆组都是杆组类（RodGroup）的子类，并在RodGroup类中设置int型静态参数（SR、RRR、RRP）用于记录各子类实体的数量
		\end{itemize}
		
		\column{.5\textwidth} 
		\begin{figure}[!hbtp]
		\centering
  		\includegraphics[scale=0.2]{rod}
		\end{figure}
		\end{columns}
	\end{frame}

	\begin{frame}
		\frametitle{界面类}
		\begin{columns}[c]

		\column{.45\textwidth} 
		\begin{itemize}
			\item MainFrame类，主界面类
			\item 包含主要的按钮、菜单和工作区，通过集合ArrayList记录操作和添加的杆组，计算函数嵌入在各个按钮的响应函数中
