\section{Trigonometrische Funktionen}
\begin{enumerate}
\item $\sin(x) = \sum\limits_{k = 0}^{\infty} \frac{(-1)^k}{(2k + 1)!} x^{2k + 1}$ (Fourier sagt \underline{ungerade} Funktionen)
\item $\cos(x)$ analog f\"ur gerade Funktionen ($2k$ statt $2k + 1$)
\item $\sinh(x)$ und $\cosh(x)$ analog, ohne den $(-1)^k$ Teil
\item Formel von De Moivre $(\cos(x) + i\sin(x))^n = \cos(nx) + i\sin(nx)$ (Because Euler's formula)
\item $cos(a \pm b) = cos(a)cos(b) \mp sin(a)sin(b)$ (folgt aus De Moivre, Vergleich Real/Imagin\"arteil)
\item $sin(a \pm b) = sin(a)cos(b) \pm sin(b)cos(b)$ (folgt aus De Moivre)
\end{enumerate}


\section{Hyperbolische Funktionen}
\begin{enumerate}
\item $\sinh(x) = \sum\limits_{k = 0}^{\infty} \frac{1}{(2k + 1)!} x^{2k + 1}$ (Fourier sagt \underline{ungerade} Funktionen)
\item $\cosh(x)$ analog f\"ur gerade Funktionen ($2k$ statt $2k + 1$)
\item $\cosh(x)^2 - \sinh(x)^2 = 1$
\item $\cosh(x) + \sinh(x) = e^x$
\item $\cosh(x) - \sinh(x) = e^{-x}$
\item $\sinh(x) = \frac{e^x - e^{-x}}{2}$
\item $\cosh(x) = \frac{e^x + e^{-x}}{2}$
\end{enumerate}
