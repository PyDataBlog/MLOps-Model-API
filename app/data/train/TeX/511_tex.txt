\begin{questions}
\item Is $f(x) = \frac{4x^2 - 23x + 15}{4x-3}$ continuous at the point $x = 1$? 
At the point $x = \frac{3}{4}$? 
What is its limit as $x \rightarrow \frac{3}{4}$?
\item Consider the function $f: [0,4] \rightarrow \R$ defined by 
\begin{equation*}
f(x) = 
\begin{cases}
4 - 2x \text{ if } x < 2 \\
2x - 4 \text{ if } x \geq 2
\end{cases}
\end{equation*}
\begin{itemize}
\item[a.] Is $f$ continuous at $x = 2$?
\item[b.] Is $f$ left-continuous at $x = 4$? 
\item[c.] Is $f$ right-continuous at $x = 0$? 
\item[d.] Where is $g(x) = \frac{f(x)}{x}$ continuous on $[0,4]$? 
\end{itemize}
\item Prove the (rather sensible) property of the sine and cosine functions that 
\begin{equation*}
\lim_{\theta \rightarrow a} \cos \theta = \cos a \qquad \lim_{\theta \rightarrow a} \sin \theta = \sin a
\end{equation*}
You will need to use the exponential representations of sine and cosine. 
\item We define the \textit{hyperbolic} functions $\cosh \theta$ and $\sinh \theta$ using exponential identities analogous to those for the sine and cosine:
\begin{equation*}
e^x = \cosh x + \sinh x \qquad e^{-x} = \cosh x - \sinh x
\end{equation*}
\begin{itemize}
	\item[a.] Find exponential representations of $\sinh x$ and $\cosh x$. 
	\item[b.] Using the representations you find, along with the analogous definition $\tanh x = \frac{\sinh x}{\cosh x}$, find the exponential representation of $\tanh x$. 
	\item[c.] Find $\lim_{x \rightarrow \infty} \sinh x$ and $\lim_{x \rightarrow \infty} \cosh x$, if they exist. 
	\item[d.] Find $\lim_{x \rightarrow \infty} \tanh x$. 
	You will need to perform some algebra. 
	\item[e.] Using the exponential representations you find, prove the lovely identities $\cosh(ix) = \cos x$ and $\sinh(ix) = i\sin x$, where $i \equiv \sqrt{-1}$. 
	This allows us to rewrite our familiar formula $e^{i \theta} = \cos \theta + i \sin \theta$ as $e^{i \theta} = \cosh(i\theta) + \sinh(i \theta)$. 
\end{itemize}
\item Let $f: \R \rightarrow \R$ be defined 
\begin{equation*}
f(x) = 
\begin{cases}
x^2 \text{ if } x < 1 \\
x - 1 \text{ if } x \geq 1
\end{cases}
\end{equation*}
Is $f$ continuous on $(-\infty, 0)$? 
On $(-\infty, 1)$? 
\end{questions}
