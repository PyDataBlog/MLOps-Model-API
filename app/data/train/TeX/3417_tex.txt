\chapter*{PRIMEIRA AVALIAÇÃO}
\label{1aval}
\vspace{-1.5cm}

\begin{enumerate}
 \item Determine os pulsos de fase máxima e fase nula que tem o mesmo espectro que o pulso $w_{t} = (9,-6,1)$.
 \item Mostre que a transformada de Radon Linear inversa,
 \begin{center}
 $
 \displaystyle f(t,X) = \int S(\tau=t-pX,p)dp
 $
 \end{center}
 mapeia um evento com moveout elíptico no domínio $\tau-p$
 \begin{center}
 $
 S(\tau,p) = w(\tau-T_{0}\sqrt{1-p^{2}v^{2}}),
 $
 \end{center}
 em que w($\tau$) representa um pulso, em um evento com moveout hiperbólico no domínio $t-X$.
\item Considere a sequência de reflexões múltiplas associadas a um refletor plano, cujo moveout é
\begin{center}
 $
 \displaystyle t_{n}^{2}(X)= (n\tau_{0})^{2}+\frac{X^{2}}{V^{2}}
 $
\end{center}
(a) Mostre que estes eventos são periódicos no domínio $\tau-p$.\\
(b) Qual a importância deste resultado para a supressão de multiplas em dados pré-empilhados.
\item Considere um evento linear registrado com aliasing espacial, com vagarosidade aparente
\begin{center}
$
\displaystyle p=2\frac{K_{N}}{w_{N}}
$
\end{center}
Em que $K_{N}$ e $w_{N}$ indicam o número de onda e a frequência de Nyquist, respectivamente. Determine as coordenadas em que este evento intercepta 
os eixos de número de onda e frequência.
\item Considere a transformada de Radon Parabólica inversa 
\begin{center}
$
\displaystyle f(t,X)=\int S(\tau=t-qX^{2},q)dq
$
\end{center}
(a) Mostre que esta expressão pode ser escrita no domínio da frequência na forma 
\begin{center}
 $
 \displaystyle F(w,X)=\int S(w,q)exp(iwqX^{2})dq
 $
\end{center}
em que $F(X,w)$ e $S(q,w)$ representam a transformada de Fourier de $f(t,X)$ e $S(\tau,q)$, respectivamente.\\
(b) Descreva uma aplicação desta transformada no processamento de dados sísmicos.

\item Descreva uma aplicação da filtragem no domínio $f-k$ no processamento de dados sísmicos.
\end{enumerate}
