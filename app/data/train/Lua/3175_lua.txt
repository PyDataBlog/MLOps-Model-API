-- Cria um widget de ACPI
batterywidget = ({ type = "textbox" })
batterywidget.text = " | Bateria | "
batterywidgettimer = timer( { timeout= 5 } )
batterywidgettimer:add_signal( "timeout",
			       function()
				fh = assert( io.popen( "acpi | cut -d, -f2,3 -", "r" ))
				batterywidget.text = " |" .. fh.read( "*1" ) .. " | "
				fh:close()
				end
)

batterywidgettimer:start()
