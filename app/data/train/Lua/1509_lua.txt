local class = require "lib/middleclass"
Color = class("Color")

function Color:initialize(red, green, blue, alpha)
	self.red = red or 255
	self.green = green or 255
	self.blue = blue or 255
	self.alpha = alpha or 255
end

function Color.static:load(table)
	local red = table["colorRed"]
	local green = table["colorGreen"]
	local blue = table["colorBlue"]
	local alpha = table["colorAlpha"]

	return Color:new(red, green, blue, alpha)
end

function Color:save()
	local out = {}
	table["colorRed"] = self.r
	table["colorGreen"] = self.g
	table["colorBlue"] = self.b
	table["colorAlpha"] = self.a

	return out
end
