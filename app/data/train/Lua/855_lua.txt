--TB2-003 Heroic Duo Son Gohan
local ds=require "expansions.utility_dbscg"
local scard,sid=ds.GetID()
function scard.initial_effect(c)
	ds.EnableBattleAttribute(c)
	ds.AddSetcode(c,CHARACTER_SON_GOHAN_ADOLESCENCE,SPECIAL_TRAIT_SAIYAN,CHARACTER_INCLUDES_SON_GOHAN,SPECIAL_TRAIT_WORLD_TOURNAMENT,SPECIAL_TRAIT_EARTHLING)
	ds.AddPlayProcedure(c,COLOR_RED,1,1)
	--ko
	ds.AddSingleAutoPlay(c,0,nil,scard.kotg,scard.koop,DS_EFFECT_FLAG_CARD_CHOOSE)
end
scard.dragon_ball_super_card=true
scard.combo_cost=0
scard.kotg=ds.ChooseCardFunction(PLAYER_PLAYER,ds.BattleAreaFilter(Card.IsPowerBelow),0,DS_LOCATION_BATTLE,0,1,DS_HINTMSG_KO,nil,10000)
scard.koop=ds.ChooseKOOperation
