--BT2-093 Terrible Creator Android 20
local ds=require "expansions.utility_dbscg"
local scard,sid=ds.GetID()
function scard.initial_effect(c)
	ds.EnableBattleAttribute(c)
	ds.AddSetcode(c,SPECIAL_TRAIT_ANDROID,NAME_INCLUDES_ANDROID,CHARACTER_ANDROID_20)
	ds.AddPlayProcedure(c,COLOR_GREEN,1,1)
	--search
	ds.AddSingleAutoPlay(c,0,nil,scard.thtg,scard.thop,DS_EFFECT_FLAG_CARD_CHOOSE)
end
scard.dragon_ball_super_card=true
scard.combo_cost=0
function scard.thfilter(c)
	return c:IsBattle() and c:IsNameIncludes(NAME_INCLUDES_ANDROID) and not c:IsCharacter(CHARACTER_ANDROID_20)
end
scard.thtg=ds.ChooseCardFunction(PLAYER_PLAYER,scard.thfilter,LOCATION_DECK,0,0,1,DS_HINTMSG_TOHAND)
scard.thop=ds.ChooseSendtoHandOperation()
