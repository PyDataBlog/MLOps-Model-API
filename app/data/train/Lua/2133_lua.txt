ruleFunc = function(cell)
    local sum;
    local west_nb_state = 0
    local east_nb_state = 0
    local west_nb = cell:getNb(direction.W)
    local east_nb = cell:getNb(direction.E)

    if west_nb ~= nil then
        west_nb_state = west_nb:getState()
    end
    if east_nb ~= nil then
        east_nb_state = east_nb:getState()
    end

    sum = west_nb_state
    sum = sum + cell:getState()
    sum = sum + east_nb_state

    if sum == 0 or sum == 3 or sum == 1 and west_nb_state == 1 then
        cell:setNewState(0)
    else
        cell:setNewState(1)
    end
end
