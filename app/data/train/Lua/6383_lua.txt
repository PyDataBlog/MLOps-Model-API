--BT3-122 Time's Judgement
local ds=require "expansions.utility_dbscg"
local scard,sid=ds.GetID()
function scard.initial_effect(c)
	--to hand
	ds.AddCounterAttackSkill(c,0,scard.thop,scard.thcost)
end
scard.dragon_ball_super_card=true
scard.thcost=ds.PayEnergyCost(COLOR_COLORLESS,0,1)
function scard.thop(e,tp,eg,ep,ev,re,r,rp)
	Duel.NegateAttack()
	local g=Duel.GetMatchingGroup(nil,tp,LOCATION_HAND,0,nil)
	if g:GetCount()<2 or not Duel.SelectYesNo(tp,aux.Stringid(sid,1)) then return end
	Duel.BreakEffect()
	Duel.Hint(HINT_SELECTMSG,tp,DS_HINTMSG_TODROP)
	local sg=g:Select(tp,2,2,nil)
	if Duel.SendtoDrop(sg,DS_REASON_SKILL)==0 then return end
	local c=e:GetHandler()
	Duel.SendtoHand(c,PLAYER_OWNER,DS_REASON_SKILL)
	Duel.ConfirmCards(1-c:GetControler(),c)
end
