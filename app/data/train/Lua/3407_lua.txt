--TB2-041 Double Impact Krillin
local ds=require "expansions.utility_dbscg"
local scard,sid=ds.GetID()
function scard.initial_effect(c)
	ds.EnableBattleAttribute(c)
	ds.AddSetcode(c,CHARACTER_KRILLIN,SPECIAL_TRAIT_WORLD_TOURNAMENT,SPECIAL_TRAIT_EARTHLING)
	ds.AddPlayProcedure(c,COLOR_GREEN,2,0)
	--ko
	ds.AddSingleAutoPlay(c,0,nil,scard.kotg,scard.koop,DS_EFFECT_FLAG_CARD_CHOOSE+DS_EFFECT_FLAG_IGNORE_BARRIER)
end
scard.dragon_ball_super_card=true
scard.combo_cost=0
scard.kotg=ds.ChooseCardFunction(PLAYER_PLAYER,ds.BattleAreaFilter(Card.IsEnergyBelow),0,DS_LOCATION_BATTLE,0,1,DS_HINTMSG_KO,nil,2)
scard.koop=ds.ChooseKOOperation
