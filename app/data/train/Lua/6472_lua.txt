sel = 0

function drawtile(collision, x, y)
  collision[y][x] = sel
end

function tilepicker(k)
  if k == "1" then
    sel = 1
  elseif k == "2" then
    sel = 0
  end
end

function tilechanger(button)
    if button == 1 then
      x = (love.mouse.getX() + camera.x) / 64 + 1
      x = math.floor(x)
      y = (love.mouse.getY() + camera.y) / 64 + 1
      y = math.floor(y)
      drawtile(collision, x, y)
    end
end
