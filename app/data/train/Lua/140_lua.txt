--TB1-011 Cabba, Universe Mediator
local ds=require "expansions.utility_dbscg"
local scard,sid=ds.GetID()
function scard.initial_effect(c)
	ds.EnableBattleAttribute(c)
	ds.AddSetcode(c,SPECIAL_TRAIT_ALIEN,CHARACTER_CABBA,SPECIAL_TRAIT_SAIYAN,SPECIAL_TRAIT_UNIVERSE_6,SPECIAL_TRAIT_INCLUDES_UNIVERSE)
	ds.AddPlayProcedure(c,COLOR_RED,1,0)
	--to hand
	ds.AddSingleAutoPlay(c,0,nil,scard.thtg,scard.thop,DS_EFFECT_FLAG_CARD_CHOOSE)
end
scard.dragon_ball_super_card=true
scard.combo_cost=0
scard.thtg=ds.ChooseDecktopTarget(Card.IsSpecialTrait,3,0,1,DS_HINTMSG_TOHAND,nil,SPECIAL_TRAIT_UNIVERSE_6)
scard.thop=ds.ChooseDecktopSendtoHandOperation(3,DECK_SEQUENCE_SHUFFLE)
