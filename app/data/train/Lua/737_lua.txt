--P-005_PR Light of Hope Trunks
local ds=require "expansions.utility_dbscg"
local scard,sid=ds.GetID()
function scard.initial_effect(c)
	ds.EnableBattleAttribute(c)
	ds.AddSetcode(c,CHARACTER_TRUNKS_FUTURE,SPECIAL_TRAIT_SAIYAN,SPECIAL_TRAIT_EARTHLING)
	ds.AddPlayProcedure(c,COLOR_BLUE,2,1)
	--power up
	ds.AddPermanentUpdatePower(c,5000,scard.powcon,DS_LOCATION_BATTLE,0,scard.powtg)
end
scard.dragon_ball_super_card=true
scard.combo_cost=0
function scard.powcon(e)
	local tp=e:GetHandlerPlayer()
	return Duel.GetTurnPlayer()==tp and Duel.IsExistingMatchingCard(ds.EnergyAreaFilter(),tp,DS_LOCATION_ENERGY,0,5,nil)
end
function scard.powtg(e,c)
	return c:IsFaceup() and (c:IsLeader() or c==e:GetHandler())
end
