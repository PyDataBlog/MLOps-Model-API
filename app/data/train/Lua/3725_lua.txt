require "src/module/json_util"
require "src/module/file_util"

local class = require "lib/middleclass"
local LanguageClass = class("LanguageClass")

--PT-BR EN-US ES-MX IT-IT FR-FR

function LanguageClass:initialize()

end

function LanguageClass:load(language)
	self._words = JSONUtil:read("langs/" .. language .. ".json")
end

function LanguageClass:languagesAvailable()
	local list = {}
	local langs = FileUtil:getFiles("langs/")
	for i,v in ipairs(langs) do
		list[i] = v
	end
	return list
end

Language = LanguageClass:new()
