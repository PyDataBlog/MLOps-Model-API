--TB2-048 Mercenary Tao, Trading Moves
local ds=require "expansions.utility_dbscg"
local scard,sid=ds.GetID()
function scard.initial_effect(c)
	ds.EnableBattleAttribute(c)
	ds.AddSetcode(c,SPECIAL_TRAIT_WORLD_TOURNAMENT,SPECIAL_TRAIT_EARTHLING)
	ds.AddPlayProcedure(c,COLOR_GREEN,2,2)
	--double strike
	ds.EnableDoubleStrike(c)
	--to drop
	ds.AddSingleAutoPlay(c,0,nil,scard.tgtg,scard.tgop,DS_EFFECT_FLAG_CARD_CHOOSE,scard.tgcon)
end
scard.dragon_ball_super_card=true
scard.combo_cost=0
function scard.lfilter(c)
	return c:IsColor(COLOR_GREEN) and c:IsSpecialTrait(SPECIAL_TRAIT_WORLD_TOURNAMENT)
end
function scard.tgcon(e,tp,eg,ep,ev,re,r,rp)
	return ds.ldscon(scard.lfilter)(e,tp,eg,ep,ev,re,r,rp)
		and Duel.IsExistingMatchingCard(ds.BattleAreaFilter(Card.IsCode),tp,DS_LOCATION_BATTLE,0,1,nil,CARD_TIEN_SHINHAN_TRADING_MOVES)
end
scard.tgtg=ds.ChooseCardFunction(PLAYER_OPPONENT,nil,0,LOCATION_HAND,3,3,DS_HINTMSG_TODROP)
scard.tgop=ds.ChooseSendtoDropOperation
