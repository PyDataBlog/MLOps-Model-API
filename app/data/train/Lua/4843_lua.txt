--Beliqua the Ascender
local ka=require "expansions.utility_ktcg"
local scard,sid=ka.GetID()
function scard.initial_effect(c)
	ka.EnableCreatureAttribute(c)
	--tap
	ka.AddSingleAttackTriggerAbility(c,0,nil,scard.postg,scard.posop,EFFECT_FLAG_CARD_TARGET)
end
scard.kaijudo_card=true
function scard.posfilter(c)
	return c:IsFaceupUntapped() and c:IsLevelBelow(3)
end
scard.postg=ka.TargetCardFunction(PLAYER_PLAYER,scard.posfilter,0,KA_LOCATION_BATTLE,1,1,KA_HINTMSG_TAP)
scard.posop=ka.TargetTapUntapOperation(POS_FACEUP_TAPPED)
