--TB1-080 Ally of Justice Toppo
local ds=require "expansions.utility_dbscg"
local scard,sid=ds.GetID()
function scard.initial_effect(c)
	ds.EnableBattleAttribute(c)
	ds.AddSetcode(c,SPECIAL_TRAIT_ALIEN,SPECIAL_TRAIT_UNIVERSE_11,SPECIAL_TRAIT_INCLUDES_UNIVERSE,CHARACTER_TOPPO)
	ds.AddPlayProcedure(c,COLOR_YELLOW,3,1)
	--play
	ds.AddSingleAutoPlay(c,0,nil,scard.pltg,scard.plop,nil,scard.plcon)
end
scard.dragon_ball_super_card=true
scard.combo_cost=1
scard.plcon=ds.ldscon(aux.FilterBoolFunction(Card.IsSpecialTrait,SPECIAL_TRAIT_UNIVERSE_11))
function scard.plfilter1(c)
	return c:IsEnergyBelow(3) and c:IsSpecialTrait(SPECIAL_TRAIT_UNIVERSE_11)
end
function scard.plfilter2(c,e)
	return scard.plfilter1(c) and c:IsCanBeSkillTarget(e)
end
function scard.pltg(e,tp,eg,ep,ev,re,r,rp,chk,chkc)
	if chkc then return chkc:IsLocation(LOCATION_DECK) and chkc:IsControler(tp) and scard.plfilter1(chkc) end
	if chk==0 then return true end
	Duel.Hint(HINT_OPSELECTED,1-tp,e:GetDescription())
	local ct=Duel.GetLocationCount(tp,DS_LOCATION_BZONE)
	if ct>2 then ct=2 end
	local g=Duel.GetDecktopGroup(tp,10)
	Duel.ConfirmCards(tp,g)
	Duel.Hint(HINT_SELECTMSG,tp,DS_HINTMSG_PLAY)
	local sg=g:FilterSelect(tp,scard.plfilter2,0,ct,nil,e)
	if sg:GetCount()>0 then
		Duel.SetTargetCard(sg)
	else Duel.ShuffleDeck(tp) end
end
scard.plop=ds.ChoosePlayOperation(DS_POS_FACEUP_REST)
