local Player = require "player"
local MediumAI = {}
setmetatable(MediumAI,
  { __call = function(self) return MediumAI:new() end })

function MediumAI:new()
  local ai = Player("Medium AI", "ai")
  local dictionary = require "dictionary"
  
  ai.selectWord = function (self)
    --return a random word out of the ones we can make
    local words = dictionary.getWordsFromLetters(self.getCurrentTiles())
    return words[math.random(#words)]
  end
  
  return ai
end

return MediumAI
