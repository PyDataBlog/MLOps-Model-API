import Base: /


immutable Dimensions
    width::Int64
    height::Int64
    Dimensions(width::Number, height::Number) = new(floor(Int64, width), floor(Int64, height))
end

immutable Point
    x::Int64
    y::Int64
    Point(x::Number, y::Number) = new(floor(Int64, x), floor(Int64, y))
end

immutable Rectangle
    point::Point
    dimensions::Dimensions
end

function aspect_ratio(dimensions::Dimensions)
    dimensions.width / dimensions.height
end

function /(dimensions::Dimensions, scale::Number)
    Dimensions(dimensions.width / scale,
               dimensions.height / scale)
end
