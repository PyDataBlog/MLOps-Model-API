type SGDParams
	## Size of minibatches
	minibatch_size::Int

	## Learning Rate
	lr::Float64 # learning rate
	decay::Float64 # learning rate decay

	## Momentum
	momentum::Float64 # momentum scalar
	v::AbstractArray # momentum direction
end

function SGDParams(minibatch_size;
				   lr::Float64=1e-3,
	               decay::Float64=0.0,
	               momentum::Float64=0.0,
	               v::AbstractArray=zeros(minibatch_size))
	SGDParams(minibatch_size,lr,decay,momentum,v)
end
