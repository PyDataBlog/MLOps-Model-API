# Box Constraint, quadratic program
f(x) = sum(x.^2)
g = ProxAlgs.BoxConstraint(ones(3),2*ones(3))
result = ProxAlgs.minimize([f,g],randn(3),method=:apd)
@test vecnorm(result.x - 1) < 1e-6
