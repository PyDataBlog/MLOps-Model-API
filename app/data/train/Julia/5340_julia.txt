#!/usr/bin/env julia
_REPORT  = ARGS[2]
_INFILE  = ARGS[1]
_OUTFILE = ARGS[3]
######################################################

function ConvertFQ2FA()
  FIN = open(_INFILE , "r")
  FOUT= open(_OUTFILE, "w")
  iterator = 0
	while !eof(FIN)
		line = readline(FIN)
		if ismatch(r"^[ACGTUacgtu]+$", line)
      iterator += 1
      write(FOUT, ">seq" * string(iterator) * "-\n")
      write(FOUT, line)
    end
	end
	close(FIN)
  close(FOUT)
  FREP = open(_REPORT, "w")
  write(FREP, "Converted "*string(iterator)*" sequences\n")
  close(FREP)
end

ConvertFQ2FA()
