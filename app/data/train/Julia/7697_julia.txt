module MyUtil

struct Sellmeier
    a::AbstractFloat
    b::AbstractFloat
    c::AbstractFloat
    d::AbstractFloat
end

struct Crystal
    sellmeier_o::Sellmeier
    sellmeier_e::Sellmeier
    lambda_ff::Real
    n_o::Float32
    n_e::Float32
    dn_o::Float32
    dn_e::Float32
end

struct System
    temp::Float32
    c::Float32
    


function get_refractive_index(lambda::Real, sellmeier::Sellmeier, t::Real=0)::Float32
    n = sellmeier.D * lambda ^ 2
    n += sellmeier.A
    n += sellmeier.B / ((lambda ^ 2) + sellmeier.C)
    n = sqrt(n)
    n += T * sellmeier.D
    return n
end

end
