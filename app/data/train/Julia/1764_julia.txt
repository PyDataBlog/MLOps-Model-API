using RLESUtils, RunUtils, IFTTTUtils

const CONFIG = "flat"
const DATA = "nvn_libcas098smallfilt_10K"
const COMP = gethostname()

function template_gp(i)
    """
    using GrammarExpts,ACASX_GP
    config=configure(ACASX_GP, "$CONFIG", "$DATA")
    acasx_gp(; seed=$i, outdir=Pkg.dir("GrammarExpts/results/ACASX_GP_$(CONFIG)_$(DATA)_seed$i"), config...)
    """
end

notifydone() = sendifttt(;value1="$COMP done")

A = JuliaSource[]
append!(A, [JuliaSource(template_gp(i)) for i=1:10])
