use super::mmu;
use super::cpu;

#[derive(Debug)]
pub struct GameBoyAdvance {
    pub cpu: cpu::Cpu,
    mmu: mmu::Mmu
    // audio
    // graphics
    // network
}

impl GameBoyAdvance {
    pub fn new(biosROM: Box<[u8]>, gameROM: Box<[u8]>) -> GameBoyAdvance {
        GameBoyAdvance {
            cpu: cpu::Cpu::new(),
            mmu: mmu::Mmu::new(biosROM, gameROM)
        }
    }

    pub fn emulate(&mut self) {
        self.cpu.run(&mut self.mmu)
    }
}
