/// Enum for symmetric matrix inputs.
#[repr(u8)]
#[derive(Clone, Copy, PartialEq, Eq)]
pub enum Symmetric {
    /// Read elements from the upper-triangular portion of the matrix
    Upper = b'U',

    /// Read elements from the lower-triangular portion of the matrix
    Lower = b'L',
}
