extern crate num;

use self::num::traits::Num;
use std::string::ToString;

pub trait Digits {
  fn digits(self) -> Vec<u32>;
}

impl<T> Digits for T where T: Num + ToString {
  fn digits(self) -> Vec<u32> {
    self.to_string().chars().filter(|c| c.is_numeric()).map(|c| c.to_digit(10).unwrap()).collect()
  }
}
