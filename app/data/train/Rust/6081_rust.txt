extern crate rusthub;

use rusthub::oauth_web;

    let result = oauth_web::capture_authorization(
        "aaaaaaaaaxaaaaaaaaax".to_string(),
        "aaaaaaaaaxaaaaaaaaaxaaaaaaaaaxaaaaaaaaax".to_string(),
        60

    match result {

}
