
!		Subroutine per la definizione della matrice Hermitiana
!		secondo la funzione specificata

		subroutine matrix(lev_mat,k)
		external f
		complex*16 lev_mat(2,2),f
		real*16 a,k(2),gamma0
		
		gamma0 = 2._16 !elettronVolt
!		Costante reticolare del grafene
		a = 2.46 !Angstrom

!		Definizione matrice 2X2
!		Entrate non nulle
		lev_mat(1,2) = gamma0 * f(k)
		lev_mat(2,1) = conjg( gamma0 * f(k) )
		
		return
		end
