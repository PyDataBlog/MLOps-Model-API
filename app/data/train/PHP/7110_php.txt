<?php if ( ! defined('BASEPATH')) exit('No direct script access allowed');

class Background_model extends CI_Model {

	function __construct(){
		// Call the Model constructor
		parent::__construct();
		$this->load->model('helper/helper_model');
	}

	public function getMapcount()
	{
		return $this->db->query("SELECT COUNT(background_id) as cnt from layout_background_mapping")->result_array();
	}

	public function maplisting($limit)
	{
		return $this->db->query("SELECT bm.*,c.cust_first_name, c.cust_last_name, c.cust_id, l.layout_id, l.layout_name FROM layout_background_mapping bm, customer_master c, layout_table l WHERE bm.cust_id=c.cust_id AND bm.layout_id=l.layout_id LIMIT $limit")->result_array();
	}

}
