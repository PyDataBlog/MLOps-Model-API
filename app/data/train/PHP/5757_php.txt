if ( ! function_exists('tgl_indo'))
{

	function dateinsert($data)
	{
		$garing="/";
		$cek=strpos($data,$garing);
		if($cek!=null)
		{
			$x=explode("/", $data);
			$date="$x[2]-$x[0]-$x[1]";
			return $date;
		}
		else
		{
			return $data;
		}
		
	}
}
