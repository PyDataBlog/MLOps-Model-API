namespace HkApps\App\Http\ViewComposers;

use HkApps\App\Support\RequestTrait;
use Illuminate\Contracts\View\View;

/**
 * Class ViewComposer.
 *
 * @package HkApps\App\Http\ViewComposers
 */
abstract class ViewComposer implements ViewComposerInterface
{
    use RequestTrait;

    /**
     * ViewComposer constructor.
     */
    public function __construct()
    {
    }

    /**
     * Bind data to the view.
     *
     * @param View $view
     */
    abstract public function compose(View $view);
}
