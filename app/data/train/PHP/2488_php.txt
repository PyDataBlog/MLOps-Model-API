namespace AntiMattr\Bundle\ContentMongoDBBundle\Form\Document;

use Symfony\Component\Form\AbstractType;
use Symfony\Component\OptionsResolver\OptionsResolverInterface;
use Symfony\Component\Form\FormBuilderInterface;

class PageSectionFormType extends AbstractType
{
    public function buildForm(FormBuilderInterface $builder, array $options)
    {
        $builder
            ->add('position', 'integer', [
                'attr' => [
                    'input_group' => [
                        'prepend' => 'Order',
                        'size' => 'small',
                    ],
                ],
                'label' => false,
            ])
            ->add('slug', 'text', [
                'attr' => [
                    'input_group' => [
                        'prepend' => 'Slug',
                        'size' => 'small',
                    ],
                ],
                'label' => false,
            ])
            ->add('title', 'text', [
                'attr' => [
                    'input_group' => [
                        'prepend' => 'Subtitle',
                        'size' => 'small',
                    ],
                ],
                'label' => false,
            ])
            ->add('description', 'textarea', [
                'label' => false,
            ])
        ;
    }

    public function setDefaultOptions(OptionsResolverInterface $resolver)
    {
        $resolver->setDefaults([
            'data_class' => 'AntiMattr\Bundle\ContentMongoDBBundle\Document\Page\Section',
        ]);
    }

    public function getName()
    {
        return 'content_page_section';
    }
}
