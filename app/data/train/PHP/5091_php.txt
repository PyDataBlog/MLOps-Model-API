/**
 * Smarty capitalize modifier plugin
 * 
 * Type:     modifier<br>
 * Name:     capitalize<br>
 * Purpose:  capitalize words in the string
 * 
 * @link 
 * @author Monte Ohrt <monte at ohrt dot com> 
 * @param string $ 
 * @return string 
 */
function smarty_modifier_hours($minutos)
{ 
	return str_pad((integer)($minutos/60), 2, "0", STR_PAD_LEFT).":".str_pad($minutos%60, 2, "0", STR_PAD_LEFT);
}
