<?php
namespace App\Controller;

use App\Controller\AppController;

/**
 * BoardsMembers Controller
 *
 * @property \App\Model\Table\BoardsMembersTable $BoardsMembers
 */
class BoardsMembersController extends AppController
{

    /**
     * Index method
     *
     * @return void
     */
    public function index()
    {
        $this->paginate = [
            'contain' => ['Boards', 'Members']
        ];
        $this->set('boardsMembers', $this->paginate($this->BoardsMembers));
        $this->set('_serialize', ['boardsMembers']);
    }

    /**
     * View method
     *
     * @param string|null $id Boards Member id.
     * @return void
     * @throws \Cake\Network\Exception\NotFoundException When record not found.
     */
    public function view($id = null)
    {
        $boardsMember = $this->BoardsMembers->get($id, [
            'contain' => ['Boards', 'Members']
        ]);
        $this->set('boardsMember', $boardsMember);
        $this->set('_serialize', ['boardsMember']);
    }

    /**
     * Add method
     *
     * @return void Redirects on successful add, renders view otherwise.
     */
    public function add()
    {
        $boardsMember = $this->BoardsMembers->newEntity();
        if ($this->request->is('post')) {
            $boardsMember = $this->BoardsMembers->patchEntity($boardsMember, $this->request->data);
            if ($this->BoardsMembers->save($boardsMember)) {
                $this->Flash->success(__('The boards member has been saved.'));
                return $this->redirect(['action' => 'index']);
            } else {
                $this->Flash->error(__('The boards member could not be saved. Please, try again.'));
            }
        }
        $boards = $this->BoardsMembers->Boards->find('list', ['limit' => 200]);
        $members = $this->BoardsMembers->Members->find('list', ['limit' => 200]);
        $this->set(compact('boardsMember', 'boards', 'members'));
        $this->set('_serialize', ['boardsMember']);
    }

    /**
     * Edit method
     *
     * @param string|null $id Boards Member id.
     * @return void Redirects on successful edit, renders view otherwise.
     * @throws \Cake\Network\Exception\NotFoundException When record not found.
     */
    public function edit($id = null)
    {
        $boardsMember = $this->BoardsMembers->get($id, [
            'contain' => []
        ]);
        if ($this->request->is(['patch', 'post', 'put'])) {
            $boardsMember = $this->BoardsMembers->patchEntity($boardsMember, $this->request->data);
            if ($this->BoardsMembers->save($boardsMember)) {
                $this->Flash->success(__('The boards member has been saved.'));
                return $this->redirect(['action' => 'index']);
            } else {
                $this->Flash->error(__('The boards member could not be saved. Please, try again.'));
            }
        }
        $boards = $this->BoardsMembers->Boards->find('list', ['limit' => 200]);
        $members = $this->BoardsMembers->Members->find('list', ['limit' => 200]);
        $this->set(compact('boardsMember', 'boards', 'members'));
        $this->set('_serialize', ['boardsMember']);
    }

    /**
     * Delete method
     *
     * @param string|null $id Boards Member id.
     * @return \Cake\Network\Response|null Redirects to index.
     * @throws \Cake\Network\Exception\NotFoundException When record not found.
     */
    public function delete($id = null)
    {
        $this->request->allowMethod(['post', 'delete']);
        $boardsMember = $this->BoardsMembers->get($id);
        if ($this->BoardsMembers->delete($boardsMember)) {
            $this->Flash->success(__('The boards member has been deleted.'));
        } else {
            $this->Flash->error(__('The boards member could not be deleted. Please, try again.'));
        }
        return $this->redirect(['action' => 'index']);
    }
}
