<?php

namespace App\Repositories\Logs;

interface BuildLogInterface
{
    /**
     * Get the path.
     *
     * @return string
     */
    public function getPath();

    /**
     * Get the build at timestamp.
     *
     * @return \Carbon\Carbon
     */
    public function getBuildAt();

    /**
     * Render the build log as HTML output.
     *
     * @return string
     */
    public function render();

    /**
     * Load the build log.
     *
     * @return string
     */
    public function load();

    /**
     * Delete the build log.
     *
     * @return bool|null
     * @throws \RuntimeException
     */
    public function delete();
}
