namespace Lestroistoques\RecettesBundle\Form;

use Symfony\Component\Form\AbstractType;
use Symfony\Component\Form\FormBuilderInterface;
use Symfony\Component\OptionsResolver\OptionsResolverInterface;

class RecetteIngredientEditType extends RecetteIngredientType
{
    /**
     * @param FormBuilderInterface $builder
     * @param array $options
     */
    public function buildForm(FormBuilderInterface $builder, array $options)
    {
        parent::buildForm($builder, $options);

        $builder
            ->add   ('ingredient', new IngredientType(), array('disabled' => 'true'))
        ;
    }

    /**
     * @return string
     */
    public function getName()
    {
        return 'lestroistoques_recettesbundle_recetteingredientedit';
    }
}
