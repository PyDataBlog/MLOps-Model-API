<?php

namespace DmMailerAdminTest\Factory\Controller;

use DmMailerAdmin\Factory\Controller\Dashboard;

use DmTest\ServiceManager\ServiceLocatorDummy;

class DashboardTest extends \PHPUnit_Framework_TestCase
{
    /** @var Dashboard */
    private $sut;

    public function setUp()
    {
        $this->sut = new Dashboard;
    }

    /**
     * @covers DmMailerAdmin\Factory\Controller\Dashboard
     */
    public function testCreateService()
    {
        $serviceLocator = new ServiceLocatorDummy($this);

        $result = $this->sut->createService($serviceLocator);

        $this->assertInstanceOf('DmMailerAdmin\Controller\DashboardController', $result);
    }
}
