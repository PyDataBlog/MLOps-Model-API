<?php

namespace JovemNF\Criteria\Utils;

interface iString{

    /**
     * @param void void
     */
    public function size();

    /**
     * @param string $searchTo String que sera procutada na variavel
     * @param string $replaceTo String que ira entrar no local da outra
     */
    public function replace( $searchTo, $replaceTo );
}
