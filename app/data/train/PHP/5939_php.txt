<?php namespace DigitalPins\Models; 

use Illuminate\Database\Eloquent\Model;

/**
 * Class Callback
 * @package DigitalPins\Models
 */
class Callback extends Model {

    /**
     * The database table used by the model.
     *
     * @var string
     */
    protected $table = 'callbacks';

    /**
     * The database fields that can be mass-assigned
     *
     * @var string
     */
    protected $fillable = ['topic', 'operation_id'];

    /**
     * @return \Illuminate\Database\Eloquent\Relations\HasMany
     */
    public function orders()
    {
        return $this->hasMany('DigitalPins\Models\Order');
    }

}
