<?php

class ExtensionManager{

    var $extensions;

    function ExtensionManager(){
        $this->extensions = array();
    }
    
    function loadExtension($name, $path, $enabled){
        if($enabled){
            require_once($path);
        }
        $this->extensions[] = array("name" => $name,
                              "path" => $path,
                              "enabled" => $enabled);
    }
}

?>
