namespace Auth\GoogleClient;

//use Auth\GoogleClient\GoogleRequest;

class GoogleUtils
{
    public $token;

    public function setToken($token)
    {
        $this->token = $token;
    }

    public function getCustomerInfo()
    {
        return json_decode($this->getData(), true);
    }

    private function getData()
    {
        $request = new GoogleRequest();
        return $request->sendRequest($this->token);
    }


}
