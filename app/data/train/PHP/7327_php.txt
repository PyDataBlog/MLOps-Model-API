#!/usr/local/bin/php
<?php
// REMOTE_ADDR should always be set by the server, if it is we're not running on the CLI - exit
if(isset($_SERVER['REMOTE_ADDR'])){
	print "Must be called from the command line.";
	exit(1);
}
error_reporting(1);
ini_set('display_errors','On');
set_time_limit(0);
require_once(dirname(__FILE__).'/../libraries/init.php');

if(main::memcache_connected()) {
	$stats = main::$mc->getStats();
	echo "Stats:\n";
	foreach ($stats as $stat => $value) {
		echo "$stat: $value\n";
	}
	$keys = main::memcache_key_list();
	echo "\nVars:\n";
	foreach ($keys as $key => $value) {
		echo "$key: $value\n";
	}
} else {
	echo "No memcache connection\n";
}
?>
