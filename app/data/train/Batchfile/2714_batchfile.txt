@echo off

if "%1" == "" (
    echo You must supply a path to continue
) else (
	:: change directory
	cd %1
	:: run commands
	call git rev-parse --abbrev-ref HEAD
	echo ^|^|
	git log -1 --pretty=%%H
	echo ^|^|
	git log -1 --pretty=%%an
	echo ^|^|
	git log -1 --pretty=%%ae
	echo ^|^|
	git log -1 --pretty=%%ad
	echo ^|^|
	:: do not echo last one so we can get line breaks
	git log -1 --pretty=%%B
)
