@echo off
set DBNAME=TestBackup

echo generate update script for %DBNAME%
python run.py %DBNAME% > generated_update_testbackup.sql
if errorlevel 1 goto errgenerate
goto okgenerate

:errgenerate
echo failed generate script
exit /b 1 

:okgenerate
echo successfully generated script:
type generated_update_testbackup.sql
echo run restore
sqlcmd -E -i generated_update_testbackup.sql
if errorlevel 1 goto errrestore
goto okrestore

:errrestore
echo failed restore
exit /b 1 

:okrestore
echo successfully restored %DBNAME%
