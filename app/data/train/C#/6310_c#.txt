using System;

namespace zComp.Core
{
    /// <summary>Elément possédant une propriété readonly</summary>
    public interface IReadOnlyElement
    {
        /// <summary>Peux-t-on réaliser des modifications ?</summary>
        /// <returns>True si l'on peut modifier l'élément</returns>
        bool CanModify();

        /// <summary>Etat de lecture seule</summary>
        zReadOnlyMode ReadOnly { get; set; }
    }

    /// <summary>Etat de lecture seule</summary>
    public enum zReadOnlyMode
    {
        ParentReadOnly,
        ReadOnly,
        ReadAndWrite
    }
}
