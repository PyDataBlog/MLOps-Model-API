using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace DlBot4
{
    class Helper{
        public static void changeWindow(dynamic windowFrom, dynamic windowTo)
        {
            windowTo.Top = windowFrom.Top;
            windowTo.Left = windowFrom.Left;
            windowTo.Height = windowFrom.Height;
            windowTo.Width = windowFrom.Width;
            windowTo.Show();
            windowFrom.Hide();
        }
    }
}
