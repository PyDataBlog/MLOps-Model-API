using Newtonsoft.Json;
using System;
using System.Collections.Generic;

namespace QnAMaker.Model
{
    [Serializable]
    public class FeedbackRecords
    {
        [JsonProperty(PropertyName = "feedbackRecords")]
        public ICollection<FeedbackRecord> Records { get; set; }

        public FeedbackRecords(FeedbackRecord record)
        {
            Records = new List<FeedbackRecord>
            {
                record
            };
        }
    }
}
