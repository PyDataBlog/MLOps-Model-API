namespace NetCreditHub.Data.FastCrud.Configuration.StatementOptions.Builders
{
    using Aggregated;

    /// <summary>
    /// 抽象一个可以有条件语句的 SQL 语句选项生成器接口.
    ///     用于函数计算, 如: MIN(column_name),MAX(column_name)...
    /// </summary>
    public interface INColumnConditionalSqlStatementOptionsBuilder<TEntity>
        : IStandardSqlStatementOptionsSetter<TEntity, INColumnConditionalSqlStatementOptionsBuilder<TEntity>>,
        INColumnConditionalSqlStatementOptionsSetter<TEntity, INColumnConditionalSqlStatementOptionsBuilder<TEntity>>,
        IRelationalSqlStatementOptionsSetter<TEntity, INColumnConditionalSqlStatementOptionsBuilder<TEntity>>,
        IParameterizedSqlStatementOptionsSetter<TEntity, INColumnConditionalSqlStatementOptionsBuilder<TEntity>>
    {
    }

    /// <summary>
    /// 一个可以有条件语句的 SQL 语句选项生成器.
    ///     用于函数计算, 如: MIN(column_name),MAX(column_name)...
    /// </summary>
    internal class NColumnConditionalSqlStatementOptionsBuilder<TEntity>
        : AggregatedSqlStatementOptionsBuilder<TEntity, INColumnConditionalSqlStatementOptionsBuilder<TEntity>>
        , INColumnConditionalSqlStatementOptionsBuilder<TEntity>
    {
        protected override INColumnConditionalSqlStatementOptionsBuilder<TEntity> Builder => this;
    }
}
