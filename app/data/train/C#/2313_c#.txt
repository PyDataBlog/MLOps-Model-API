namespace Rutt.Messaging.Models
{
    public sealed class MessagingClientInfo
    {
        public MessagingClientType ClientType { get; set; }

        public string DisplayName { get; set; }

        public string ExecutorName { get; set; }

        public string ExecutorSource { get; set; }

        public string Id { get; set; }

        public string Url { get; set; }
        

        public MessagingClientInfo()
        {
            ClientType = MessagingClientType.Unknown;
            DisplayName = "";
            Id = "";
            ExecutorName = "";
            ExecutorSource = "";
            Url = "";
        }
    }
}
