namespace TurboRango.Dominio
{
    public class Contato : Entidade
    {
        // Atributos
        public string Site { get; set; }
        public string Telefone { get; set; }
        // Construtores
        public Contato()
        {
            
        }
        public Contato(string site, string telefone)
        {
            this.Site = site;
            this.Telefone = telefone;
        }

    }
}
