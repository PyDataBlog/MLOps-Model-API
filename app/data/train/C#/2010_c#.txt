using System.Collections.Generic;
using System.Linq;
using System.Web.Http;
using Business.Logic.Factories;
using Business.Logic.Objects;
using Business.Logic.Queries;

namespace DFWOutdoors.Controllers
{
    public class CampInfoController : ApiController
    {
        [HttpPost]
        public IEnumerable<CampSite> CheckCampsiteAvailability(CampsiteSearchCriteria campsiteSearchCriteria)
        {
            var campsiteQuery = new QueryFactory<GetCampsitesByCampsiteSearchCriteria>().Create();

            campsiteQuery.Url = Urls.GetAllCampsitesBySearchCriteria;
            campsiteQuery.CampsiteSearchCriteria = campsiteSearchCriteria;

            return campsiteQuery.Execute().OfType<CampSite>().ToList();
        }
    }
}
