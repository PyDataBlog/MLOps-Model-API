using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using Arch.CFX.AppInternals.Components;
using Arch.CFX.AppInternals.Manager.Components.Impl;
using Arch.CFX.AppInternals.Utility;
using Arch.CFX.Ioc;


namespace Arch.CFX.AppInternals.Manager
{
    public class ComponentsManager : IComponentManager
    {

        public List<IComponent> GetAllComponents()
        {
            return VenusContainerLoader.Container.LookupList<IComponent>().ToList<IComponent>();
        }

        public IComponent GetComponentByID(string id)
        {
            return VenusContainerLoader.Container.TryLookup<IComponent>(id);
        }

        public void Register(IComponent component)
        {
            VenusContainerLoader.Container.Define(typeof(IComponent), component.GetType(), component.GetID());
        }
    }
}
