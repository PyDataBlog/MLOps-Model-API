using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using Game.Utils;

namespace Game.Commands
{
    public class SelectSecondTowerTypeCommand : Command
    {
        public SelectSecondTowerTypeCommand(Board board) : base(board)
        {
        }

        public override void Execute()
        {
            var towerToBuild = this._board.Towers[2];
            this._board.Selector.SelectTowerToBuild(towerToBuild);
        }
    }
}
