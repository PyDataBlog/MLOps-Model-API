using System;
using System.Collections.Generic;
using System.Threading.Tasks;

namespace Vstack.Common.Utilities
{
    public class VstackTask
    : Task
    {
        public VstackTask(Action action)
            : base(action)
        {
            this.Dependencies = new List<VstackTask>() { };
        }

        public VstackTask(Action action, params VstackTask[] require)
            : base(action)
        {
            this.Dependencies = require;
        }

        public IEnumerable<VstackTask> Dependencies { get; }
    }
}
