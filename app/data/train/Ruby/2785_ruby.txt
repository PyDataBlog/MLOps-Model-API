class Sound < Media
  
  # we store the uploaded mime-type to distinguish these from videos,
  # but we convert them to mp4's so we can play them via the HTML5 Flowplayer.
  
  
  def get_styles
    {
      # for mp4 audio-only movie from mp3.
      :preview => { :format => :mp4, :processors => [:sound_converter] }
    }
  end
  
  
  def accepts
    # we accept video/mp4 to cover the times we make these in seeds.rb
    [ 'audio/mpeg', 'audio/x-mpeg', 'audio/mp3', 'audio/x-mp3', 'audio/mpeg3', 'audio/x-mpeg3', 'audio/mpg', 'audio/x-mpg', 'audio/x-mpegaudio', 'video/mp4' ]
  end

  
  def dimensions
    "600x0"
  end
  
end
