module CloudFormer
  module CloudFront
    class StreamingDistribution
      class StreamingDistributionConfig < ResourceProperty
        aws_attribute :logging, type: StreamingDistribution::Logging, embed: true
        aws_attribute :comment, type: String, required: true
        aws_attribute :price_class, type: String
        aws_attribute :s3_origin, type: StreamingDistribution::S3Origin, embed: true, required: true
        aws_attribute :enabled, type: Boolean, required: true
        aws_attribute :aliases, type: String, list: true
        aws_attribute :trusted_signers, type: StreamingDistribution::TrustedSigners, embed: true, required: true
      end
    end
  end
end
