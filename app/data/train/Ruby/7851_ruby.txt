class KuberRouter < FPM::Cookery::Recipe
  name 'kube-router'
  version '0.1.0-rc2'
  source "https://github.com/cloudnativelabs/kube-router/releases/download/v#{version}/kube-router_#{version}_linux_amd64.tar.gz"

  def build; end

  def install
    sbin.install name
    prefix('lib/systemd/system').install workdir("#{name}.service")
  end
end
