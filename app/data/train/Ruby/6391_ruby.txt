require 'json'

module SimpleFirebase
  class Response
    def initialize(http_response)
      @http_response = http_response
    end

    def success?
      @http_response.code.to_i >= 200 && @http_response.code.to_i < 300
    end

    def json_body
      JSON.parse(@http_response.body || 'null')
    end

    def body
      @http_response.body
    end
  end
end
