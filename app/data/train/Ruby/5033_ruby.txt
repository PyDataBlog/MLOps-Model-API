class SchoolsController < ApplicationController
  def index
    renter = Renter.find( params[:renter_id])
    if renter
      @schools = renter.schools
    end
    render json: @schools
  end

  def get_school_cycles
  	school = School.find( params[:id])
  	if school
  		render json:school.cycles
  	end
  end
end
