# frozen_string_literal: true
FactoryGirl.define do
  sequence :unique_name do |n|
    "project#{n}"
  end
  factory :project do
    name { generate :unique_name }
    description { Faker::Hipster.paragraph 5, true, 3 }
    # rubocop:disable all
    public { [true, false].sample }
    # rubocop:enable all
    project_status { Project.project_statuses.keys.sample }
    created_at { rand(3.years).seconds.ago }
    updated_at { DateTime.strptime(rand(created_at.to_i..DateTime.now.to_i).to_s, '%s') }
  end
end
