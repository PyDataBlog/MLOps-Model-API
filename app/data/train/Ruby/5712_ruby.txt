require 'wfstatus/wfmessage'

class Wfstatus::Builder
  attr_reader :config, :messages
  def initialize(config, messages)
    @config = config
    @messages = messages
  end

  def relevant_messages
    @relevant_messages ||= @messages.select { |m| m.relevant? }
  end

  def content
    relevant_messages.join("\n")
  end

end
