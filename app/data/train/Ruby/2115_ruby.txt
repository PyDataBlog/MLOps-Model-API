module Tremor
  module Seismi
    class Response

      prepend Tremor::Response::Base

      def initialize(data)
        @data = JSON.load(data)
      end

      def count
        @data['count'.freeze]
      end

      def earthquakes
        @data['earthquakes'.freeze].map do |earthquake|
          Tremor::Seismi::Earthquake.new(earthquake)
        end
      end
    end
  end
end
