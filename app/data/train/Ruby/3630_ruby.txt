class BabylonianMethod
  attr_accessor :working_root, :number
  def initialize(num)
    self.number = num
    self.working_root = number / 2.to_f
  end

  def calculate(precision=0.000001)
    until [(working_root*working_root), self.number].max - [(working_root*working_root), self.number].min < precision
      approximate
    end
    return working_root
  end
  def approximate
    self.working_root = ((self.number / self.working_root) + self.working_root) / 2
  end
end
