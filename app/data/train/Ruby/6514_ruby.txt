feature "The Staff Website home page", :network => 'staff', :production_safe => true do

  before(:each) do
    set_host_for('http://staff.wrdsb.ca')
  end

  it 'should require a authentication' do
    visit('/')
    expect(page).to have_css('input#user_login')
    expect(page).to have_css('input#user_pass')
  end

end
