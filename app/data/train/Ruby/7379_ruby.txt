class Web::Profile::EventsController < Web::Profile::ApplicationController
  authorize_actions_for Event
  def index
    @collection = Event.all.order("date desc").decorate
  end

  def new
    @event = ModerateEventType.new(params[:event])
    respond_with @event
  end

  def create
    @event = ModerateEventType.new(params[:event])
    @event.assign_attributes(params[:event])
    @event.save
    respond_with @event, location: profile_events_path
  end

  def show
    @event = Event.find(params[:id]).decorate
    respond_with @event
  end

  def edit
    @event = Event.find(params[:id]).becomes(ModerateEventType)
    respond_with @event
  end

  def update
    @event = Event.find(params[:id]).becomes(ModerateEventType)
    @event.assign_attributes(params[:event])
    @event.save
    respond_with @event, location: profile_events_path
  end

  def destroy
    @event = Event.find(params[:id])
    @event.destroy
    respond_with @event, location: profile_events_path
  end
end
