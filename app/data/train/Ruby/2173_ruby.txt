include_recipe 'yum-epel'
node.set['build-essential']['compile_time'] = true
include_recipe 'build-essential'

# install sysadmin utilities missing from centos-6.5
# and keep them up-to-date for security patches
%w(
  bash
  lsof
  man
  nc
  redhat-lsb-core
  screen
  telnet
  tree
  wget
  vim
).each do |pkg|
  package pkg do
    action :upgrade
  end
end
