Imports Visomes.MySQL
Imports Visomes.SGV.Core

Public Class ExternoTipoOcorrenciaArea
    Implements ICadastro(Of ExternoTipoOcorrenciaArea)
    Private idValue As Integer
    Private descricaoValue As String
    Private cadastroValue As Date
    Private usuarioValue As Usuario
    Private statusValue As Boolean = True

#Region "Property"

    Public Property ID() As Integer Implements ICadastro(Of Visomes.SGV.ExternoTipoOcorrenciaArea).ID
        Get
            Return idValue
        End Get
        Set(ByVal value As Integer)
            idValue = value
        End Set
    End Property

    Public Property Usuario() As Usuario Implements ICadastro(Of Visomes.SGV.ExternoTipoOcorrenciaArea).Usuario
        Get
            Return usuarioValue
        End Get
        Set(ByVal value As Usuario)
            usuarioValue = value
        End Set
    End Property

    Public Property Descricao() As String Implements ICadastro(Of Visomes.SGV.ExternoTipoOcorrenciaArea).Descricao
        Get
            Return descricaoValue
        End Get
        Set(ByVal value As String)
            descricaoValue = value
        End Set
    End Property

    Public Property Status() As Boolean Implements ICadastro(Of Visomes.SGV.ExternoTipoOcorrenciaArea).Status
        Get
            Return statusValue
        End Get
        Set(ByVal value As Boolean)
            statusValue = value
        End Set
    End Property

    Public Property Cadastro() As Date Implements ICadastro(Of Visomes.SGV.ExternoTipoOcorrenciaArea).Cadastro
        Get
            Return cadastroValue
        End Get
        Set(ByVal value As Date)
            cadastroValue = value
        End Set
    End Property

#End Region

    Public Sub New(ByVal id As Integer)
        Me.idValue = id
        CarregaDados()
    End Sub

    Public Sub New()
    End Sub

    Private Sub CarregaDados()
        Dim idUsuario As Integer = 0
        Connection.ConnectionMysql()
        MyData = DML.Select("*", "externo_ocorrencia_area t", "id = " & Me.idValue & "")
        If MyData.Read() Then
            idUsuario = MyData.GetInt32(1)
            Me.descricaoValue = MyData.GetString(2)
            Me.cadastroValue = MyData.GetDateTime(3)
            Me.statusValue = MyData.GetBoolean(4)
        End If
        Connection.DisconnectMySQL()
        MyData.Dispose()
        If idUsuario > 0 Then Me.usuarioValue = New Usuario(idUsuario)
    End Sub

    Public Function Commit() As Boolean Implements ICadastro(Of Visomes.SGV.ExternoTipoOcorrenciaArea).Commit
        If Me.idValue = 0 Then
            Return DML.Insert("externo_ocorrencia_area", "descricao,dtCadastro,status,idUsuario", "'" & Me.descricaoValue & "',CURRENT_DATE()," & Me.statusValue & "," & Me.usuarioValue.ID & "")
        Else
            Return DML.Update("externo_ocorrencia_area", "descricao='" & Me.descricaoValue & "',dtCadastro=CURRENT_DATE(),status=" & Me.statusValue & ",idUsuario=" & Me.usuarioValue.ID & "", "id=" & Me.idValue & "")
        End If
    End Function

    Public Function GetLista() As Object() Implements ICadastro(Of Visomes.SGV.ExternoTipoOcorrenciaArea).GetLista
        Return Core.FeedsLists("ts.descricao", "externo_ocorrencia_area ts  ", "ts.status = 1 ")
    End Function

    Public Function CarregaLocalizacoes() As Generic.List(Of ExternoTipoOcorrenciaArea) Implements ICadastro(Of Visomes.SGV.ExternoTipoOcorrenciaArea).CarregaCadastros
        Dim lista As New Generic.List(Of ExternoTipoOcorrenciaArea)
        Dim listaID As New Generic.List(Of Integer)
        Connection.ConnectionMysql()
        Core.MyData = DML.Select("id", "externo_ocorrencia_area", "status=1")
        While MyData.Read
            listaID.Add(MyData.GetInt32(0))
        End While
        Connection.DisconnectMySQL()
        Core.MyData.Dispose()
        For Each num As Integer In listaID
            lista.Add(New ExternoTipoOcorrenciaArea(num))
        Next
        Return lista
    End Function

    Public Function Inativar() As Boolean Implements ICadastro(Of Visomes.SGV.ExternoTipoOcorrenciaArea).Inativar
        Return DML.Update("externo_ocorrencia_area", "status=0", "id=" & Me.idValue & "")
    End Function

    Public Function Delete() As Boolean Implements ICadastro(Of Visomes.SGV.ExternoTipoOcorrenciaArea).Delete
        Return DML.Delete("externo_ocorrencia_area", "id=" & Me.idValue & "")
    End Function

    Public Shared Function GetIDByDescricao(ByVal str As String) As Integer
        Dim id As Integer = 0
        Connection.ConnectionMysql()
        MyData = DML.Select("id", "externo_ocorrencia_area", "descricao = '" & str & "'")
        If MyData.Read() Then id = MyData.GetInt32(0)
        Connection.DisconnectMySQL()
        MyData.Dispose()
        Return id
    End Function

    Public Shared Function GetDescricaoByID(ByVal id As Integer) As String
        Dim descricao As String = String.Empty
        Connection.ConnectionMysql()
        Core.MyData = DML.Select("descricao", "externo_ocorrencia_area", "id = '" & id & "'")
        If Core.MyData.Read() Then descricao = Core.MyData.GetString(0)
        Connection.DisconnectMySQL()
        Core.MyData.Dispose()
        Return descricao
    End Function

    Public Overrides Function ToString() As String
        Return Me.Descricao
    End Function
End Class
