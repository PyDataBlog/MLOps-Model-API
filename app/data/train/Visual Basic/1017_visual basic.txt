Imports System.Math
Public Class mainForm


    Dim aInteger, bInteger, cInteger, xSolution1, xSolution2 As Decimal

    Private Sub calculateButton_Click(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles calculateButton.Click
        aInteger = Val(aTextBox.Text)
        bInteger = Val(bTextBox.Text)
        cInteger = Val(cTextBox.Text)
        xSolution1 = -bInteger + Math.Sqrt((bInteger ^ 2) - (4 * (aInteger * cInteger))) / (2 * aInteger)
        xSolution2 = (-bInteger - Math.Sqrt((bInteger ^ 2) - (4 * (aInteger * cInteger)))) / (2 * aInteger)
        answerTextBox.Text = "X = " & xSolution1.ToString & " & X = " & xSolution2.ToString

    End Sub

    Private Sub clearButton_Click(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles clearButton.Click
        answerTextBox.Text = ""
        aTextBox.Text = ""
        bTextBox.Text = ""
        cTextBox.Text = ""
    End Sub

    Private Sub mainForm_Load(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles MyBase.Load

    End Sub
End Class
