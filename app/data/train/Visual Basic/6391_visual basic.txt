        '28/01/2016     Login failure Exception. Thrown when a login attempt fails due to no accounts returned from the MultiLogin.

            Inherits Exception

                MyBase.New("A login attempt failed. No accounts were returned from the Linnworks API.")
            End Sub
        End Class
