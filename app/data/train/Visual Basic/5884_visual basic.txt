#Region " Shared "

Private Shared _DBDetails As DMS.Base.Models.DBDetails = Nothing

Public Shared Async Function Delete(dbAccess As DMS.Base.Data.IDBAccess, entities As IEnumerable(Of Sound)) As Task
   Dim Tasks As New List(Of Task)
   For Each Entity As Sound In entities
      Tasks.Add(Delete(dbAccess, Entity))
   Next
   Await Task.WhenAll(Tasks)
End Function

Public Shared Async Function Delete(dbAccess As DMS.Base.Data.IDBAccess, entity As Sound) As Task
      Await dbAccess.ExecuteNonQuery(GetDBDetails.Delete, entity.GetParameters())
End Function

Public Shared Async Function Upsert(dbAccess As DMS.Base.Data.IDBAccess, entities As IEnumerable(Of Sound)) As Task
   If entities.Count = 1 Then
      Await Upsert(dbAccess, entities.FirstOrDefault())
   Else
         Dim DBDetails As DMS.Base.Models.DBDetails = GetDBDetails()
         Using DataTable As DataTable = GetDataTable()
            For Each Entity As Sound In entities
               Dim DataRow As DataRow = DataTable.NewRow
               Entity.PopulateDataRow(DataRow)
               DataTable.Rows.Add(DataRow)
            Next
            Await dbAccess.Merge(DBDetails.CreateTemp, DBDetails.DropTemp, DataTable, DBDetails.Merge, DBDetails.TableName)
         End Using
   End If
End Function

Public Shared Async Function Upsert(dbAccess As DMS.Base.Data.IDBAccess, entity As Sound) As Task
      Dim Parameters As IDictionary(Of String, Object) = entity.GetParameters()
      If String.IsNullOrEmpty(entity.ID) OrElse entity.ID.Equals("-1") OrElse ToIntegerDB(Await dbAccess.ExecuteScalar(GetDBDetails.CountSingle, Parameters)) <= 0 Then
         entity.SoundID = ToIntegerDB(Await dbAccess.ExecuteScalar(GetDBDetails.Insert, Parameters))
      Else
         Await dbAccess.ExecuteNonQuery(GetDBDetails.Update, Parameters)
      End If
End Function

Public Shared Async Function GetAll(dbAccess As DMS.Base.Data.IDBAccess) As Threading.Tasks.Task(Of IEnumerable(Of Sound))
      Return Await dbAccess.ExecuteReader(Of Sound)(GetDBDetails.SelectAll)
End Function

Public Shared Function GetDataTable() As DataTable
   Dim DataTable As New DataTable
   DataTable.Columns.Add("SoundFileName")
   DataTable.Columns.Add("Description")
   DataTable.Columns.Add("EditStatus")
   DataTable.Columns.Add("EditStatusParentID")
   DataTable.Columns.Add("EditStatusUserID")
   DataTable.Columns.Add("SoundID")
   Return DataTable
End Function

Public Shared Function GetDBDetails() As DMS.Base.Models.DBDetails
   If _DBDetails Is Nothing Then
      _DBDetails = New DMS.Base.Models.DBDetails
      _DBDetails.CountAll = "SELECT COUNT(*) FROM [Sounds]"
      _DBDetails.CountSingle = "SELECT COUNT(*) FROM [Sounds] WHERE ()"
      _DBDetails.CreateTemp = "CREATE TABLE #Sounds ([SoundFileName] [varchar](255) NULL,[Description] [varchar](255) NULL,[EditStatus] [varchar](1) NULL,[EditStatusParentID] [varchar](50) NULL,[EditStatusUserID] [int] NULL,[SoundID] [int] NULL)"
      _DBDetails.Delete = "DELETE FROM [Sounds] WHERE ()"
      _DBDetails.DropTemp = "DROP TABLE #Sounds"
      _DBDetails.GetDataTable = New DMS.Base.Delegates.GetDataTable(AddressOf GetDataTable)
      _DBDetails.Insert = "INSERT INTO [Sounds] ([SoundFileName],[Description],[EditStatus],[EditStatusParentID],[EditStatusUserID]) OUTPUT Inserted.[SoundID] VALUES (@SoundFileName,@Description,@EditStatus,@EditStatusParentID,@EditStatusUserID)"
      _DBDetails.Merge = "MERGE INTO [Sounds] As [Target] USING #Sounds As [Source] ON  WHEN MATCHED THEN UPDATE SET [Target].[SoundFileName]=[Source].[SoundFileName],[Target].[Description]=[Source].[Description],[Target].[EditStatus]=[Source].[EditStatus],[Target].[EditStatusParentID]=[Source].[EditStatusParentID],[Target].[EditStatusUserID]=[Source].[EditStatusUserID] WHEN NOT MATCHED THEN INSERT ([SoundFileName],[Description],[EditStatus],[EditStatusParentID],[EditStatusUserID]) VALUES ([Source].[SoundFileName],[Source].[Description],[Source].[EditStatus],[Source].[EditStatusParentID],[Source].[EditStatusUserID]);"
      _DBDetails.ModelName = "Sound"
      _DBDetails.SelectAll = "SELECT [SoundFileName],[Description],[EditStatus],[EditStatusParentID],[EditStatusUserID],[SoundID] FROM [Sounds]"
      _DBDetails.TableName = "Sounds"
      _DBDetails.Update = "UPDATE [Sounds] SET [SoundFileName]=@SoundFileName,[Description]=@Description,[EditStatus]=@EditStatus,[EditStatusParentID]=@EditStatusParentID,[EditStatusUserID]=@EditStatusUserID WHERE ()"
   End If
   Return _DBDetails
End Function
