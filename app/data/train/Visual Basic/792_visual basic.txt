Imports System.Windows.Forms

Public Class CustomDialog
    Private buttonETextValue As String = ""
    Public Property buttonEText() As String
        Get
            Return buttonETextValue
        End Get
        Set(ByVal value As String)
            buttonETextValue = value
        End Set
    End Property
    Private buttonDTextValue As String = ""
    Public Property buttonDText() As String
        Get
            Return buttonDTextValue
        End Get
        Set(ByVal value As String)
            buttonDTextValue = value
        End Set
    End Property
    Private buttonCTextValue As String = ""
    Public Property buttonCText() As String
        Get
            Return buttonCTextValue
        End Get
        Set(ByVal value As String)
            buttonCTextValue = value
        End Set
    End Property
    Private buttonBTextValue As String = ""
    Public Property buttonBText() As String
        Get
            Return buttonBTextValue
        End Get
        Set(ByVal value As String)
            buttonBTextValue = value
        End Set
    End Property
    Private buttonATextValue As String = ""
    Public Property buttonAText() As String
        Get
            Return buttonATextValue
        End Get
        Set(ByVal value As String)
            buttonATextValue = value
        End Set
    End Property
    Private buttonEResultValue As Integer = 0
    Public Property buttonEResult() As Integer
        Get
            Return buttonEResultValue
        End Get
        Set(ByVal value As Integer)
            buttonEResultValue = value
        End Set
    End Property
    Private buttonDResultValue As Integer = 0
    Public Property buttonDResult() As Integer
        Get
            Return buttonDResultValue
        End Get
        Set(ByVal value As Integer)
            buttonDResultValue = value
        End Set
    End Property
    Private buttonCResultValue As Integer = 0
    Public Property buttonCResult() As Integer
        Get
            Return buttonCResultValue
        End Get
        Set(ByVal value As Integer)
            buttonCResultValue = value
        End Set
    End Property
    Private buttonBResultValue As Integer = 0
    Public Property buttonBResult() As Integer
        Get
            Return buttonBResultValue
        End Get
        Set(ByVal value As Integer)
            buttonBResultValue = value
        End Set
    End Property
    Private buttonAResultValue As Integer = 0
    Public Property buttonAResult() As Integer
        Get
            Return buttonAResultValue
        End Get
        Set(ByVal value As Integer)
            buttonAResultValue = value
        End Set
    End Property
    Private dialogTextValue As String = ""
    Public Property dialogMessage() As String
        Get
            Return dialogTextValue
        End Get
        Set(ByVal value As String)
            dialogTextValue = value
        End Set
    End Property
    Private dialogTitleValue As String = ""
    Public Property dialogTitle() As String
        Get
            Return dialogTitleValue
        End Get
        Set(ByVal value As String)
            dialogTitleValue = value
        End Set
    End Property
    Private useIconValue As Boolean = False
    Public Property useIcon() As Boolean
        Get
            Return useIconValue
        End Get
        Set(ByVal value As Boolean)
            useIconValue = value
        End Set
    End Property
    Private dialogIconValue As Image = My.Resources.NoImage64
    Public Property dialogIcon() As Image
        Get
            Return dialogIconValue
        End Get
        Set(ByVal value As Image)
            dialogIconValue = value
        End Set
    End Property
    Private choiceValue As Integer
    Public Property choice() As Integer
        Get
            Return choiceValue
        End Get
        Set(ByVal value As Integer)
            choiceValue = value
        End Set
    End Property



    Private Sub CustomDialog_Load(sender As System.Object, e As System.EventArgs) Handles MyBase.Load
        If Not buttonAText = "" Then CustomBtnA.Visible = True
        If Not buttonBText = "" Then CustomBtnB.Visible = True
        If Not buttonCText = "" Then CustomBtnC.Visible = True
        If Not buttonDText = "" Then CustomBtnD.Visible = True
        If Not buttonEText = "" Then CustomBtnE.Visible = True
        CustomBtnA.Text = buttonAText
        CustomBtnB.Text = buttonBText
        CustomBtnC.Text = buttonCText
        CustomBtnD.Text = buttonDText
        CustomBtnE.Text = buttonEText
        If useIcon Then
            Try
                DialogIconCtnr.Image = dialogIcon
            Catch ex As Exception
                DialogIconCtnr.Image = My.Resources.NoImage64
            End Try
        Else
            DialogIconCtnr.Visible = False
            DialogText.Location = New Point(12, 12)
        End If
        DialogText.Text = dialogMessage
        Me.Text = dialogTitle
    End Sub

    Private Sub CustomBtnE_Click(sender As System.Object, e As System.EventArgs) Handles CustomBtnE.Click
        Me.choice = buttonEResult
        Me.Close()
    End Sub

    Private Sub CustomBtnD_Click(sender As System.Object, e As System.EventArgs) Handles CustomBtnD.Click
        Me.choice = buttonDResult
        Me.Close()
    End Sub

    Private Sub CustomBtnC_Click(sender As System.Object, e As System.EventArgs) Handles CustomBtnC.Click
        Me.choice = buttonCResult
        Me.Close()
    End Sub

    Private Sub CustomBtnB_Click(sender As System.Object, e As System.EventArgs) Handles CustomBtnB.Click
        Me.choice = buttonBResult
        Me.Close()
    End Sub

    Private Sub CustomBtnA_Click(sender As System.Object, e As System.EventArgs) Handles CustomBtnA.Click
        Me.choice = buttonAResult
        Me.Close()
    End Sub
End Class
