Public Class miscFunctions
    Inherits System.Web.UI.Page

    Protected Sub Page_Load(ByVal sender As Object, ByVal e As System.EventArgs) Handles Me.Load
        If IsNothing(Session("User_Name")) Then
            Response.Redirect("Restricted.aspx")
        Else
            If IsPostBack Then
            Else
                Dim action As String = Request.QueryString("action")

                If action = "deleteNote" Then
                    deleteNote()
                End If
            End If
        End If
    End Sub


    Private Sub deleteNote()
        Dim strSql As String = "Delete from server_notes where server_id = " & Request.QueryString("Serverid") & " and server_notes_id = " & Request.QueryString("NoteID")
        g_IO_Execute_SQL(strSql, False)

        Response.Redirect("serverdetail.aspx?id=" & Request.QueryString("ServerID"))
    End Sub
End Class
