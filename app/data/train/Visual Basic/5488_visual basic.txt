        ''' <summary>
        ''' Structure to resemble the Button FlatAppearance property.
        ''' </summary>
        Public Structure ButtonFlatAppearance

            Public BorderColor As Color
            Public BorderSize As Integer
            Public MouseDownBackColor As Color
            Public MouseOverBackColor As Color

            ''' <summary>
            ''' Creates a new structure to resemble the Button FlatAppearance property.
            ''' </summary>
            ''' <param name="_bClr">Border color.</param>
            ''' <param name="_bSz">Border size.</param>
            ''' <param name="_mdClr">Mouse-down back color.</param>
            ''' <param name="_moClr">Mouse-over back color.</param>
            Public Sub New(ByVal _bClr As Color, ByVal _bSz As Integer, ByVal _mdClr As Color, ByVal _moClr As Color)
                BorderColor = _bClr
                BorderSize = _bSz
                MouseDownBackColor = _mdClr
                MouseOverBackColor = _moClr
            End Sub

        End Structure

        ''' <summary>
        ''' Determines whether the theme is partial or full.
        ''' </summary>
        Public Enum ThemeTypes As Byte
            Full = 0
            FormOnly = 1
            StudyTabAndForm = 2
            LearnTabAndForm = 3
            TestTabAndForm = 4
        End Enum

        ''' <summary>
        ''' Applies the current theme to the main form.
        ''' </summary>
        ''' <param name="_form">The main form object.</param>
        Public Sub ApplyTheme(ByRef _form As MainForm)
            ApplyFormTheme(_form)
            ApplyStudyTheme(_form)
            ApplyLearnTheme(_form)
            ApplyTestTheme(_form)
        End Sub
        ''' <summary>
        ''' Creates a new full theme.
        ''' </summary>
        Public Sub New()
            MyBase.New()
            _themeType = ThemeTypes.Full
        End Sub
        ''' <summary>
        ''' Creates a partial/full theme(used for custom themes). DO NOT USE FOR BASE THEME!
        ''' </summary>
        ''' <param name="_thTyp">The type of theme. Determines whether the theme is partial or full.</param>
        Public Sub New(ByVal _thTyp As ThemeTypes)
            MyBase.New()
            _themeType = _thTyp
        End Sub

    End Class
