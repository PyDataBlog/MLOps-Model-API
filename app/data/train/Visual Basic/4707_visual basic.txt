Public Class TTListBox : Inherits ListBox

  Private p_fDataBinding As Boolean

  Protected Overrides Sub OnSelectedIndexChanged(e As EventArgs)
    If p_fDataBinding Then Return
    MyBase.OnSelectedIndexChanged(e)
  End Sub

  Protected Overrides Sub OnSelectedValueChanged(e As EventArgs)
    If p_fDataBinding Then Return
    MyBase.OnSelectedValueChanged(e)
  End Sub

  Public Sub SetDataSource(oList As IList)
    p_fDataBinding = True
    Me.DataSource = oList
    Me.SelectedIndex = -1
    p_fDataBinding = False
  End Sub

End Class
