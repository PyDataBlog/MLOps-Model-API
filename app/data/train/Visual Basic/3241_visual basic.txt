
Partial Class SubPage
    Inherits System.Web.UI.Page

    Protected Sub Page_Load(ByVal sender As Object, ByVal e As System.EventArgs) Handles Me.Load
        Dim aryAccountID() As String = {"1234", "5678", "2234"}
        GridView1.DataSource = aryAccountID
        GridView1.DataBind()
    End Sub

    Protected Sub GridView1_SelectedIndexChanged(ByVal sender As Object, ByVal e As System.EventArgs) Handles GridView1.SelectedIndexChanged
        Dim strAccountID As String = GridView1.SelectedRow.Cells(1).Text
        Dim sbrJS As New StringBuilder
        sbrJS.Append(" window.opener.document.getElementById('txtAccountID').value='" & strAccountID & "'; ")
        sbrJS.Append(" window.open('','_self',''); ")
        sbrJS.Append(" window.close(); ")
        ClientScript.RegisterStartupScript(Page.GetType(), "Close", sbrJS.ToString, True)
    End Sub
End Class
