    Private Sub ResetPanForCableWrap()
        _log.Debug(_main.LoggerComboBox.Text & " -Starting sub 'resetPanForCableWrap()'.")
        Dim fcommand As String = String.Empty
        Try
            _positioner.ResetPan()
            fcommand = _positioner.GetPanPosition
            If fcommand < 0 Then fcommand += 360
            Gauge1.Value = CDbl(fcommand)

        End Try

    End Sub

    Public Sub GetPositionResolution()
        _log.Debug(_main.LoggerComboBox.Text & " -Starting sub 'getPositionResolution()'")
        Try

            'Me.mbSession.Flush(BufferTypes.InBuffer, True)
            'Me.QueryPTU300("ED")
            Dim answer As String = _positioner.Resolution 'Me.QueryPTU300("PR")
            Dim tmpAnswer() = answer.Split(Chr(32))

            Try

                'the first cases apply to the positioner in van8
                If CDbl(tmpAnswer(1)) > 80 And CDbl(tmpAnswer(1)) < 90 Then
                    DegreesPerPanPosition = DegreesPerPanPositionF
                ElseIf CDbl(tmpAnswer(1)) > 40 And CDbl(tmpAnswer(1)) < 50 Then
                    DegreesPerPanPosition = DegreesPerPanPositionH
                ElseIf CDbl(tmpAnswer(1)) > 20 And CDbl(tmpAnswer(1)) < 30 Then
                    DegreesPerPanPosition = DegreesPerPanPositionQ
                ElseIf CDbl(tmpAnswer(1)) > 10 And CDbl(tmpAnswer(1)) < 20 Then
                    DegreesPerPanPosition = DegreesPerPanPositionE
                    'these cases are for the positioner in van 8
                ElseIf CDbl(tmpAnswer(1)) > 180 And CDbl(tmpAnswer(1)) < 190 Then
                    DegreesPerPanPosition = DegreesPerPanPositionFvan7
                ElseIf CDbl(tmpAnswer(1)) > 90 And CDbl(tmpAnswer(1)) < 100 Then
                    DegreesPerPanPosition = DegreesPerPanPositionHvan7
                End If
            Catch ex As Exception
                _log.Error(_main.LoggerComboBox.Text & " -Error in sub 'getPositionResolution()': " & ex.Message)
                ErrorMessageFormat("Error finding pan resolution.", ex)
            End Try

        Catch ex As Exception
            _log.Error(_main.LoggerComboBox.Text & " -Error in sub 'getPositionResolution()': " & ex.Message)
            ErrorMessageFormat("Error! Power cycle positioner.", ex)
        End Try

        DegreesPerTiltPosition = DegreesPerPanPosition

    End Sub

    Function QueryPtu300(ByVal commandString As String) As String
        _log.Debug(_main.LoggerComboBox.Text & " -Starting sub 'QueryPTU300()'.")

        Dim returnString As String = String.Empty

        Try
            ReturnString = _positioner.SendCommand(CommandString)
            Threading.Thread.Sleep(100)
            Return ReturnString
        Catch ex As Exception
            _log.Error(_main.LoggerComboBox.Text & " -Error in sub 'QueryPTU300': " & ex.Message)
            ErrorMessageFormat("Error. ", ex)
            Dim tmpStr As String = (ex.Message)
            If tmpStr.ToUpper.Contains("TIMEOUT") Then
                Dim result As MsgBoxResult = MessageBox.Show("Cannot establish comm with the positioner.  Please check that the correct COM port has been chosen, " & vbCrLf & "and the positioner has power. Would you like to abort?", "Error Box", MessageBoxButtons.YesNo)
                If result = MsgBoxResult.Yes Then
                    ReturnString = "EXIT"
                End If
            Else
                Dim result As MsgBoxResult = MessageBox.Show("Would you like to abort the positioner control?", "Box", MessageBoxButtons.YesNo)
                If result = MsgBoxResult.Yes Then
                    ReturnString = "EXIT"

                Else
                    ReturnString = "error"
                End If
            End If
        End Try
        Return ReturnString
    End Function

    Function GetVDC_and_Temp() As Boolean
        Dim vdcTemp As String = String.Empty

        _log.Debug(_main.LoggerComboBox.Text & " -Starting function 'GetVDC_and_Temp'.") 'write log file
        Try
            VDCTemp = _positioner.GetVoltageAndTemp()



            'Dim VDCTemp As String = QueryPTU300("O") 'O is command for retrieving VDC and Temp data
            Dim vdcTempPattern = "/* Input (?<VDC>\S+) VDC @ (?<Temp>\S+) degF, motors: pan (?<PanTemp>\S+) degF, tilt (?<TiltTemp>\S+).+"
            'Sample Reply  :       "O * Input 29 VDC @ 106 degF, motors: pan 105 degF, tilt 102 degF"
            '\s whitespace char     \S  non-whitespace char
            Dim vdcTempRegex As New System.Text.RegularExpressions.Regex(VDCTempPattern)
            Dim vdcTempMatch = VDCTempRegex.Match(VDCTemp)
            If VDCTempMatch.Success Then '# xxxx   
                VDCSupply = CInt(VDCTempMatch.Groups("VDC").Value.TrimEnd())
                PositionerTemp = CInt(VDCTempMatch.Groups("Temp").Value.TrimEnd())
                PanMotorTemp = CInt(VDCTempMatch.Groups("PanTemp").Value.TrimEnd())
                TiltMotorTemp = CInt(VDCTempMatch.Groups("TiltTemp").Value.TrimEnd())
            End If
        Catch ex As Exception
            _log.Error(_main.LoggerComboBox.Text & " -Error in function 'GetVDC_and_Temp': " & ex.Message)
            ErrorMessageFormat("Error getting voltage and temp.", ex)
        End Try
    End Function

                        Gauge1.Caption = "Azimuth(M) : " & Gauge1.Value.ToString("000.00")

            ErrorMessageFormat("Error updating pan position.", ex)
        Finally
            UpdateGauges()
        End Try

    End Sub

    Private Sub lblHandler_Click(ByVal sender As Object, ByVal e As EventArgs) Handles lblStartAzimuth.Click, lblStopAzimuth.Click, lblInterval.Click, lblTrueNorth.Click
        _log.Debug(_main.LoggerComboBox.Text & " -Sub 'lblHandler_Click()' triggered by event.")
        Dim proxy As Label = CType(sender, Label)
        Try
            Select Case Proxy.Text.ToUpper
                Case "CENTER"
                    txtStartRange.Text = Gauge1.Value.ToString
                Case "SPAN"
                    _rangeSpan = Double.Parse(txtStopRange.Text)
                    Gauge1.Value = Gauge1.Value + 1

                '  txtStopRange.Text = Gauge1.Value.ToString
                Case "START"
                    If Not Gauge1.Value >= Gauge1.RangeFills(0).Range.Maximum Then 'test new range IF 
                        Gauge1.RangeFills(0).Range = New Range(Gauge1.Value, Gauge1.RangeFills(0).Range.Maximum)
                        Gauge1.RangeFills(1).Visible = False
                        txtStartRange.Text = Gauge1.Value.ToString("000.00")
                    Else
                        Gauge1.RangeFills(0).Range = New Range(Gauge1.Value, 359.0#)
                        Gauge1.RangeFills(1).Visible = True
                        Gauge1.RangeFills(1).Range = New Range(0.0#, Gauge1.RangeFills(1).Range.Maximum)
                    End If
                Case "STOP"
                    If Not Gauge1.Value <= Gauge1.RangeFills(0).Range.Minimum Then 'test new range IF 
                        Gauge1.RangeFills(0).Range = New Range(Gauge1.RangeFills(0).Range.Minimum, Gauge1.Value)
                        Gauge1.RangeFills(1).Visible = False
                        txtStopRange.Text = Gauge1.Value.ToString("000.00")
                    Else
                        Gauge1.RangeFills(1).Visible = True
                        Gauge1.RangeFills(1).Range = New Range(0.0#, Gauge1.Value)
                        Gauge1.RangeFills(0).Range = New Range(Gauge1.RangeFills(0).Range.Minimum, 359.0#)
                    End If
                Case "N (T)", "N (MAG)"
                    txtTrueNorth.Text = Gauge1.Value.ToString
                    Gauge1.CustomDivisions(0).Value = Double.Parse(txtTrueNorth.Text)
                    If Double.Parse(txtTrueNorth.Text) + 180 < 360 Then
                        Gauge1.CustomDivisions(1).Value = Double.Parse(txtTrueNorth.Text) + 180
                    Else
                        Gauge1.CustomDivisions(1).Value = Double.Parse(txtTrueNorth.Text) + 180 - 360
                    End If
                    If Double.Parse(txtTrueNorth.Text) + 90 < 360 Then
                        Gauge1.CustomDivisions(2).Value = Double.Parse(txtTrueNorth.Text) + 90
                    Else
                        Gauge1.CustomDivisions(2).Value = Double.Parse(txtTrueNorth.Text) + 90 - 360

                If Not Gauge1.Value + Gauge1.CoercionInterval > 359.9999# Then

                Else
                    Gauge1.Value += Gauge1.CoercionInterval - 360
                End If

            Case "UP"
                If Not Gauge2.Value + Gauge2.CoercionInterval > 29.9999# Then
                    Gauge2.Value += Gauge2.CoercionInterval
                End If
                If Gauge2.Value > 270 Then
                    value = Gauge2.Value + Gauge2.CoercionInterval
                    If value = 360 Then value = 0
                    Gauge2.Value = value
                End If
            Case "DN"
                If Not Gauge2.Value - Gauge2.CoercionInterval < 270.0# Then
                    Gauge2.Value -= Gauge2.CoercionInterval
                End If
                If Gauge2.Value < 30 Then
                    value = Gauge2.Value - Gauge2.CoercionInterval
                    If value < 0 Then value += 360
                    Gauge2.Value = value
                End If
        End Select
    End Sub

    Private Sub UpdateGauges()

    Private Sub ErrorMessageFormat(ByVal message As String, ByVal ex As Exception)

        If ex.InnerException IsNot Nothing Then
            MsgBox(message & vbNewLine & "Message:" & vbNewLine & ex.Message, vbMsgBoxSetForeground)

        Else
            MsgBox(message & vbNewLine & "Message:" & vbNewLine & ex.Message & vbNewLine & "Inner Exception" & vbNewLine &
                   ex.InnerException.Message, vbMsgBoxSetForeground)
        End If
    End Sub
