Namespace MessageBox

    Public Module _Shared

        Public Function Show(message As String) As System.Windows.Forms.DialogResult
            Return Show(message, "Message...", MessageBoxButtons.OK, MessageBoxIcon.Information)
        End Function

        Public Function Show(message As String, icon As System.Windows.Forms.MessageBoxIcon) As System.Windows.Forms.DialogResult
            If icon = MessageBoxIcon.Error Then
                Return Show("Error: {0}".FormatWith(message), "Error...", MessageBoxButtons.OK, MessageBoxIcon.Error)
            End If
            Return Show(message, "Message...", MessageBoxButtons.OK, icon)
        End Function

        Public Function Show(message As String, title As String, buttons As System.Windows.Forms.MessageBoxButtons, icon As System.Windows.Forms.MessageBoxIcon) As System.Windows.Forms.DialogResult
            Dim Form As New frmMessageBox(message, title, buttons, icon)
            Return Form.ShowDialog()
        End Function

    End Module

End Namespace
