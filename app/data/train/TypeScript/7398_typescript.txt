export class IndexController {
    constructor() { }
    static home(req, res) {
        res.render('home', {
            title: 'Home',
            homeActive: true
        });
    }
    static login(req, res) {
        res.render('login', {
            title: 'Login',
            loginActive: true
        });
    }
    static register(req, res) {
        res.render('register', {
            title: 'Register',
            registerActive: true
        });
    }
    static notFound(req, res) {
        res.render('not-found');
    }
}
