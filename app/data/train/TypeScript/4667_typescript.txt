import { Action } from './action';

/**
 * Interface which defines the reducer service
 *
 * @export
 * @interface Reducer
 * @template T
 */
export interface Reducer<T> {
  /**
   * Method to update the state of the store
   *
   * @param {T} state - Current state in the store
   * @param {Action} action - Action to run on the current state
   * @returns {T} - New state as result of the action
   * @memberof Reducer
   */
  reduce(state: T, action: Action): T;
}
