declare module 'postgres-fill' {
  function PostgresFill(
    log: (message: string) => any,
    connection: {
      host: string,
      user: string,
      password: string,
      database: string,
      port: number,
    },
    structure: DBStructure[],
    verbose?: boolean
  ): Promise<any>;
  module PostgresFill {}

  export = PostgresFill;

  interface DBStructure {
    name: string,
    description?: string,
    disabled?: boolean,
    columns: {
      name: string,
      type: string,
      isNull?: boolean,
      disabled?: boolean,
      size?: number,
      primaryKey?: boolean,
      defaultValue?: any,
    }[],
    foreignKeys?: {
      column: string,
      referenceTable: string,
      referenceColumn: string,
      onDelete?: 'setNull'|'setDefault'|'restrict'|'cascade'|'noAction',
      onUpdate?: 'setNull'|'setDefault'|'restrict'|'cascade'|'noAction',
    }[],
    unique?: string[],
    index?: string[],
  }
}
