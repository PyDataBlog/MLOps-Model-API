export function* merge(...generators) {
  for (const generator of generators) {
    yield* generator;
  }
}

export function collect(generator) {
  const res = [];
  for (const el of generator) {
    console.log(el);
    res.push(el);
  }
  return res;
}

export function pickPrefixed<T>(obj: { [key: string]: T }, prefix: string): { [key: string]: T } {
  const newObj = Object.create(null);
  for (const key of Object.keys(obj))
    if (key.indexOf(prefix) === 0)
      newObj[key.substring(prefix.length)] = obj[key];
  return newObj;
}
