import IParticipantIdentity = require('./../../interfaces/match/participant-identity');
import Player = require('./player');


class ParticipantIdentity
{
    constructor(participantIdentityData : IParticipantIdentity) {
        this.participantId = participantIdentityData.participantId;
        this.player = new Player(participantIdentityData.player);
    }

    public participantId : number;
    public player : Player;

    toObject() : IParticipantIdentity {
        return {
            participantId: this.participantId,
            player: this.player.toObject()
        };
    }
}

export = ParticipantIdentity;
