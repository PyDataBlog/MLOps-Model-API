import { NgModule } from '@angular/core';
import { Routes, RouterModule } from '@angular/router';
import { BanComponent } from "app/quan-ly/ban/ban.component";
import { DanhSachBanComponent } from "app/quan-ly/ban/danh-sach/danh-sach.component";
import { ThemBanComponent } from "app/quan-ly/ban/them/them.component";
import { SuaBanComponent } from "app/quan-ly/ban/sua/sua.component";

const routes: Routes = [
  {
    path: '', component: BanComponent,
    children: [
      { path: 'danh-sach', component: DanhSachBanComponent },
      { path: 'them', component: ThemBanComponent },
      { path: 'sua/:id', component: SuaBanComponent },
    ]
  }
];

@NgModule({
    imports: [RouterModule.forChild(routes)],
    exports: [RouterModule]
})
export class BanRoutingModule { }
