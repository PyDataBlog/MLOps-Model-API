import {Graphics} from 'controller/tiling/graphics'
import Point from 'controller/geom/point'

export default class LevelPoint {
  tile :Graphics
  pos :Point
  seen :boolean
  persistent :boolean
  moving :boolean

  constructor(pos :Point,
              tile :Graphics,
              persistent :boolean,
              moving :boolean){
    this.pos = pos
    this.tile = tile
    this.persistent = persistent
    this.moving = moving
    this.seen = false
  }

  public getRenderedTile() :Graphics {
    if (this.seen) {
      return this.tile
    }
    return Graphics.Void
  }

  public getTile() :Graphics {
      return this.tile
  }

  public setTile(tile :Graphics){
    this.tile = tile
  }
}
