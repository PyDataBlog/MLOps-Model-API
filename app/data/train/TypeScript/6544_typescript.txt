import { Component, OnInit } from "@angular/core";
import { AuthService } from "../../services/auth.service";
import { Router } from "@angular/router";
import { FlashMessagesService } from "angular2-flash-messages";
import { ValidateService } from "../../services/validate.service";

import { User } from "../../models/user";

@Component({
  selector: "app-login",
  templateUrl: "./login.component.html",
  styleUrls: ["./login.component.css"]
})
export class LoginComponent implements OnInit {
  username: string;
  password: string;

  constructor(
    private authService: AuthService,
    private router: Router,
    private flashMessage: FlashMessagesService,
    private validateService: ValidateService
  ) {}

  ngOnInit() {}

  onLoginSubmit() {
    if (this.valid(this.username)) {
      const creds = {
        username: this.username,
        password: this.password
      };
      this.authUser(creds);
    } else {
      this.flashMessage.show(
        "Invalid Username: Use Alphanumeric Characters Only",
        { cssClass: "alert-danger", timeout: 10000 }
      );
      this.router.navigate(["/login"]);
    }
  }

  valid(field: string): boolean {
    return this.validateService.validateField(field);
  }

  authUser(creds: object) {
    this.authService.authenticateUser(creds).subscribe(data => {
      if (data.success) {
        this.authService.storeUserData(data.token, data.user);
        this.flashMessage.show("Login Successful", {
          cssClass: "alert-success",
          timeout: 2500
        });
        this.router.navigate(["/"]);
      } else {
        this.flashMessage.show(data.msg, {
          cssClass: "alert-danger",
          timeout: 10000
        });
        this.router.navigate(["/login"]);
      }
    });
  }
}
