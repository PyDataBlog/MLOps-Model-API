import { User } from '@api/auth';
import { Column, Entity, ManyToOne, PrimaryColumn } from 'typeorm';

/**
 * Represents a light novel editor.
 */
@Entity()
export class Editor {
  @PrimaryColumn('uuid')
  id: string;
  @Column()
  name: string;
  @Column()
  website: string;
  @ManyToOne(t => User)
  associatedUser: User;

  constructor(props: Partial<Editor>) {
    Object.assign(this, props);
  }
}
