import { Routes, RouterModule } from '@angular/router';
import { DashboardComponent } from "./Dashboard/dashboard.component";

const appRoutes: Routes = [
	{
		path: '',
		redirectTo: '/dashboard',
		pathMatch: 'full'

		path: 'dashboard',

	}
