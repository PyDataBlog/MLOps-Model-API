import {CollectionObject} from './collection-object.model';

export interface Project extends CollectionObject{
  developer_id: string;
  developerName?: string;
  description?: string;
  name: string;
  webPage?: string;
  owner?: string;
  logo?: string;
  phone?: string;
  facebook?: string;
  location: Location;
  images?: string[];
  active: boolean;
  lastModified: Date;
  email?: string;
  priceStart?: string;
}

interface Location {
  address?: string;
  province?: string;
  canton?: string;
  lat?: number;
  lng?: number;
}
