import {Pipe, PipeTransform} from '@angular/core';
import { Images } from '../../../../both/collections/images.collection';
import { Track } from '../../../../both/models/track.model';

@Pipe({
  name: 'displayMainImage'
})
export class DisplayMainImagePipe implements PipeTransform {
  transform(track: Track) {
    if (!track) {
      return;
    }

    let imageUrl: string;
    let imageId: string = (track.images || [])[0];

    const found = Images.findOne(imageId);

    if (found) {
      imageUrl = found.url;
    }

    return imageUrl;
  }
}
