import {
  parse,
  format,
} from 'url'
import {
  stringify,
  parse as parseQuery,
} from 'querystring'

const parsed = parse('https://naver.com/?hello=world&foo=bar')
console.log(parsed)
const query = parseQuery(parsed.query)
query.add = 'add'
parsed.search = '?' + stringify(query)
console.log(format(parsed))
