import {Pipe, PipeTransform} from '@angular/core';
import { TopicObject } from './topic.model';

@Pipe({
 name: 'weightSorting'
})
export class TopicWeightSort implements PipeTransform{

 transform(array: Array<TopicObject>, args: string): Array<TopicObject> {

  if(!array || array === undefined || array.length === 0) return null;

    array.sort((a: TopicObject, b: TopicObject) => {
      if (a.weight > b.weight) {
        return -1;
      } else if (a.weight < b.weight) {
        return 1;
      } else {
        return 0;
      }
    });
    return array;
  }

}
