    angular

        .config([

            (   $stateProvider: angular.ui.IStateProvider,
                $urlRouterProvider: angular.ui.IUrlRouterProvider
            ) => {
                    return new AppRouteConfig($stateProvider, $urlRouterProvider);
            }
        ]);
}
