export interface IState{
    id:number;
    name:string;
    uf:string;
}

export class StateEntity implements IState{
    id:number;
    name:string;
    uf:string;

    constructor(data?:any) {
        if (data) {
            this.id = data.id || null;
            this.name = data.name || null;
            this.uf = data.ud || null;
        }
    }
}
