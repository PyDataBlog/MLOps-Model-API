package com.bjorktech.cayman.idea.designpattern.structure.proxy;

public class TargetClass implements TargetInterface {

	@Override
	public long add(long a, long b) {
		long temp = a + b;
		System.out.println(temp);
		return temp;
	}

	@Override
	public long sub(long a, long b) {
		long temp = a - b;
		System.out.println(temp);
		return temp;
	}

}
