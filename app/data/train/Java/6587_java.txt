package net.xiguawanou.ingwer;

import org.junit.Assert;

public class TestUtility {
	public static void assertAnimationEquals(Animation animation,
			int startFrame, int endFrame, float fps) {
		Assert.assertEquals(startFrame, animation.getStartFrame());
		Assert.assertEquals(endFrame, animation.getEndFrame());
		Assert.assertEquals(fps, animation.getFrameRate(), 0.001f);
	}
}
