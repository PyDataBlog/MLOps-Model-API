using System;
using System.Collections.Generic;
using System.Data;
using System.Linq;
using BlogForFZO.DataCommon;
using Dapper;

namespace BlogForFZO.DataCommon
{
    public class BaseDAL<T> where T:class, new()
    {
        public IDbConnection conn = ConnFactory.GetConnection();
        public virtual int ExecuteSql(string sql,T t)
        {
            using(conn)
            {
                conn.Open();
                return conn.Execute(sql, t);
            }
        }

        public virtual IEnumerable<T> Query(string sql, Object param = null)
        {
            using(conn)
            {
                conn.Open();
                return conn.Query<T>(sql, param);
            }
        }
    }
}
