using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;
using Treinamento.Classes;
using Treinamento.Codigos;

namespace Treinamento.Interfaces.PerfisView
{
    public partial class Cadastra : System.Web.UI.Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {

        }

        protected void btnSalvar_Click(object sender, EventArgs e)
        {
            Perfil perfil = new Perfil
            {
                Nome = tbNome.Text
            };
            lblMensagem.Text = PerfilController.Insert(perfil);
            Limpar();
        }

        private void Limpar()
        {
            tbNome.Text = "";
        }
    }
}
