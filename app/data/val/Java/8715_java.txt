package coordinateSystem;

public class Point {
	private double x;
	private double y;

	public Point() {
		this.setX(0);
		this.setY(0);
	}

	public double getX() {
		return x;
	}

	public void setX(double x) {
		this.x = x;
	}

	public double getY() {
		return y;
	}

	public void setY(double y) {
		this.y = y;
	}
}
