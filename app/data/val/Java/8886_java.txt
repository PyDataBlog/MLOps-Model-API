package pers.wzq.hb.util.table;

import org.apache.poi.ss.usermodel.Cell;

/**
 * excel表格格式化接口
 * 
 * @author RP_S
 * @since 2017年9月28日
 */
public interface CellFormatter {
	/**
	 * 使用自定义方式，格式化表格内容<br>
	 * 可在table表格上自定义属性，编程转换成excel单元格样式
	 * 
	 * @param tableCell table表格
	 * @param cell excel表格
	 */
	void formatCell(TableCell tableCell, Cell cell);
}
