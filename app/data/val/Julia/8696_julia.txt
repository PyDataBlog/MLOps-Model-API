module ImgCreater

export
  chessboardimg,
  sinimg

function chessboardimg(m::Int, n::Int, blkwidth::Int, blkheight::Int, )
  img = zeros(m, n)

  togglex = true
  toggley = !togglex
  for i=1:m
    togglex = true
    for j=1:n
      img[i, j] = abs(togglex - toggley)
      if j % blkwidth <= 0
        togglex = !togglex
      end
    end
    if i % blkheight <= 0
      toggley = !toggley
    end
  end
  return img
end


function sinimg(m::Int64, n::Int64, freq::Int64, phase::Float64 = 0.0)
  img = zeros(m, n)

  for i=1:m
    for j=1:n
      img[i, j] = sin((2*pi*freq*(j-1)/n) + phase)
    end
  end
  return img
end

end # module
