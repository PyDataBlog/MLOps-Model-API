function foo($arr){
    //将数组中所有的键名都转成大写
    $re= array_change_key_case($arr,CASE_UPPER);
    foreach ($re as $k=>$v){
        if(is_array($v)){
            $re[$k]=foo($v);
        }
    }
//    返回为了接收转成大写后的数据
    return $re;
}

$arr1=array('one'=>'dmmkk',
    'two'=>array(
        'three'=>'nfcjkdk',
        'four'=>array(
            'five'=>'djjfnjmd',
            'six'=>'anbfdh'
        )
    ),
);
print_r(foo($arr1));
