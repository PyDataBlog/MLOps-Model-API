namespace org\afterspring\test {
    use org\afterspring\io\File;

	require_once (__DIR__ . '/../../../../../../target/afterspring-core-test.phar');
    require_once ("simpletest/unit_tester.php");
    
    use UnitTestCase;  
    
    /**
     * 
     */
    class TestFile extends UnitTestCase {
        
        /**
         * 
         */
        public function testFileClass() {
            $tempFile = sys_get_temp_dir()."/testfile.tmp";
            file_put_contents($tempFile, "ASDF123");
            
            $file = new File($tempFile);
            $this->assertEqual($file->getFilename(), "testfile.tmp");
            $this->assertEqual($file->getFilepath(), $tempFile);
            $this->assertEqual($file->getFilesize(), filesize($tempFile));
            $this->assertEqual($file->getContent(), "ASDF123");
            
            unlink($tempFile);
        }
        
        /**
         * 
         */
        public function __construct() {
        }
    }
}
