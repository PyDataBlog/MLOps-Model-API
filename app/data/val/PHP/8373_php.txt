<?php
namespace f2r\Fydrate\Adapter\Mysql;

use f2r\Fydrate\Adapter\StatementInterface;

class Statement implements StatementInterface
{
    /**
     * @var \mysqli_result
     */
    protected $statement;

    public function __construct(\mysqli_result $resullt)
    {
        $this->statement = $resullt;
    }

    /**
     * @return $this
     */
    public function rewind()
    {
        $this->statement->data_seek(0);
        return $this;
    }

    /**
     * @return array
     */
    public function fetch()
    {
        return $this->statement->fetch_assoc();
    }

    /**
     * @return int
     */
    public function count()
    {
        return $this->statement->num_rows;
    }
}
