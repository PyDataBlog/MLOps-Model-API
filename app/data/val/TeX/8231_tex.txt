\section{?`Como construimos informaci\'on a partir de una imagen satelital?}
\subsection{P\'ixeles}

\begin{frame}
    \frametitle{\secname-\subsecname}
    Comenzaremos a pensar a los pixeles como un conjunto de numeros. 
    Este conjunto tendra dos partes
    \begin{itemize}
        \item Una parte espacial
        \item Una parte espectral
    \end{itemize}
    y vamos a realizar operaciones con estos numeros.
\end{frame}

\begin{frame}
    \frametitle{\secname-\subsecname}
    \begin{block}{Observaci\'on}
        Nos vamos a centrar, en las proximas clases, en trabajar con la parte
        espectral de los p\'ixeles.
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{\secname-\subsecname}
    \begin{block}{Observacion}

    \end{block}
\end{frame}

    \'area de interes.

\begin{frame}
    \frametitle{\secname-\subsecname}
    \begin{block}{Observacion}
        Todos los cambios en la firma espectral de la vegetacion estan asociados
        a fenomenos biofisicos. Y a que especies estamos hablando.
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{\secname-\subsecname}
    \begin{block}{Simple ratio}
        El primer indice que uno puede construir es la relacion entre la
        absorcion de la clorofila y la reflectancia debido a la estructura
        celular
        \begin{equation}
            SR = \frac{\rho_{NIR}}{\rho_{rojo}}
        \end{equation}
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{\secname-\subsecname}
    \begin{block}{NDVI}
        Es una mejora del simple ratio normalizado por diferencia
        \begin{equation}
            NDVI = \frac{\rho_{NIR}-\rho_{rojo}}{\rho_{NIR}+\rho_{rojo}} 
        \end{equation}
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{\secname-\subsecname}
    \begin{figure}
    \begin{center}
        \includegraphics[width=0.8\textwidth]{Figures/avndvi.png}
    \end{center}
    \caption{NDVI vs Biomasa. El NDVI correlaciona
        con la cantidad de biomasa humeda.  }
    \end{figure}
\end{frame}

\begin{frame}
    \frametitle{\secname-\subsecname}
    \begin{block}{EVI}
        Creado como una mejora del NDVI para usar con MODIS, tiene 3 parametros
        que deben determinarse empiricamente, $C_1=6.0$, $C_2=7.5$ y $L=1.0$. $G$ es un
        factor de ganancia..
        \begin{equation}
            EVI =
            G\times\frac{\rho_{NIR}-\rho_{rojo}}{\rho_{nir}+C_1\rho_{rojo}-C_2\rho_{azul}+L}(1+L) 
        \end{equation}
        Tiene un mayor rango dinamico que el NDVI\@.
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{\secname-\subsecname}
    \begin{block}{NDWI}
        Relaciona el contenido de humedad de las plantas con el area foliar
         \begin{equation}
            NDWI = \frac{\rho_{NIR}-\rho_{SWIR1}}{\rho_{NIR}+\rho_{SWIR1}} 
        \end{equation}
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{\secname-\subsecname}
    Cada uno de los indices relaciona distintas propiedades de absorcion o
    refleccion para obtener variables biofisicas de la vegetacion.
    Permiten analizar por separado, y en forma cuantitativa, distintos fenomenos
    que se producen en la vegetacion.
\end{frame}

\begin{frame}
    \frametitle{\secname-\subsecname}
    Nuevamente a partir de la firma espectral, nos inspiramos para construir
    indices que nos hablen de distintas propiedades.
\end{frame}

\begin{frame}
    \frametitle{\secname-\subsecname}
    \begin{block}{Batimetria}
        Analizando la radiancia obtenida para dos longitudes de onda cercanas es
        posible estimar la profundidad de un cuerpo de agua, mientras el mismo
        presente pocos sedimentos
        \begin{equation}
            z = m_0 \log \frac{L(\lambda_i)}{(L(\lambda_j)} - m_1
        \end{equation}
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{\secname-\subsecname}
    \begin{figure}
    \begin{center}
        \includegraphics[width=0.8\textwidth]{Figures/snow.png}
    \end{center}
    \caption{Comparacion de la firma espectral de nieve, nubes y agua.}
    \end{figure}
\end{frame}

\begin{frame}
    \frametitle{\secname-\subsecname}
    \begin{block}{NDSI}
        Este indice permite detectar y separar nieve de nubes.
        \begin{equation}
            NDSI = \frac{\rho_{verde}-\rho_{SWIR1}}{\rho_{verde}+\rho_{SWIR1}} 
        \end{equation}
    \end{block}
\end{frame}

        espectrales de suelo con distinto contenido de vegetacion.

        Es posible invertir estas relaciones para obtener la cantidad vegetacion
        en cada pixel.
