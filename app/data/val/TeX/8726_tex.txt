\enabstract{Gerich~M.}%
{Clarification of basic factorization identity is for the almost semi-continuous latticed Poisson processes on the Markov chain}%
{Let $\{\xi(t), x(t)\}$ be a homogeneous semi-continuous lattice  Poisson process on the Markov chain.
The jumps of one sign are geometrically distributed, and jumps of the opposite sign are arbitrary latticed distribution. For a such
processes the relations for the components of two-sided matrix factorization are established.
This relations define the moment genereting functions for extremumf of the process and their complements.
}

\textbf{p}_{x}^{-}(s)=[\textbf{p}_{x}^{-}(s)+\textbf{E}[\textbf{C}\,^{\mid\xi(\theta_{s})\mid}, \, \xi(\theta_{s})<x]
\cdot\textbf{C}\,^{x-1}(\textbf{C}-\textbf{Q}_{s}^{-1})](\textbf{p}^{+}(s))^{-1}\textbf{P}_{s}, \,\, x\in \mathbb{Z}^{-}\cup \{0\},
\end{equation}
äå $\mathbf{Q}_{s}^{-1}=\textbf{P}_{s}^{-1}\textbf{q}^{+}(s)+\textbf{C}\,\textbf{P}_{s}^{-1}\textbf{p}^{+}(s)$.
\end{theorem}
\begin{proof}
Íà îñíîâ³ ñòîõàñòè÷íèõ ñï³ââ³äíîøåíü äëÿ $\tau_{kr}^{-}(x)$, ($x\in \mathbb{Z}_{-}\cup \{0\}$), äå íèæí³ ³íäåêñè
îçíà÷àþòü ïî÷àòêîâå çíà÷åííÿ  $x(t)$  òà çíà÷åííÿ $x(t)$ â ìîìåíò äîñÿãíåííÿ
ð³âíÿ  $x$ â³äïîâ³äíî ($x(0)=k, \,\, x(\tau^{-}(x))=r$). Òðåáà çàçíà÷èòè, ùî
ðîçãëÿäàþòüñÿ
ò³ëüêè ò³ òðàºêòîð³¿ ïðîöåñó, äëÿ ÿêèõ  $\tau^{-}(x)<\infty$
      $$
\tau_{kr}^{-}(x)\doteq
\left\{
\begin{array}{c}
\zeta'_{k}, \,\,\,\,  \xi_{1}^{(k)}<x, \,\, \zeta'_{k}<\zeta_{k};\\
\zeta_{k}, \,\,\,\,  \chi_{kr}<x, \,\, \zeta'_{k}>\zeta_{k};\\
\zeta'_{k}+\tau_{kr}^{-}(x-\xi_{k}), \,\,  \xi_{1}^{(k)}\geq x, \,\, \zeta'_{k}<\zeta_{k};\\
\zeta_{k}+\tau_{kr}^{-}(x-\chi_{kj}), \,\,  \chi_{kj}\geq x, \,\,
\zeta'_{k}>\zeta_{k}, \,\, (x(\zeta_{k})=j).
\end{array}
\right.
$$
Íà îñíîâ³ öèõ ñòîõàñòè÷íèõ ñï³ââ³äíîøåíü âèâîäèìî  ð³âíÿííÿ
\begin{equation}\label{fvst220}
\begin{split}
   T_{\ast kr}^{-}(s,x)&=E[e^{-s\tau^{\pm}(x)}, \,\, x(\tau^{\pm}(x))=r
| x(0)=k]  \\
&=\lambda_{k}\int_{0}^{+\infty}e^{-(s+\lambda_{k}+n_{k})y}dy \sum
\limits_{l=-\infty}^{x-1} P\{\xi_{k}=l\}  \\
   &+n_{k}\int_{0}^{+\infty}e^{-(s+\lambda_{k}+n_{k})y}dy
\sum\limits_{l=-\infty}^{x-1} p_{kr}P\{\chi_{kr}=l\}  \\
    &+\lambda_{k}\int_{0}^{+\infty}e^{-(s+\lambda_{k}+n_{k})y}dy
\sum\limits_{l=x}^{+\infty} e^{-s\tau_{kr}^{-}(x-l)}P\{\xi_{k}=l\}  \\
   &+n_{k}\sum\limits_{j=1}^{n}\int_{0}^{+\infty}e^{-(s+\lambda_{k}+n_{k})y}dy
\sum\limits_{l=x}^{+\infty} e^{-s\tau_{kr}^{-}(x-l)}p_{kj}P\{\chi_{kj}=l\}.
\end{split}
\end{equation}
Çàïèøåìî  (\ref{fvst220}) â ìàòðè÷í³é ôîðì³:
\begin{equation}
\label{fvst221}
(s\mathbf{I}+\mathbf{\Lambda}+\mathbf{N})\mathbf{T}_{\ast}^{-}(s,x)=\sum_{l\,=-\infty}^{x-1}\left(\mathbf{\Lambda p}(l)+\mathbf{Nf}(l)\right)
+\sum_{l\,=x}^{+\infty}\left(\mathbf{\Lambda p}(l)+\mathbf{Nf}(l)\right)\mathbf{T}_{\ast}^{-}(s,x-l).
\end{equation}
Çàñòîñîâóþ÷è äî (\ref{fvst221}) òâ³ðíå ïåðåòâîðåííÿ ïî
$x\in \mathbb{Z}_{-}\cup \{0\}$
òà âðàõóâàâøè óìîâó ìàéæå íàï³âíåïåðåðâíîñò³ çâåðõó, îòðèìàºìî
\begin{eqnarray}
\label{fvst222}
(s\mathbf{I}-\mathbf{K}(z))\widetilde{\mathbf{T}}_{\ast}^{-}(s,z)=\frac{z}{z-1}[
-\mathbf{\Lambda}_{2}(\widetilde{\mathbf{p}}_{2}(z)-\mathbf{I})-\mathbf{N}
(\widetilde{\mathbf{f}}(z)-\mathbf{P})]\nonumber\\
      -\mathbf{\Lambda}_{1}(\mathbf{I}-\mathbf{C})z(\mathbf{I}-\mathbf{C}z)^{-1}\widetilde{\mathbf{T}}_{\ast}^{-}(s,\, \mathbf{C}\,^{-1}),
\end{eqnarray}
äå
$$
\widetilde{\mathbf{T}}_{\ast}^{-}\left(s,z\right)=\sum_{x=-\infty}^{0}z^{x}\mathbf{T}_{\ast}^{-}(s,x)=\frac{z}{z-1}(\mathbf{I}-\mathbf{g}_{-}(s,z)\mathbf{P}_{s}^{-1}).
$$
Ï³äñòàâëÿþ÷è â ($\ref{fvst222}$) îñòàííº ñï³ââ³äíîøåííÿ òà (\ref{fvst22}),
îòðèìàºìî
\begin{eqnarray}
\label{fvst223}
      (s\mathbf{I}-\mathbf{K}(z))\textbf{g}_{-}(s,z)=[s\mathbf{I}+\mathbf{\Lambda}_{1}(1-z)(\mathbf{I}-\mathbf{C}z)^{-1}\mathbf{g}_{-}(s,\mathbf{C}^{-1})].
\end{eqnarray}
Ç (\ref{fvst223}) ï³ñëÿ âðàõóâàííÿ äðóãîãî ñï³ââ³äíîøåííÿ (\ref{fvst23}), îäåðæèìî,
     \begin{eqnarray}
\label{fvst224}
\textbf{g}^{+}(s,z)=\textbf{P}_{s}[\mathbf{I}+\mathbf{\Lambda}_{1}s^{-1}(1-z)(\mathbf{I}-\mathbf{C}z)^{-1}\mathbf{g}_{-}(s,\mathbf{C}^{-1})]^{-1}.
     \end{eqnarray}
Ï³ñëÿ ãðàíè÷íîãî ïåðåõîäó ïðè $z\rightarrow 0$ â (\ref{fvst223}), îòðèìàºìî
\begin{eqnarray}
\label{fvst225}
\textbf{p}^{+}(s)=\textbf{P}_{s}[\mathbf{I}+\mathbf{\Lambda}_{1}s^{-1}\mathbf{g}_{-}(s,\mathbf{C}^{-1})]^{-1}.
\end{eqnarray}
Ï³äñòàâëÿþ÷è (\ref{fvst225}) â ôîðìóëó (\ref{fvst224}), îäåðæèìî (\ref{fvst215}). Ï³ñëÿ îáåðòàííÿ (\ref{fvst215}) ïî $z$ âèïëèâàº ôîðìóëà (\ref{fvst216}).
Âèêîðèñòîâóþ÷è (\ref{fvst215}) òà äðóãèé ðÿäîê (\ref{fvst23}), îòðèìàºìî,
ïîïåðåäíüî çàñòîñóâàâøè îïåðàö³þ ïðîåêòóâàííÿ íà $[\,\,]_{-}^{0}$,
ñï³ââ³äíîøåííÿ (\ref{fvst218}).
Ôîðìóëà (\ref{fvst219}) âèâîäèòüñÿ ç (\ref{fvst218}) çà äîïîìîãîþ îáåðòàííÿ ïî $z$.
\end{proof}

Ç ïîïåðåäíüî¿ òåîðåìè, ç âðàõóâàííÿì óìîâè íàï³âíåïåðåðâíîñò³ çâåðõó, âèïëèâàº íàñòóïíèé íàñë³äîê.
\begin{corollary}
Äëÿ íàï³âíåïåðåðâíèõ çâåðõó ïðîöåñ³â ãåíåðàòðèñà òà ðîçïîä³ë
$\check{\xi}(\theta_{s})$ âèçíà÷àþòüñÿ ñï³ââ³äíîøåííÿìè:
\[
\textbf{g}^{+}(s,z)=\textbf{p}^{+}(s)[\mathbf{I}-\textbf{Q}_{s}^{-1}z]^{-1},
\]
\[
\check{\textbf{p}}_{x}^{+}(s)=\textbf{p}_{+}(s)\mathbf{Q}_{s}^{-x}, \,\, x\in \mathbb{Z}^{+},
\]
\[
\textbf{p}^{+}(s)=\textbf{P}_{s}(\mathbf{I}-\mathbf{Q}_{s}^{-1});
\]
äëÿ  $\xi^{-}(\theta_{s})$ ìàºìî:
\[
\textbf{g}_{-}(s,z)=([\textbf{g}(s,z)]_{-}^{0}-[\textbf{g}(s,z)]_{-}\mathbf{Q}_{s}^{-1}z)(\textbf{p}_{+}(s))^{-1}\textbf{P}_{s},
\]
\[
\textbf{p}_{x}^{-}(s)=(\textbf{p}_{x}(s)-\textbf{p}_{x-1}(s)\textbf{Q}_{s}^{-1})(\textbf{p}_{+}(s))^{-1}\textbf{P}_{s}, \,\, x\in\mathbb{Z}^{-}\cup\{0\},
\]
äå $\mathbf{Q}_{s}^{-1}=\textbf{P}_{s}^{-1}\textbf{q}^{+}(s)$.
\end{corollary}
Íåõàé $Z(t)=\{\xi(t), \, x(t)\}$ --- ìàéæå íàï³âíåïåðåðâíèé çíèçó ïðîöåñ, òîä³
$Z_{1}(t)=\{\xi_{1}(t), \, x_{1}(t)\}=\{-\xi(t), \, x(t)\}$
º ìàéæå íàï³âíåïåðåðâíèì çâåðõó. Âèêîðèñòîâóþ÷è íà\-ñòóï\-í³ ñï³ââ³äíîøåííÿ ì³æ
ãåíåðàòðèñàìè åêñòðåìóì³â ïðîöåñ³â $Z(t)$ òà $Z_{1}(t)$ òà ¿õ ðîçïîä³ë³â:
\[
\textbf{g}_{\mp}(s, z) =\textbf{g}_{\pm}^{1}(s,z^{-1}), \,\, \textbf{g}^{\mp}(s,z)=\textbf{g}_{1}^{\pm}(s,z^{-1}),
\]
\[
\textbf{p}_{\mp}(s) =\textbf{p}_{\pm}^{1}(s), \,\, \textbf{p}^{\mp}(s)=\textbf{p}_{1}^{\pm}(s),
\]
\[
\textbf{q}_{\mp}(s) =\textbf{q}_{\pm}^{1}(s), \,\, \textbf{q}^{\mp}(s)=\textbf{q}_{1}^{\pm}(s),
\]
\[
\textbf{p}_{x}^{\mp}(s) =(\textbf{p}_{-x}^{\pm}(s))_{1}, \,\, \check{\textbf{p}}^{+}_{x}(s)=(\check{\textbf{p}}_{-x}^{-}(s))_{1}, \,\,
\check{\textbf{p}}^{-}_{x}(s)=(\check{\textbf{p}}_{-x}^{+}(s))_{1},
\]
ìîæíà îòðèìàòè òâåðäæåííÿ ïðî óòî÷íåííÿ êîìïîíåíò ôàêòîðèçàö³¿ (3) äëÿ ìàéæå
íàï³âíåïåðåðâíèõ çíèçó ïðîöåñ³â.
\begin{theorem}
Äëÿ ìàéæå íàï³âíåïåðåðâíîãî çíèçó ïðîöåñó, çàäàíîãî íà ËÌ, ìàþòü ì³ñöå

äëÿ $\xi^{-}(\theta_{s})$ ìàºìî:
\[
\textbf{g}_{-}(s,z)=(z\textbf{I}-\textbf{B})(z\textbf{I}-\textbf{Z}_{s})^{-1}\textbf{p}_{-}(s),
\]

\textbf{g}^{+}(s,z)=\textbf{P}_{s}(\textbf{p}_{-}(s))^{-1}([\textbf{g}(s,z)]_{+}
^{0}
\]
\[
+(\textbf{B}-\mathbf{Z}_{s})(\textbf{B}-\textbf{I}z)^{-1}\textbf{E}[\textbf{B}\,^{\xi(\theta_{s})}-z^{\xi(\theta_{s})}, \,\, \xi(\theta_{s})>0]),

\begin{corollary}

\end{corollary}
