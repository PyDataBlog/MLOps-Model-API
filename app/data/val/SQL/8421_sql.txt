
SET statement_timeout = 0;
SET client_encoding = 'UTF8';
SET standard_conforming_strings = on;
SET check_function_bodies = true;
SET client_min_messages = warning;

\connect project_track

SET search_path = ptrack, pg_catalog;

CREATE OR REPLACE FUNCTION role__delete (
        _id integer )
    RETURNS boolean
    SECURITY DEFINER
    LANGUAGE plpgsql
    AS $$
BEGIN
    DELETE FROM pt_role
        WHERE id = _id ;
    RETURN true ;
END ;
$$ ;

ALTER FUNCTION role__delete ( integer ) OWNER TO ptrack_updater ;

GRANT EXECUTE ON FUNCTION role__delete(integer) TO ptrack_updater;
GRANT EXECUTE ON FUNCTION role__delete(integer) TO ptrack_user;
REVOKE ALL ON FUNCTION role__delete(integer) FROM public;
