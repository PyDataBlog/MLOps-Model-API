\connect central

INSERT INTO indicator_electronic_medical_record(indicator_id, electronic_medical_record_id)
SELECT
    i.id,
    (SELECT id FROM electronic_medical_record WHERE name ILIKE 'Wolf')
FROM indicator i
WHERE
    i.name LIKE 'Average Contact Counts%'
    or i.name LIKE 'Patient Counts%'
    or i.name LIKE 'Contact Counts%'
	ON CONFLICT DO NOTHING;
