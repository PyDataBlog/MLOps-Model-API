require("../core/core.module");

import { provide } from "../core";
import { LogoEditorComponent } from "./logo-editor.component";
import { LogoListComponent } from "./logo-list.component";
import { LogoComponent } from "./logo.component";
import { LogosContainerComponent } from "./logos-container.component";
import { LogoActionCreator } from "./logo.actions";
import { LogoService } from "./logo.service";
import *  as reducers from "./logo.reducers";

var app = (<any>angular.module("app.logo", [
    "app.core"    
]));

provide(app,LogoActionCreator);
provide(app,LogoService);
app.component(LogoEditorComponent);
app.component(LogoListComponent);
app.component(LogoComponent);
app.component(LogosContainerComponent);

app.config(["reducersProvider", reducersProvider => {	
    for (var reducer in reducers) { reducersProvider.configure(reducers[reducer]); }
}]);
