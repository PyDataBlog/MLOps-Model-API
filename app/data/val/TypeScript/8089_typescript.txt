import { NgModule } from '@angular/core';
import { RouterModule, Routes } from '@angular/router';

import { ThemeViewComponent } from './view/theme/themeView.component';
import { GridSectionComponent } from './view/theme/sections/gridSection/gridSection.component';

const routes: Routes = [
    { path: '', redirectTo: '/theme', pathMatch: 'full' },
    { path: 'theme', component: ThemeViewComponent },
    { path: 'gridDetail/:name', component: GridSectionComponent }
];

@NgModule({
    imports: [RouterModule.forRoot(routes, { useHash: true })],
    exports: [RouterModule]
})

export class AppRoutingModule { }
