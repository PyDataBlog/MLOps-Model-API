%define	BOOTSECT
%include "..\include\defs.h"

bits 16
org 0x7c00

; ---------------------------------------------- ;

%macro write 3				; write(buf,len,color)
	mov 	bp,%1
	mov 	cx,%2
	mov 	bx,%3
	call 	write_proc
%endmacro


; ---------------------------------------------- ;

__entry_point:
  jmp start			; [0x00] Áåçóñëîâíûé ïåðåõîä íà çàãðóçî÷íûé êîä 
  OEM 		db "DoOrS 95"	; [0x03] Àááðåâèàòóðà è íîìåð âåðñèè ÎÑ 
  SectSize 	dw 512 		; [0x0B] ×èñëî áàéò â ñåêòîðå
  ClustSize 	db 1 		; [0x0D] ×èñëî ñåêòîðîâ â êëàñòåðå
  Ressect 	dw 1 		; [0x0E] ×èñëî ðåçåðâíûõ ñåêòîðîâ â ðåçåðâíîé îáëàñòè ðàçäåëà, íà÷èíàÿ ñ ïåðâîãî ñåêòîðà ðàçäåëà 
  FatCnt 	db 2 		; [0x10] ×èñëî êîïèé FAT 
  RootSize 	dw 224 		; [0x11] Êîëè÷åñòâî 32-áàéòíûõ äåñêðèïòîðîâ ôàéëîâ â êîðíåâîì êàòàëîãå
  TotalSect 	dw 2880 	; [0x13] Îáùåå ÷èñëî ñåêòîðîâ 
  Media 	db 0xF0 	; [0x15] Òèï íîñèòåëÿ èíôîðìàöèè (Ãèáêèé äèñê: 2 ñòîðîíû, 18 ñåêòîðîâ íà äîðîæêå)
  FatSize 	dw 9 		; [0x16] Êîëè÷åñòâî ñåêòîðîâ, çàíèìàåìûõ FAT 
  TrackSect 	dw 18 		; [0x18] ×èñëî ñåêòîðîâ íà äîðîæêå 
  HeadCnt 	dw 2 		; [0x1A] ×èñëî ãîëîâîê 
  HidenSect 	dd 0 		; [0x1C] ×èñëî ñêðûòûõ ñåêòîðîâ ïåðåä ðàçäåëîì
  HugeSect 	dd 0 		; [0x20] Èñïîëüçóåòñÿ FAT32 
  BootDrv 	db 0 		; [0x24] Íîìåð äèñêîâîäà 
  Reserv 	db 0 		; [0x25] Çàðåçåðâèðîâàíî äëÿ Windows NT
  BootSign 	db 0x29 	; [0x26] Ïðèçíàê ðàñøèðåííîé çàãðóçî÷íîé çàïèñè 
  VolID 	dd 0 		; [0x27] Íîìåð ëîãè÷åñêîãî äèñêà 
  VoLabel 	db "System     "; [0x2B] Ìåòêà äèñêà 
  FSType 	db "FAT12   " 	; [0x36] Àááðåâèàòóðà òèïà ôàéëîâîé ñèñòåìû   
  
; ---------------------------------------------- ;  
  
start:  
  	cli
  	mov	ax,cs
  	mov	ds,ax
  	mov	ss,ax
  	mov	sp,__entry_point
  	sti
    
%include "bootlogo.asm"
 
clrscr:					; Î÷èñòèì ýêðàí
	mov 	ax,0x03
	int 	vid
  
load_kernel:				; Ãðóçèì ÿäðî - ïåðâûé ôàéë â êîðíåâîì êàòàëîãå
	mov	ax,kernseg
	mov	es,ax			; Òî÷êà âõîäà - [kernseg:0] di=0
     		; 33 ñåêòîð - çäåñü íà÷èíàåòñÿ ÿäðî =))
    mov dx,33
	
nextsect:    
   
	push	dx
	push 	es
	push	di
	call	readsect
	pop	di
	pop	es
	pop	dx
   
	add	di,512
	inc	dx
	cmp	dx,160			; Âñåãî (160-33+1)*512 = 128 ñåêòîðîâ = 64 êá
	jne	nextsect
	
; --------- À ÿäðî ëè ìû çàãðóçèëè? =) --------- ;  	

	mov	di,2
	mov	ax,kernseg
	mov	es,ax
	mov	ax, word [es:di]
	cmp	ax,word 0xABCD
	jne	readerror

	jmp	kernseg:0		; Ïåðåäàåì óïðàâëåíèå íà ñ÷èòàííûé êîä
  
; ----- Ïåðåâîä ¹ ñåêòîðà â ïàðàìåòðû BIOS ----- ;   

readsect:
	push	di
	push  	es
	push  	cs
	pop   	ds
	mov   	cx,[TrackSect]	; cx=Sectors
	mov   	si,dx		; si = Sect
  
; ------------- Tmp=(Sect/Sectors) ------------- ;

	mov	ax,si		; ax = Sect
	xor	dx,dx
	div	cx		; ax = Tmp, dx=?
	mov	di,ax		; di = Tmp
  
; ----------- Sec=Sect-(Tmp*Sectors)+1 --------- ; 

	imul	cx		; ax = Tmp*Sectors
	mov	dx,si		; dx = Sect
	sub	dx,ax		; dx = Sect-(Tmp*Sectors)
	inc	dx		; dx = Sec
	mov	[ABSSECT],dx	; ABSSECT = Sec
  
; ----------------- Hea=Tmp & 1 ---------------- ;  

	mov	ax,di		; ax = Tmp
	and	ax,1		; ax = Hea
	mov	[ABSHEAD],ax	; ABSHEAD = Hea
  
;  Trk=(Sect-(Hea*Sectors)-(Sec-1))/(Sectors*2)  ;

	imul	cx		; ax = Hea*Sectors
	push	ax
	mov	ax,si		; ax = Sect
	pop	dx		; dx = Hea*Sectors
	sub	ax,dx		; ax = Sect-(Hea*Sectors)
	mov	dx,[ABSSECT]	; dx = Sec
	dec	dx		; dx = Sec-1
	sub	ax,dx		; ax = Sect-(Hea*Sectors)-(Sec-1)
	mov	dx,cx		; dx = Sectors
	shl	dx,1		; dx = Sectors*2
	push	dx
	xor	dx,dx
	pop	bx		; bx = Sectors*2
	div	bx      	; ax = Trk

	mov	cx,ax
	mov	al,cl
	shr	cx,2
	and	cl,0x00C0
	mov	ch,al
	and	cx,0xFFC0
	mov	ax,[ABSSECT]
	or	cl,al
	pop	es
	pop	bx
 
	mov	dx,[ABSHEAD]	; dh - ãîëîâêà
	mov	dh,dl
	mov	dl,0		; dl - äèñê (0 äëÿ fd0)
	mov	al,1		; al - êîë-âî ñ÷èòûâàåìûõ ñåêòîðîâ
	mov	ah,0x02		; ×òåíèå ñ äèñêà
	int	disk
  
	jnc 	readdone	; Îøèáêà, åñëè CF=1, ah = êîä îøèáêè
  
readerror:
	mov	ax,cs		; Ñäåëàåì âñ¸ êàê áûëî =)
	mov	es,ax
	cmp	di,2
	je	kernel_not_found
	write	MSG1,2,cllight_red
	jmp	short exit
kernel_not_found:	
	write	MSG0,5,cllight_red
exit:
	xor	ax,ax
	int 	keyb		; Æä¸ì íàæàòèÿ êëàâèøè
	int 	0x19  
    
readdone:  
	ret
	
; ---------------------------------------------- ; 

write_proc:
	mov 	ax,0x1301
	int 	vid
	
	ret
	
; ------------------- ÄÀÍÍÛÅ ------------------- ;

	MSG0 db "Krnl?"				; ßäðî íå íàéäåíî
	MSG1 db "RE"				; Îøèáêà ÷òåíèÿ
	MSG2 db "Now Loading..."
	MSG3 db "*"
	MSG4 db "* DoOrS  '95 *"
	
	ABSSECT dw 0
	kern_sect dw 33
	ABSHEAD dw 0
	times 510-($-$$) db 0
    ;db 0x55,0xaa 	; Â êîíöå çàãðóçî÷íîãî ñåêòîðà äîëæíû
  	SIGN dw 0xAA55		; ñîäåðæàòüñÿ äàííûå 2 áàéòà
