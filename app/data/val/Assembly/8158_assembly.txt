	processor	pic16f690
	include		<p16f690.inc>

	global		set_pinout_tris

libot	code
set_pinout_tris:
	banksel		TRISA		
	movlw		B'00000000'	
	movwf		TRISA		
	movlw		B'00000000'	
	movwf		TRISB		
	movlw		B'00011111'	
	movwf		TRISC		
	banksel		PORTA		
	clrf		PORTA		
	clrf		PORTB		
	clrf		PORTC		
	return
	end
