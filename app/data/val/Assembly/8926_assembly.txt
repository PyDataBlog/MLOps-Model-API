;; Title: BUILDING MACROS
;; -----------------------

READ MACRO
;; MACRO FOR READING INPUT FROM KEYBOARD
;; THE CHARACTER'S ASCII IS RETURNED THROUGH
;; THE AL REGISTER
	 MOV AH,8
	 INT 21H
ENDM

PRINT MACRO CHAR
;; MACRO THAT PRINTS A CHARACTER TO SCREEN
;; THE CHAR PARAMETER SHOULD BE DEFINED BEFORE
;; THE MARCO GETS COMPILED
	  MOV DL,CHAR
	  MOV AH,2
	  INT 21
ENDM

PRINT_STR MACRO STRING
;; MACRO THAT PRINTS A STRING OF CHARACTERS
	   MOV DX,OFFSET STING
	   MOV AH,9
	   INT 21H
ENDM

EXIT MACRO
;; MACRO USED FOR PROGRAM EXIT
	   MOV AX,4C00H
	   INT 21H
ENDM
