class Score
  include Mongoid::Document
  include Mongoid::Timestamps
  include Mongoid::Autoinc

  belongs_to :quiz_instance, foreign_key: :quiz_instance_id

  field :correct_answers, type: Integer, :default => 0
  field :incorrect_answers, type: Integer, :default => 0
  field :hints_taken, type: Integer, :default => 0
  field :hints_total, type: Integer, :default => 0
end
