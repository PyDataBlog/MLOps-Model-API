# Init script to setup $HOME

cd $HOME
# install oh my zsh
wget --no-check-certificate \
https://raw.github.com/robbyrussell/oh-my-zsh/master/tools/install.sh \
-O - | sh

# ok, now get everything
if [ -z "$TMP" ]; then
    export TMP=/tmp
fi

cd $TMP
git clone --recursive https://github.com/paulollivier/dotfiles dotfiles
rm -rf dotfiles/.git
cp -r dotfiles/.* $HOME/
