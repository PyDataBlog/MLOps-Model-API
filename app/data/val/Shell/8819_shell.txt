#!/bin/sh

image () {

file ${1} |grep 'JPEG';
bool=$? ;

    case "${1}" in

    *\.jpg    )  if [ ${bool} -eq 0 ]
                 then echo "<img src='${1}'>"
		 else echo "<!-- ${1} ne vérifie pas le format de l'énoncé ou n'existe pas -->"
		 fi ;;

    * 	      )   echo "<!-- ${1} ne vérifie pas le format de l'énoncé ou n'existe pas -->" ;;

esac

}
