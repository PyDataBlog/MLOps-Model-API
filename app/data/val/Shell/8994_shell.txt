#!/bin/bash

protoc_path=$(which protoc)
if [ ! -x "$protoc_path" ]; then
    echo "ERROR: Protobuf Compiler not found"
    exit 1
fi


if [ ! -d "schema" ]; then
    echo "ERROR: Protobuf Schema not found"
    exit 1
fi

mkdir -p py-sportsbook
protoc --proto_path=schema --python_out=py-sportsbook schema/*.proto

touch py-sportsbook/__init__.py
