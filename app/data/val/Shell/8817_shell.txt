workspace=$(pwd)
appName=$(basename ${workspace})
env=$1

logDir=${workspace}/log
targetDir=${workspace}/target

echo "<<<<<<<<<<<<<<<<<<<<<<<<"
echo "build apidoc ..."
echo ">>>>>>>>>>>>>>>>>>>>>>>>"

apiTemplateDir=${workspace}/
apiName=${appName/%-demo/-doc}


apidoc -i ${workspace}/src/main/java/www/liujijiji/cn/vuespringboot/controller -o ${workspace}/apidoc -c ${apiTemplateDir}
