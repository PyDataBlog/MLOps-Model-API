package kumoi.shell.pm

import kumoi.shell.or._

/**
 * Statistical information of physical machines.
 * 
 * @author Akiyoshi SUGIKI
 */
abstract class PMStat extends UStat

case class FreeMem(memFree: Long) extends PMStat
case class MaxMem(memMax: Long) extends PMStat
case class FreeSwap(swapFree: Long) extends PMStat
case class MaxSwap(swapMax: Long) extends PMStat
case class Fds(swapFree: Long) extends PMStat
case class MaxFds(swapFree: Long) extends PMStat

case class CPURatio(user: Double, sys: Double, total: Double) extends PMStat
case class LoadAverage(load: Double) extends PMStat
