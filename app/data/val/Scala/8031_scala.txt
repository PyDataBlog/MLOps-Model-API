package com.transgee.ebook.pdf

private class ProcessingContext {

  private var id = 0

  private var currentPageIndex: Int = 0
  private var opLog: OpLog = null

  def nextId() = {
    id += 1
    id
  }

  def setOpLog(currentPageIndex: Int, opLog: OpLog) = {
    this.currentPageIndex = currentPageIndex
    this.opLog = opLog
  }

  def getOpLog(pageIndex: Int) = {
    if (currentPageIndex == pageIndex) opLog else null
  }

}
