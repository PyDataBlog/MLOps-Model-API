$filecontent = Get-Content "work\Jet Kanji.txt" -Raw -Encoding UTF8
$filecontent = "Number`tGrade`tKanji`tReading`tVocabulary`tNotes`tDiagram`tStrokeCount`tRadical`tReadings`n" + $filecontent
$input = $filecontent | ConvertFrom-CSV -Delimiter "`t"
$exp = "Kanji`tGrade`tReading`tVocabulary`n"
foreach ($card in $input) {
    $exp += $card.Kanji + "`t" + $card.Grade + "`t" + $card.Reading + "`t" + $card.Vocabulary
    $exp += "`n"
}
$exp | out-file work\Kanji.csv
