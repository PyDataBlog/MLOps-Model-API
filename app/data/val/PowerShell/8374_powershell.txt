<#
    .Synopsis
    Restore a window (bring to front).
    .Description
    Attempts to restore a window that may or may not have disappeared from the taskbar.
    .Parameter ProcessName
    The process name to find
    .Example
    .\BringToFront.ps1 -ProcessName dstermserv
#>

Param(
    [Parameter(Mandatory=$true,HelpMessage="Process (Task) Name")][string]$ProcessName
)

$swa = '[DllImport("user32.dll")] public static extern bool ShowWindowAsync(IntPtr hWnd, int nCmdShow);'
Add-Type -MemberDefinition $swa -Name NativeMethods -Namespace Win32

# Get list of processes matching $ProcessName
$processes = Get-Process "*$($ProcessName)*"
if ($processes.Length -eq 0)
{
    Write-Host "Could not find task: $($ProcessName)"
    exit
}

# Grab the first process that matches and attempt to restore it
$mwh = $processes[0].MainWindowHandle
$restored = [Win32.NativeMethods]::ShowWindowAsync($mwh, 4)
if ($restored)
{
    Write-Host "$($ProcessName) restored"
}
