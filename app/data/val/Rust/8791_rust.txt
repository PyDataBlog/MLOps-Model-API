#![feature(core)]
use euler::*; mod euler;
#[no_mangle]
pub extern "C" fn solution() -> EulerType {
	for a in range_inclusive(5, 500) {
		let r = 1000 - a;
		for b in range_inclusive(a + 1, r) {
			let c = r - b;
			if triplet(a, b, c) {
				return EulerU(a * b * c);
			}
		}
	}
	unreachable!()
}
fn main() { print_euler(solution()) }
