Option Infer On

Imports System
Imports System.Runtime.InteropServices

Namespace Examples
    Friend Class Program
        <STAThread>
        Shared Sub Main(ByVal args() As String)
            Dim application As SolidEdgeFramework.Application = Nothing
            Dim partDocument As SolidEdgePart.PartDocument = Nothing
            Dim models As SolidEdgePart.Models = Nothing
            Dim model As SolidEdgePart.Model = Nothing
            Dim body As SolidEdgeGeometry.Body = Nothing
            Dim edges As SolidEdgeGeometry.Edges = Nothing
            Dim edgeCollection As SolidEdgeGeometry.Edges = Nothing

            Try
                ' See "Handling 'Application is Busy' and 'Call was Rejected By Callee' errors" topic.
                OleMessageFilter.Register()

                ' Attempt to connect to a running instance of Solid Edge.
                application = DirectCast(Marshal.GetActiveObject("SolidEdge.Application"), SolidEdgeFramework.Application)
                partDocument = TryCast(application.ActiveDocument, SolidEdgePart.PartDocument)

                If partDocument IsNot Nothing Then
                    models = partDocument.Models
                    model = models.Item(1)
                    body = CType(model.Body, SolidEdgeGeometry.Body)

                    Dim EdgeType = SolidEdgeGeometry.FeatureTopologyQueryTypeConstants.igQueryAll
                    edges = CType(body.Edges(EdgeType), SolidEdgeGeometry.Edges)

                    Dim CollectionType = SolidEdgeGeometry.TopologyCollectionTypeConstants.seEdgeCollection
                    edgeCollection = CType(body.CreateCollection(CollectionType), SolidEdgeGeometry.Edges)

                    edgeCollection.Add(edges.Item(1))
                    edgeCollection.Add(edges.Item(3))
                End If
            Catch ex As System.Exception
                Console.WriteLine(ex)
            Finally
                OleMessageFilter.Unregister()
            End Try
        End Sub
    End Class
End Namespace
