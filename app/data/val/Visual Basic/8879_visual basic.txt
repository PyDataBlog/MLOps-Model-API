Public Class SituacaoCotacaoLicitacao
    Private situacaoValue As SituacaoEnum

    Public Property Situacao() As SituacaoEnum
        Get
            Return Me.situacaoValue
        End Get
        Set(ByVal value As SituacaoEnum)
            Me.situacaoValue = value
        End Set
    End Property

    Enum SituacaoEnum As Integer
        EM_ABERTO = 1
        APROVADO
        ABERTA_LICITAÇÃO
        PERDIDO
        INATIVO
    End Enum

    Public Shared Function GetIDSituacao(ByVal name As String) As SituacaoEnum
        For Each value In CType([Enum].GetValues(GetType(SituacaoEnum)), Integer())
            If name.Equals(CType(value, SituacaoEnum).ToString().Replace("_", " ")) Then
                Return CType(value, SituacaoEnum)
            End If
        Next
    End Function

End Class
