Imports System.Web.Routing
Imports System.Web.Mvc

Public Class AppRoutes
    Shared Sub MapRoutes(routes As RouteCollection)
        routes.MapRoute(
            "CustomersRoute",
            "Home",
            New With {.controller = "Home", .action = "Index"}
        )
        routes.MapRoute(
            "CustomersListRoute",
            "Home/List",
            New With {.controller = "Home", .action = "List"}
        )
        routes.MapRoute(
            "CustomersBrowseRoute",
            "Home/Browse/{customerId}",
            New With {.controller = "Home", .action = "Browse", .customerId = String.Empty}
        )
        routes.MapRoute(
            "CustomersEditRoute",
            "Home/Edit/{customerId}",
            New With {.controller = "Home", .action = "Edit", .customerId = String.Empty}
        )
        routes.MapRoute(
            "OrdersProcesssRoute",
            "Orders/Process/{customerId}/{contactName}",
            New With {.controller = "Orders", .action = "Process", .customerId = String.Empty, .contactName = String.Empty}
        )
        routes.MapRoute(
            "DefaultRoute",
            "{controller}/{action}",
            New With {.controller = "Home", .action = "Index"}
        )
    End Sub
End Class
