--TB1-042 Universe 9 Striker Oregano
local ds=require "expansions.utility_dbscg"
local scard,sid=ds.GetID()
function scard.initial_effect(c)
	ds.EnableBattleAttribute(c)
	ds.AddSetcode(c,SPECIAL_TRAIT_ALIEN,SPECIAL_TRAIT_UNIVERSE_9,SPECIAL_TRAIT_INCLUDES_UNIVERSE)
	ds.AddPlayProcedure(c,COLOR_BLUE,1,1)
	--to deck
	ds.AddSingleAutoPlay(c,0,true,scard.tdtg,scard.tdop)
end
scard.dragon_ball_super_card=true
scard.combo_cost=0
function scard.tdfilter(c)
	return c:IsSpecialTrait(SPECIAL_TRAIT_UNIVERSE_9) and c:IsAbleToDeck()
end
scard.tdtg=ds.CheckCardFunction(scard.tdfilter,LOCATION_HAND,0)
function scard.tdop(e,tp,eg,ep,ev,re,r,rp)
	Duel.Hint(HINT_SELECTMSG,tp,DS_HINTMSG_TODECK)
	local tc=Duel.SelectMatchingCard(tp,Card.IsSpecialTrait,tp,LOCATION_HAND,0,1,1,nil,SPECIAL_TRAIT_UNIVERSE_9):GetFirst()
	if not tc then return end
	if not tc:IsPublic() then Duel.ConfirmCards(1-tc:GetControler(),tc) end
	if Duel.SendtoDeck(tc,PLAYER_OWNER,DECK_SEQUENCE_BOTTOM,DS_REASON_SKILL)==0 then return end
	Duel.Draw(tp,2,DS_REASON_SKILL)
end
