--Panopter
local ka=require "expansions.utility_ktcg"
local scard,sid=ka.GetID()
function scard.initial_effect(c)
	ka.EnableCreatureAttribute(c)
	--blocker
	ka.EnableBlocker(c)
	--skirmisher
	ka.EnableSkirmisher(c)
	--draw
	ka.AddComeIntoPlayAbility(c,0,true,scard.drtg,scard.drop,nil,scard.drcon)
end
scard.kaijudo_card=true
function scard.cfilter(c,tp)
	return c:IsFaceup() and c:IsHasAbility(KA_ABILITY_BLOCKER) and c:IsControler(tp)
end
function scard.drcon(e,tp,eg,ep,ev,re,r,rp)
	return eg:IsExists(scard.cfilter,1,e:GetHandler(),tp)
end
scard.drtg=ka.DrawTarget(PLAYER_PLAYER)
function scard.drop(e,tp,eg,ep,ev,re,r,rp)
	if not e:GetHandler():IsRelateToAbility(e) or e:GetHandler():IsFacedown() then return end
	Duel.Draw(tp,1,KA_REASON_ABILITY)
end
